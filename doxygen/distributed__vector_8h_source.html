501 template <typename T>
502 inline
503 void DistributedVector<T>::init (const numeric_index_type n,
504  const numeric_index_type n_local,
505  const bool fast,
506  const ParallelType ptype)
507 {
508  // This function must be run on all processors at once
509  parallel_object_only();
510 
511  libmesh_assert_less_equal (n_local, n);
512 
513  if (ptype == AUTOMATIC)
514  {
515  if (n == n_local)
516  this->_type = SERIAL;
517  else
518  this->_type = PARALLEL;
519  }
520  else
521  this->_type = ptype;
522 
523  libmesh_assert ((this->_type==SERIAL && n==n_local) ||
524  this->_type==PARALLEL);
525 
526  // Clear the data structures if already initialized
527  if (this->initialized())
528  this->clear();
529 
530  // Initialize data structures
531  _values.resize(n_local);
532  _local_size = n_local;
533  _global_size = n;
534 
535  _first_local_index = 0;
536 
537 #ifdef LIBMESH_HAVE_MPI
538 
539  std::vector<numeric_index_type> local_sizes (this->n_processors(), 0);
540 
541  local_sizes[this->processor_id()] = n_local;
542 
543  this->comm().sum(local_sizes);
544 
545  // _first_local_index is the sum of _local_size
546  // for all processor ids less than ours
547  for (processor_id_type p=0; p!=this->processor_id(); p++)
548  _first_local_index += local_sizes[p];
549 
550 
551 # ifdef DEBUG
552  // Make sure all the local sizes sum up to the global
553  // size, otherwise there is big trouble!
554  numeric_index_type dbg_sum=0;
555 
556  for (processor_id_type p=0; p!=this->n_processors(); p++)
557  dbg_sum += local_sizes[p];
558 
559  libmesh_assert_equal_to (dbg_sum, n);
560 
561 # endif
562 
563 #else
564 
565  // No other options without MPI!
566  if (n != n_local)
567  libmesh_error_msg("ERROR: MPI is required for n != n_local!");
568 
569 #endif
570 
571  _last_local_index = _first_local_index + n_local;
572 
573  // Set the initialized flag
574  this->_is_initialized = true;
575 
576  // Zero the components unless directed otherwise
577  if (!fast)
578  this->zero();
579 }
