102 void ExplicitSystem::assemble_qoi_derivative (const QoISet& qoi_indices)

126 void ExplicitSystem::add_system_rhs ()
127 {
128  // Possible that we cleared the _vectors but
129  // forgot to NULL-out the rhs?
130  if (this->n_vectors() == 0) rhs = NULL;
131 
132 
133  // Only need to add the rhs if it isn't there
134  // already!
135  if (rhs == NULL)

138  libmesh_assert(rhs);
