705 template <typename T>
706 inline
707 PetscVector<T>::PetscVector (Vec v,
708  const Parallel::Communicator &comm_in)
709  : NumericVector<T>(comm_in, AUTOMATIC),
710  _array_is_present(false),
711  _local_form(NULL),
712  _values(NULL),
713  _global_to_local_map(),
714  _destroy_vec_on_exit(false)
715 {
716  this->_vec = v;
717  this->_is_closed = true;
718  this->_is_initialized = true;
719 
720  /* We need to ask PETSc about the (local to global) ghost value
721  mapping and create the inverse mapping out of it. */
722  PetscErrorCode ierr=0;
723  PetscInt petsc_local_size=0;
724  ierr = VecGetLocalSize(_vec, &petsc_local_size);
725  LIBMESH_CHKERRABORT(ierr);
726 
727  // Get the vector type from PETSc.
728  // As of Petsc 3.0.0, the VecType #define lost its const-ness, so we
729  // need to have it in the code
730 #if PETSC_VERSION_LESS_THAN(3,0,0) || !PETSC_VERSION_LESS_THAN(3,4,0)
731  // Pre-3.0 and petsc-dev (as of October 2012) use non-const versions
732  VecType ptype;
733 #else
734  const VecType ptype;
735 #endif
736  ierr = VecGetType(_vec, &ptype);
737  LIBMESH_CHKERRABORT(ierr);
738 
739  if((std::strcmp(ptype,VECSHARED) == 0) || (std::strcmp(ptype,VECMPI) == 0))
740  {
741 #if PETSC_RELEASE_LESS_THAN(3,1,1)
742  ISLocalToGlobalMapping mapping = _vec->mapping;
743 #else
744  ISLocalToGlobalMapping mapping;
745  ierr = VecGetLocalToGlobalMapping(_vec, &mapping);
746  LIBMESH_CHKERRABORT(ierr);
747 #endif
748 
749  // If is a sparsely stored vector, set up our new mapping
750  if (mapping)
751  {
752  const numeric_index_type my_local_size = static_cast<numeric_index_type>(petsc_local_size);
753  const numeric_index_type ghost_begin = static_cast<numeric_index_type>(petsc_local_size);
754 #if PETSC_RELEASE_LESS_THAN(3,4,0)
755  const numeric_index_type ghost_end = static_cast<numeric_index_type>(mapping->n);
756 #else
757  PetscInt n;
758  ierr = ISLocalToGlobalMappingGetSize(mapping, &n);
759  LIBMESH_CHKERRABORT(ierr);
760  const numeric_index_type ghost_end = static_cast<numeric_index_type>(n);
761 #endif
762 #if PETSC_RELEASE_LESS_THAN(3,1,1)
763  const PetscInt *indices = mapping->indices;
764 #else
765  const PetscInt *indices;
766  ierr = ISLocalToGlobalMappingGetIndices(mapping,&indices);
767  LIBMESH_CHKERRABORT(ierr);
768 #endif
769  for(numeric_index_type i=ghost_begin; i<ghost_end; i++)
770  _global_to_local_map[indices[i]] = i-my_local_size;
771  this->_type = GHOSTED;
772 #if !PETSC_RELEASE_LESS_THAN(3,1,1)
773  ierr = ISLocalToGlobalMappingRestoreIndices(mapping, &indices);
774  LIBMESH_CHKERRABORT(ierr);
775 #endif
776  }
777  else
778  this->_type = PARALLEL;
779  }
780  else
781  this->_type = SERIAL;
782 
783  this->close();
784 }

798 template <typename T>
799 inline
800 void PetscVector<T>::init (const numeric_index_type n,
801  const numeric_index_type n_local,
802  const bool fast,
803  const ParallelType ptype)
804 {
805  PetscErrorCode ierr=0;
806  PetscInt petsc_n=static_cast<PetscInt>(n);
807  PetscInt petsc_n_local=static_cast<PetscInt>(n_local);
808 
809 
810  // Clear initialized vectors
811  if (this->initialized())
812  this->clear();
813 
814  if (ptype == AUTOMATIC)
815  {
816  if (n == n_local)
817  this->_type = SERIAL;
818  else
819  this->_type = PARALLEL;
820  }
821  else
822  this->_type = ptype;
823 
824  libmesh_assert ((this->_type==SERIAL && n==n_local) ||
825  this->_type==PARALLEL);
826 
827  // create a sequential vector if on only 1 processor
828  if (this->_type == SERIAL)
829  {
830  ierr = VecCreateSeq (PETSC_COMM_SELF, petsc_n, &_vec);
831  CHKERRABORT(PETSC_COMM_SELF,ierr);
832 
833  ierr = VecSetFromOptions (_vec);
834  CHKERRABORT(PETSC_COMM_SELF,ierr);
835  }
836  // otherwise create an MPI-enabled vector
837  else if (this->_type == PARALLEL)
838  {
839 #ifdef LIBMESH_HAVE_MPI
840  libmesh_assert_less_equal (n_local, n);
841  ierr = VecCreateMPI (this->comm().get(), petsc_n_local, petsc_n,
842  &_vec);
843  LIBMESH_CHKERRABORT(ierr);
844 #else
845  libmesh_assert_equal_to (n_local, n);
846  ierr = VecCreateSeq (PETSC_COMM_SELF, petsc_n, &_vec);
847  CHKERRABORT(PETSC_COMM_SELF,ierr);
848 #endif
849 
850  ierr = VecSetFromOptions (_vec);
851  LIBMESH_CHKERRABORT(ierr);
852  }
853  else
854  libmesh_error_msg("Unsupported type " << this->_type);
855 
856  this->_is_initialized = true;
857  this->_is_closed = true;
858 
859 
860  if (fast == false)
861  this->zero ();
862 }

877 template <typename T>
878 inline
879 void PetscVector<T>::init (const numeric_index_type n,
880  const numeric_index_type n_local,
881  const std::vector<numeric_index_type>& ghost,
882  const bool fast,
883  const ParallelType libmesh_dbg_var(ptype))
884 {
885  PetscErrorCode ierr=0;
886  PetscInt petsc_n=static_cast<PetscInt>(n);
887  PetscInt petsc_n_local=static_cast<PetscInt>(n_local);
888  PetscInt petsc_n_ghost=static_cast<PetscInt>(ghost.size());
889 
890  // If the mesh is not disjoint, every processor will either have
891  // all the dofs, none of the dofs, or some non-zero dofs at the
892  // boundary between processors.
893  //
894  // However we can't assert this, because someone might want to
895  // construct a GHOSTED vector which doesn't include neighbor element
896  // dofs. Boyce tried to do so in user code, and we're going to want
897  // to do so in System::project_vector().
898  //
899  // libmesh_assert(n_local == 0 || n_local == n || !ghost.empty());
900 
901  libmesh_assert(sizeof(PetscInt) == sizeof(numeric_index_type));
902 
903  PetscInt* petsc_ghost = ghost.empty() ? PETSC_NULL :
904  const_cast<PetscInt*>(reinterpret_cast<const PetscInt*>(&ghost[0]));
905 
906  // Clear initialized vectors
907  if (this->initialized())
908  this->clear();
909 
910  libmesh_assert(ptype == AUTOMATIC || ptype == GHOSTED);
911  this->_type = GHOSTED;
912 
913  /* Make the global-to-local ghost cell map. */
914  for(numeric_index_type i=0; i<ghost.size(); i++)
915  {
916  _global_to_local_map[ghost[i]] = i;
917  }
918 
919  /* Create vector. */
920  ierr = VecCreateGhost (this->comm().get(), petsc_n_local, petsc_n,
921  petsc_n_ghost, petsc_ghost,
922  &_vec);
923  LIBMESH_CHKERRABORT(ierr);
924 
925  ierr = VecSetFromOptions (_vec);
926  LIBMESH_CHKERRABORT(ierr);
927 
928  this->_is_initialized = true;
929  this->_is_closed = true;
930 
931  if (fast == false)
932  this->zero ();
933 }

937 template <typename T>
938 inline
939 void PetscVector<T>::init (const NumericVector<T>& other,
940  const bool fast)
941 {
942  // Clear initialized vectors
943  if (this->initialized())
944  this->clear();
945 
946  const PetscVector<T>& v = cast_ref<const PetscVector<T>&>(other);
947 
948  // Other vector should restore array.
949  if(v.initialized())
950  {
951  v._restore_array();
952  }
953 
954  this->_global_to_local_map = v._global_to_local_map;
955 
956  // Even if we're initializeing sizes based on an uninitialized or
957  // unclosed vector, *this* vector is being initialized now and is
958  // initially closed.
959  this->_is_closed = true; // v._is_closed;
960  this->_is_initialized = true; // v._is_initialized;
961 
962  this->_type = v._type;
963 
964  if (v.size() != 0)
965  {
966  PetscErrorCode ierr = 0;
967 
968  ierr = VecDuplicate (v._vec, &this->_vec);
969  LIBMESH_CHKERRABORT(ierr);
970  }
971 
972  if (fast == false)
973  this->zero ();
974 }

996  LIBMESH_CHKERRABORT(ierr);
997  }
998

1026 template <typename T>
1027 inline
1028 void PetscVector<T>::zero ()
1029 {
1030  libmesh_assert(this->closed());
1031 
1032  this->_restore_array();
1033 
1034  PetscErrorCode ierr=0;
1035 
1036  PetscScalar z=0.;
1037 
1038  if(this->type() != GHOSTED)
1039  {
1040 #if PETSC_VERSION_LESS_THAN(2,3,0)
1041  // 2.2.x & earlier style
1042  ierr = VecSet (&z, _vec);
1043  LIBMESH_CHKERRABORT(ierr);
1044 #else
1045  // 2.3.x & newer
1046  ierr = VecSet (_vec, z);
1047  LIBMESH_CHKERRABORT(ierr);
1048 #endif
1049  }
1050  else
1051  {
1052  /* Vectors that include ghost values require a special
1053  handling. */
1054  Vec loc_vec;
1055  ierr = VecGhostGetLocalForm (_vec,&loc_vec);
1056  LIBMESH_CHKERRABORT(ierr);
1057 #if PETSC_VERSION_LESS_THAN(2,3,0)
1058  // 2.2.x & earlier style
1059  ierr = VecSet (&z, loc_vec);
1060  LIBMESH_CHKERRABORT(ierr);
1061 #else
1062  // 2.3.x & newer
1063  ierr = VecSet (loc_vec, z);
1064  LIBMESH_CHKERRABORT(ierr);
1065 #endif
