661 template <typename T>
662 inline
663 PetscVector<T>::PetscVector (Vec v,
664  const Parallel::Communicator &comm_in) :
665  NumericVector<T>(comm_in, AUTOMATIC),
666  _array_is_present(false),
667  _local_form(NULL),
668  _values(NULL),
669  _global_to_local_map(),
670  _destroy_vec_on_exit(false)
671 {
672  this->_vec = v;
673  this->_is_closed = true;
674  this->_is_initialized = true;
675 
676  /* We need to ask PETSc about the (local to global) ghost value
677  mapping and create the inverse mapping out of it. */
678  PetscErrorCode ierr=0;
679  PetscInt petsc_local_size=0;
680  ierr = VecGetLocalSize(_vec, &petsc_local_size);
681  LIBMESH_CHKERR(ierr);
682 
683  // Get the vector type from PETSc.
684  // As of Petsc 3.0.0, the VecType #define lost its const-ness, so we
685  // need to have it in the code
686 #if PETSC_VERSION_LESS_THAN(3,0,0) || !PETSC_VERSION_LESS_THAN(3,4,0)
687  // Pre-3.0 and petsc-dev (as of October 2012) use non-const versions
688  VecType ptype;
689 #else
690  const VecType ptype;
691 #endif
692  ierr = VecGetType(_vec, &ptype);
693  LIBMESH_CHKERR(ierr);
694 
695  if((std::strcmp(ptype,VECSHARED) == 0) || (std::strcmp(ptype,VECMPI) == 0))
696  {
697 #if PETSC_RELEASE_LESS_THAN(3,1,1)
698  ISLocalToGlobalMapping mapping = _vec->mapping;
699 #else
700  ISLocalToGlobalMapping mapping;
701  ierr = VecGetLocalToGlobalMapping(_vec, &mapping);
702  LIBMESH_CHKERR(ierr);
703 #endif
704 
705  // If is a sparsely stored vector, set up our new mapping
706  if (mapping)
707  {
708  const numeric_index_type my_local_size = static_cast<numeric_index_type>(petsc_local_size);
709  const numeric_index_type ghost_begin = static_cast<numeric_index_type>(petsc_local_size);
710 #if PETSC_RELEASE_LESS_THAN(3,4,0)
711  const numeric_index_type ghost_end = static_cast<numeric_index_type>(mapping->n);
712 #else
713  PetscInt n;
714  ierr = ISLocalToGlobalMappingGetSize(mapping, &n);
715  LIBMESH_CHKERR(ierr);
716  const numeric_index_type ghost_end = static_cast<numeric_index_type>(n);
717 #endif
718 #if PETSC_RELEASE_LESS_THAN(3,1,1)
719  const PetscInt *indices = mapping->indices;
720 #else
721  const PetscInt *indices;
722  ierr = ISLocalToGlobalMappingGetIndices(mapping,&indices);
723  LIBMESH_CHKERR(ierr);
724 #endif
725  for(numeric_index_type i=ghost_begin; i<ghost_end; i++)
726  _global_to_local_map[indices[i]] = i-my_local_size;
727  this->_type = GHOSTED;
728 #if !PETSC_RELEASE_LESS_THAN(3,1,1)
729  ierr = ISLocalToGlobalMappingRestoreIndices(mapping, &indices);
730  LIBMESH_CHKERR(ierr);
731 #endif
732  }
733  else
734  this->_type = PARALLEL;
735  }
736  else
737  this->_type = SERIAL;
738 }

752 template <typename T>
753 inline
754 void PetscVector<T>::init (const numeric_index_type n,
755  const numeric_index_type n_local,
756  const bool fast,
757  const ParallelType ptype)
758 {
759  parallel_object_only();
760 
761  PetscErrorCode ierr=0;
762  PetscInt petsc_n=static_cast<PetscInt>(n);
763  PetscInt petsc_n_local=static_cast<PetscInt>(n_local);
764 
765 
766  // Clear initialized vectors
767  if (this->initialized())
768  this->clear();
769 
770  if (ptype == AUTOMATIC)
771  {
772  if (n == n_local)
773  this->_type = SERIAL;
774  else
775  this->_type = PARALLEL;
776  }
777  else
778  this->_type = ptype;
779 
780  libmesh_assert ((this->_type==SERIAL && n==n_local) ||
781  this->_type==PARALLEL);
782 
783  // create a sequential vector if on only 1 processor
784  if (this->_type == SERIAL)
785  {
786  ierr = VecCreateSeq (PETSC_COMM_SELF, petsc_n, &_vec);
787  CHKERRABORT(PETSC_COMM_SELF,ierr);
788 
789  ierr = VecSetFromOptions (_vec);
790  CHKERRABORT(PETSC_COMM_SELF,ierr);
791  }
792  // otherwise create an MPI-enabled vector
793  else if (this->_type == PARALLEL)
794  {
795 #ifdef LIBMESH_HAVE_MPI
796  libmesh_assert_less_equal (n_local, n);
797  ierr = VecCreateMPI (this->comm().get(), petsc_n_local, petsc_n,
798  &_vec);
799  LIBMESH_CHKERR(ierr);
800 #else
801  libmesh_assert_equal_to (n_local, n);
802  ierr = VecCreateSeq (PETSC_COMM_SELF, petsc_n, &_vec);
803  CHKERRABORT(PETSC_COMM_SELF,ierr);
804 #endif
805 
806  ierr = VecSetFromOptions (_vec);
807  LIBMESH_CHKERR(ierr);
808  }
809  else
810  libmesh_error_msg("Unsupported type " << this->_type);
811 
812  this->_is_initialized = true;
813  this->_is_closed = true;
814 
815 
816  if (fast == false)
817  this->zero ();
818 }

833 template <typename T>
834 inline
835 void PetscVector<T>::init (const numeric_index_type n,
836  const numeric_index_type n_local,
837  const std::vector<numeric_index_type>& ghost,
838  const bool fast,
839  const ParallelType libmesh_dbg_var(ptype))
840 {
841  parallel_object_only();
842 
843  PetscErrorCode ierr=0;
844  PetscInt petsc_n=static_cast<PetscInt>(n);
845  PetscInt petsc_n_local=static_cast<PetscInt>(n_local);
846  PetscInt petsc_n_ghost=static_cast<PetscInt>(ghost.size());
847 
848  // If the mesh is not disjoint, every processor will either have
849  // all the dofs, none of the dofs, or some non-zero dofs at the
850  // boundary between processors.
851  //
852  // However we can't assert this, because someone might want to
853  // construct a GHOSTED vector which doesn't include neighbor element
854  // dofs. Boyce tried to do so in user code, and we're going to want
855  // to do so in System::project_vector().
856  //
857  // libmesh_assert(n_local == 0 || n_local == n || !ghost.empty());
858 
859  libmesh_assert(sizeof(PetscInt) == sizeof(numeric_index_type));
860 
861  PetscInt* petsc_ghost = ghost.empty() ? PETSC_NULL :
862  const_cast<PetscInt*>(reinterpret_cast<const PetscInt*>(&ghost[0]));
863 
864  // Clear initialized vectors
865  if (this->initialized())
866  this->clear();
867 
868  libmesh_assert(ptype == AUTOMATIC || ptype == GHOSTED);
869  this->_type = GHOSTED;
870 
871  /* Make the global-to-local ghost cell map. */
872  for(numeric_index_type i=0; i<ghost.size(); i++)
873  {
874  _global_to_local_map[ghost[i]] = i;
875  }
876 
877  /* Create vector. */
878  ierr = VecCreateGhost (this->comm().get(), petsc_n_local, petsc_n,
879  petsc_n_ghost, petsc_ghost,
880  &_vec);
881  LIBMESH_CHKERR(ierr);
882 
883  ierr = VecSetFromOptions (_vec);
884  LIBMESH_CHKERR(ierr);
885 
886  this->_is_initialized = true;
887  this->_is_closed = true;
888 
889  if (fast == false)
890  this->zero ();
891 }

895 template <typename T>
896 inline
897 void PetscVector<T>::init (const NumericVector<T>& other,
898  const bool fast)
899 {
900  parallel_object_only();
901 
902  // Clear initialized vectors
903  if (this->initialized())
904  this->clear();
905 
906  const PetscVector<T>& v = cast_ref<const PetscVector<T>&>(other);
907 
908  // Other vector should restore array.
909  if(v.initialized())
910  {
911  v._restore_array();
912  }
913 
914  this->_global_to_local_map = v._global_to_local_map;
915 
916  // Even if we're initializeing sizes based on an uninitialized or
917  // unclosed vector, *this* vector is being initialized now and is
918  // initially closed.
919  this->_is_closed = true; // v._is_closed;
920  this->_is_initialized = true; // v._is_initialized;
921 
922  this->_type = v._type;
923 
924  // We want to have a valid Vec, even if it's initially of size zero
925  PetscErrorCode ierr = VecDuplicate (v._vec, &this->_vec);
926  LIBMESH_CHKERR(ierr);
927 
928  if (fast == false)
929  this->zero ();
930 }

988 void PetscVector<T>::zero ()
989 {

992  libmesh_assert(this->closed());

1012  else

1023 #else
1024  // 2.3.x & newer
1025  ierr = VecSet (loc_vec, z);
1026  LIBMESH_CHKERR(ierr);
1027 #endif
1028  ierr = VecGhostRestoreLocalForm (_vec,&loc_vec);
1029  LIBMESH_CHKERR(ierr);
1030  }
