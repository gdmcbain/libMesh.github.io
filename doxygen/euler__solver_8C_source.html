89 bool EulerSolver::_general_residual (bool request_jacobian,
90  DiffContext & context,
91  ResFuncType mass,
92  ResFuncType time_deriv,
93  ResFuncType constraint,
94  ReinitFuncType reinit_func)
95 {
96  unsigned int n_dofs = context.get_elem_solution().size();
97 
98  // We might need to save the old jacobian in case one of our physics
99  // terms later is unable to update it analytically.
100  DenseMatrix<Number> old_elem_jacobian(n_dofs, n_dofs);
101  if (request_jacobian)
102  old_elem_jacobian.swap(context.get_elem_jacobian());
103 
104  // Local nonlinear solution at old timestep
105  DenseVector<Number> old_elem_solution(n_dofs);
106  for (unsigned int i=0; i != n_dofs; ++i)
107  old_elem_solution(i) =
108  old_nonlinear_solution(context.get_dof_indices()[i]);
109 
110  // Local time derivative of solution
111  context.get_elem_solution_rate() = context.get_elem_solution();
112  context.get_elem_solution_rate() -= old_elem_solution;
113  context.elem_solution_rate_derivative = 1 / _system.deltat;
114  context.get_elem_solution_rate() *=
115  context.elem_solution_rate_derivative;
116 
117  // Local nonlinear solution at time t_theta
118  DenseVector<Number> theta_solution(context.get_elem_solution());
119  theta_solution *= theta;
120  theta_solution.add(1. - theta, old_elem_solution);
121 
122  context.elem_solution_derivative = theta;
123  context.fixed_solution_derivative = theta;
124 
125  // If a fixed solution is requested, we'll use theta_solution
126  if (_system.use_fixed_solution)
127  context.get_elem_fixed_solution() = theta_solution;
128 
129  // Move theta_->elem_, elem_->theta_
130  context.get_elem_solution().swap(theta_solution);
131 
132  // Move the mesh into place first if necessary, set t = t_{\theta}
133  (context.*reinit_func)(theta);
134 
135  // Get the time derivative at t_theta
136  bool jacobian_computed =
137  (_system.*time_deriv)(request_jacobian, context);
138 
139  jacobian_computed = (_system.*mass)(jacobian_computed, context) &&
140  jacobian_computed;
141 
142  // Restore the elem position if necessary, set t = t_{n+1}
143  (context.*reinit_func)(1);
144 
145  // Move elem_->elem_, theta_->theta_
146  context.get_elem_solution().swap(theta_solution);
147  context.elem_solution_derivative = 1;
148 
149  // Add the constraint term
150  jacobian_computed = (_system.*constraint)(jacobian_computed, context) &&
151  jacobian_computed;
152 
153  // Add back (or restore) the old jacobian
154  if (request_jacobian)
155  {
156  if (jacobian_computed)
157  context.get_elem_jacobian() += old_elem_jacobian;
158  else
159  context.get_elem_jacobian().swap(old_elem_jacobian);
160  }
161 
162  return jacobian_computed;
163 }
