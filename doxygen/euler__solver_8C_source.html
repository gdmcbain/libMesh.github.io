89 bool EulerSolver::_general_residual (bool request_jacobian,
90  DiffContext &context,
91  ResFuncType mass,
92  ResFuncType time_deriv,
93  ResFuncType constraint,
94  ReinitFuncType reinit_func)
95 {
96  unsigned int n_dofs = context.get_elem_solution().size();
97 
98  // Local nonlinear solution at old timestep
99  DenseVector<Number> old_elem_solution(n_dofs);
100  for (unsigned int i=0; i != n_dofs; ++i)
101  old_elem_solution(i) =
102  old_nonlinear_solution(context.get_dof_indices()[i]);
103 
104  // Local nonlinear solution at time t_theta
105  DenseVector<Number> theta_solution(context.get_elem_solution());
106  theta_solution *= theta;
107  theta_solution.add(1. - theta, old_elem_solution);
108 
109  // Technically the elem_solution_derivative is either theta
110  // or -1.0 in this implementation, but we scale the former part
111  // ourselves
112  context.elem_solution_derivative = 1.0;
113 
114  // Technically the fixed_solution_derivative is always theta,
115  // but we're scaling a whole jacobian by theta after these first
116  // evaluations
117  context.fixed_solution_derivative = 1.0;
118 
119  // If a fixed solution is requested, we'll use theta_solution
120  if (_system.use_fixed_solution)
121  context.get_elem_fixed_solution() = theta_solution;
122 
123  // Move theta_->elem_, elem_->theta_
124  context.get_elem_solution().swap(theta_solution);
125 
126  // Move the mesh into place first if necessary
127  (context.*reinit_func)(theta);
128 
129  // We're going to compute just the change in elem_residual
130  // (and possibly elem_jacobian), then add back the old values
131  DenseVector<Number> old_elem_residual(context.get_elem_residual());
132  DenseMatrix<Number> old_elem_jacobian;
133  if (request_jacobian)
134  {
135  old_elem_jacobian = context.get_elem_jacobian();
136  context.get_elem_jacobian().zero();
137  }
138  context.get_elem_residual().zero();
139 
140  // Get the time derivative at t_theta
141  bool jacobian_computed =
142  (_system.*time_deriv)(request_jacobian, context);
143 
144  // Scale the time-dependent residual and jacobian correctly
145  context.get_elem_residual() *= _system.deltat;
146  if (jacobian_computed)
147  context.get_elem_jacobian() *= (theta * _system.deltat);
148 
149  // The fixed_solution_derivative is always theta,
150  // and now we're done scaling jacobians
151  context.fixed_solution_derivative = theta;
152 
153  // We evaluate mass_residual with the change in solution
154  // to get the mass matrix, reusing old_elem_solution to hold that
155  // delta_solution. We're solving dt*F(u) - du = 0, so our
156  // delta_solution is old_solution - new_solution.
157  // We're still keeping elem_solution in theta_solution for now
158  old_elem_solution -= theta_solution;
159 
160  // Move old_->elem_, theta_->old_
161  context.get_elem_solution().swap(old_elem_solution);
162 
163  // We do a trick here to avoid using a non-1
164  // elem_solution_derivative:
165  context.get_elem_jacobian() *= -1.0;
166  context.fixed_solution_derivative *= -1.0;
167  jacobian_computed = (_system.*mass)(jacobian_computed, context) &&
168  jacobian_computed;
169  context.get_elem_jacobian() *= -1.0;
170  context.fixed_solution_derivative *= -1.0;
171 
172  // Move elem_->elem_, old_->theta_
173  context.get_elem_solution().swap(theta_solution);
174 
175  // Restore the elem position if necessary
176  (context.*reinit_func)(1.);
177 
178  // Add the constraint term
179  jacobian_computed = (_system.*constraint)(jacobian_computed, context) &&
180  jacobian_computed;
181 
182  // Add back the old residual and jacobian
183  context.get_elem_residual() += old_elem_residual;
184  if (request_jacobian)
185  {
186  if (jacobian_computed)
187  context.get_elem_jacobian() += old_elem_jacobian;
188  else
189  context.get_elem_jacobian().swap(old_elem_jacobian);
190  }
191 
192  return jacobian_computed;
193 }
