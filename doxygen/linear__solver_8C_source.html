41 UniquePtr<LinearSolver<T> >
42 LinearSolver<T>::build(const libMesh::Parallel::Communicator &comm,
43  const SolverPackage solver_package)
44 {
45  // Build the appropriate solver
46  switch (solver_package)
47  {
48 
49 
50 #ifdef LIBMESH_HAVE_LASPACK
51  case LASPACK_SOLVERS:
52  return UniquePtr<LinearSolver<T> >(new LaspackLinearSolver<T>(comm));
53 #endif
54 
55 
56 #ifdef LIBMESH_HAVE_PETSC
57  case PETSC_SOLVERS:
58  return UniquePtr<LinearSolver<T> >(new PetscLinearSolver<T>(comm));
59 #endif
60 
61 
62 #if defined(LIBMESH_HAVE_TRILINOS) && (LIBMESH_TRILINOS_HAVE_AZTECOO)
63  case TRILINOS_SOLVERS:
64  return UniquePtr<LinearSolver<T> >(new AztecLinearSolver<T>(comm));
65 #endif
66 
67 
68 #ifdef LIBMESH_HAVE_EIGEN
69  case EIGEN_SOLVERS:
70  return UniquePtr<LinearSolver<T> >(new EigenSparseLinearSolver<T>(comm));
71 #endif
72 
73  default:
74  libmesh_error_msg("ERROR: Unrecognized solver package: " << solver_package);
75  }
76 
77  return UniquePtr<LinearSolver<T> >();

130 template <typename T>
131 std::pair<unsigned int, Real> LinearSolver<T>::adjoint_solve (SparseMatrix<T> & mat,
132  NumericVector<T>& sol,
133  NumericVector<T>& rhs,
134  const double tol,
135  const unsigned int n_iter)
136 {
137  // Log how long the linear solve takes.
138  START_LOG("adjoint_solve()", "LinearSolver");
139 
140  // Take the discrete adjoint
141  mat.close();
142  mat.get_transpose(mat);
143 
144  // Call the solve function for the relevant linear algebra library and
145  // solve the transpose matrix
146  const std::pair<unsigned int, Real> totalrval = this->solve (mat, sol, rhs, tol, n_iter);
147 
148  // Now transpose back and restore the original matrix
149  // by taking the discrete adjoint
150  mat.get_transpose(mat);
151 
152  // Stop logging the nonlinear solve
153  STOP_LOG("adjoint_solve()", "LinearSolver");
154 
155  return totalrval;
156 
157 }
