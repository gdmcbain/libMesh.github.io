88 {
89  libmesh_assert_less (i, this->n_sides());

91  // To be returned wrapped in an UniquePtr
92  Elem * face = libmesh_nullptr;
93 
94  // Set up the type of element
95  switch (i)
96  {
97  case 0: // triangular face 1
98  case 1: // triangular face 2
99  case 2: // triangular face 3
100  case 3: // triangular face 4

105  case 4: // the quad face at z=0
106  {

109  }
110  default:
111  libmesh_error_msg("Invalid side i = " << i);
112  }
113 
114  // Set the nodes
115  for (unsigned n=0; n<face->n_nodes(); ++n)
116  face->set_node(n) = this->get_node(Pyramid5::side_nodes_map[i][n]);
117 
118  return UniquePtr<Elem>(face);
