97 UniquePtr<Elem> InfHex::side_ptr (const unsigned int i)
98 {
99  libmesh_assert_less (i, this->n_sides());
100 
101  // To be returned wrapped in a UniquePtr
102  Elem * face = libmesh_nullptr;
103 
104  // Think of a unit cube: (-1,1) x (-1,1) x (-1,1),
105  // with (in general) the normals pointing outwards
106  switch (i)
107  {
108  // the face at z = -1
109  // the base, where the infinite element couples to conventional
110  // elements
111  case 0:
112  {
113  // Oops, here we are, claiming the normal of the face
114  // elements point outwards -- and this is the exception:
115  // For the side built from the base face,
116  // the normal is pointing _into_ the element!
117  // Why is that? - In agreement with build_side_ptr(),
118  // which in turn _has_ to build the face in this
119  // way as to enable the cool way \p InfFE re-uses \p FE.
120  face = new Quad4;
121  break;
122  }
123 
124  // These faces connect to other infinite elements.
125  case 1: // the face at y = -1
126  case 2: // the face at x = 1
127  case 3: // the face at y = 1
128  case 4: // the face at x = -1
129  {
130  face = new InfQuad4;
131  break;
132  }
133 
134  default:
135  libmesh_error_msg("Invalid side i = " << i);
136  }
137 
138  // Set the nodes
139  for (unsigned n=0; n<face->n_nodes(); ++n)
140  face->set_node(n) = this->node_ptr(InfHex8::side_nodes_map[i][n]);
141 
142  return UniquePtr<Elem>(face);
143 }

171 Real InfHex::quality (const ElemQuality q) const
172 {
173  switch (q)
174  {
175 
185  case DIAGONAL:
186  {
187  // Diagonal between node 0 and node 2
188  const Real d02 = this->length(0,2);
189 
190  // Diagonal between node 1 and node 3
191  const Real d13 = this->length(1,3);
192 
193  // Find the biggest and smallest diagonals
194  const Real min = std::min(d02, d13);
195  const Real max = std::max(d02, d13);
196 
197  libmesh_assert_not_equal_to (max, 0.0);
198 
199  return min / max;
200 
201  break;
202  }
203 
211  case TAPER:
212  {
213 
217  const Real d01 = this->length(0,1);
218  const Real d12 = this->length(1,2);
219  const Real d23 = this->length(2,3);
220  const Real d03 = this->length(0,3);
221 
222  std::vector<Real> edge_ratios(2);
223 
224  // Bottom
225  edge_ratios[8] = std::min(d01, d23) / std::max(d01, d23);
226  edge_ratios[9] = std::min(d03, d12) / std::max(d03, d12);
227 
228  return *(std::min_element(edge_ratios.begin(), edge_ratios.end())) ;
229 
230  break;
231  }
232 
233 
241  case STRETCH:
242  {
246  const Real sqrt3 = 1.73205080756888;
247 
251  const Real d02 = this->length(0,2);
252  const Real d13 = this->length(1,3);
253  const Real max_diag = std::max(d02, d13);
254 
255  libmesh_assert_not_equal_to ( max_diag, 0.0 );
256 
260  std::vector<Real> edges(4);
261  edges[0] = this->length(0,1);
262  edges[1] = this->length(1,2);
263  edges[2] = this->length(2,3);
264  edges[3] = this->length(0,3);
265 
266  const Real min_edge = *(std::min_element(edges.begin(), edges.end()));
267  return sqrt3 * min_edge / max_diag ;
268  }
269 
270 
275  default:
276  return Elem::quality(q);
277  }
278 
279  libmesh_error_msg("We'll never get here!");
280  return 0.;
281 }
