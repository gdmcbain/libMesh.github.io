98 AutoPtr<Elem> InfHex::side (const unsigned int i) const
99 {
100  libmesh_assert_less (i, this->n_sides());
101 
102  /*
103  *Think of a unit cube: (-1,1) x (-1,1)x (-1,1),
104  * with (in general) the normals pointing outwards
105  */
106  switch (i)
107  {
108  case 0: // the face at z = -1
109  // the base, where the infinite element couples to conventional
110  // elements
111  {
112  Elem* face = new Quad4;
113  AutoPtr<Elem> ap_face(face);
114 
115  /*
116  * Oops, here we are, claiming the normal of the face
117  * elements point outwards -- and this is the exception:
118  * For the side built from the base face,
119  * the normal is pointing _into_ the element!
120  * Why is that? - In agreement with build_side(),
121  * which in turn _has_ to build the face in this
122  * way as to enable the cool way \p InfFE re-uses \p FE.
123  */
124  face->set_node(0) = this->get_node(0);
125  face->set_node(1) = this->get_node(1);
126  face->set_node(2) = this->get_node(2);
127  face->set_node(3) = this->get_node(3);
128 
129  return ap_face;
130  }
131 
132  case 1: // the face at y = -1
133  // this face connects to another infinite element
134  {
135  Elem* face = new InfQuad4;
136  AutoPtr<Elem> ap_face(face);
137 
138  face->set_node(0) = this->get_node(0);
139  face->set_node(1) = this->get_node(1);
140  face->set_node(2) = this->get_node(4);
141  face->set_node(3) = this->get_node(5);
142 
143  return ap_face;
144  }
145 
146  case 2: // the face at x = 1
147  // this face connects to another infinite element
148  {
149  Elem* face = new InfQuad4;
150  AutoPtr<Elem> ap_face(face);
151  //AutoPtr<Elem> face(new InfQuad4);
152 
153  face->set_node(0) = this->get_node(1);
154  face->set_node(1) = this->get_node(2);
155  face->set_node(2) = this->get_node(5);
156  face->set_node(3) = this->get_node(6);
157 
158  return ap_face;
159  }
160 
161  case 3: // the face at y = 1
162  // this face connects to another infinite element
163  {
164  Elem* face = new InfQuad4;
165  AutoPtr<Elem> ap_face(face);
166  //AutoPtr<Elem> face(new InfQuad4);
167 
168  face->set_node(0) = this->get_node(2);
169  face->set_node(1) = this->get_node(3);
170  face->set_node(2) = this->get_node(6);
171  face->set_node(3) = this->get_node(7);
172 
173  return ap_face;
174  }
175 
176  case 4: // the face at x = -1
177  // this face connects to another infinite element
178  {
179  Elem* face = new InfQuad4;
180  AutoPtr<Elem> ap_face(face);
181  //AutoPtr<Elem> face(new InfQuad4);
182

185  face->set_node(2) = this->get_node(7);

187 
188  return ap_face;
189  }
190 
191  default:
192  libmesh_error_msg("Invalid side i = " << i);
193  }
194 
195  libmesh_error_msg("We'll never get here!");
196  AutoPtr<Elem> ap(NULL);
197  return ap;
198 }

226 Real InfHex::quality (const ElemQuality q) const
227 {
228  switch (q)
229  {
230 
240  case DIAGONAL:
241  {
242  // Diagonal between node 0 and node 2
243  const Real d02 = this->length(0,2);
244 
245  // Diagonal between node 1 and node 3
246  const Real d13 = this->length(1,3);
247 
248  // Find the biggest and smallest diagonals
249  const Real min = std::min(d02, d13);
250  const Real max = std::max(d02, d13);
251 
252  libmesh_assert_not_equal_to (max, 0.0);
253 
254  return min / max;
255 
256  break;
257  }
258 
266  case TAPER:
267  {
268 
272  const Real d01 = this->length(0,1);
273  const Real d12 = this->length(1,2);
274  const Real d23 = this->length(2,3);
275  const Real d03 = this->length(0,3);
276 
277  std::vector<Real> edge_ratios(2);
278 
279  // Bottom
280  edge_ratios[8] = std::min(d01, d23) / std::max(d01, d23);
281  edge_ratios[9] = std::min(d03, d12) / std::max(d03, d12);
282 
283  return *(std::min_element(edge_ratios.begin(), edge_ratios.end())) ;
284 
285  break;
286  }
287 
288 
296  case STRETCH:
297  {
301  const Real sqrt3 = 1.73205080756888;
302 
306  const Real d02 = this->length(0,2);
307  const Real d13 = this->length(1,3);
308  const Real max_diag = std::max(d02, d13);
309 
310  libmesh_assert_not_equal_to ( max_diag, 0.0 );
311 
315  std::vector<Real> edges(4);
316  edges[0] = this->length(0,1);
317  edges[1] = this->length(1,2);
318  edges[2] = this->length(2,3);
319  edges[3] = this->length(0,3);
320 
321  const Real min_edge = *(std::min_element(edges.begin(), edges.end()));
322  return sqrt3 * min_edge / max_diag ;
323  }
324 
325 
330  default:
331  return Elem::quality(q);
332  }
333 
334  libmesh_error_msg("We'll never get here!");
335  return 0.;
336 }
