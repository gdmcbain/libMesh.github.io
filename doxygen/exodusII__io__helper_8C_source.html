324 void ExodusII_IO_Helper::open(const char * filename, bool read_only)
325 {
326  // Version of Exodus you are using
327  float ex_version = 0.;
328 
329  // Word size in bytes of the floating point variables used in the
330  // application program. Exodus only supports 4-byte and 8-byte
331  // floats.
332  int comp_ws = std::min(sizeof(Real), std::size_t(8));
333 
334  // Word size in bytes of the floating point data as they are stored
335  // in the ExodusII file. "If this argument is 0, the word size of the
336  // floating point data already stored in the file is returned"
337  int io_ws = 0;
338 
339  ex_id = exII::ex_open(filename,
340  read_only ? EX_READ : EX_WRITE,
341  &comp_ws,
342  &io_ws,
343  &ex_version);
344 
345  std::string err_msg = std::string("Error opening ExodusII mesh file: ") + std::string(filename);
346  EX_CHECK_ERR(ex_id, err_msg);
347  if (verbose) libMesh::out << "File opened successfully." << std::endl;
348 
349  if (read_only)
350  opened_for_reading = true;
351  else
352  opened_for_writing = true;
353 
354  current_filename = std::string(filename);
355 }

389 void ExodusII_IO_Helper::read_qa_records()
390 {
391  // The QA records are four MAX_STR_LENGTH-byte character strings.
392  int num_qa_rec =
393  this->inquire(exII::EX_INQ_QA, "Error retrieving number of QA records");
394 
395  if (verbose)
396  libMesh::out << "Found "
397  << num_qa_rec
398  << " QA record(s) in the Exodus file."
399  << std::endl;
400 
401  if (num_qa_rec > 0)
402  {
403  // How to dynamically allocate an array of fixed-size char * arrays in C++.
404  // http://stackoverflow.com/questions/8529359/creating-a-dynamic-sized-array-of-fixed-sized-int-arrays-in-c
405  typedef char * inner_array_t[4];
406  inner_array_t * qa_record = new inner_array_t[num_qa_rec];
407 
408  for (int i=0; i<num_qa_rec; i++)
409  for (int j=0; j<4; j++)
410  qa_record[i][j] = new char[MAX_STR_LENGTH+1];
411 
412  ex_err = exII::ex_get_qa (ex_id, qa_record);
413  EX_CHECK_ERR(ex_err, "Error reading the QA records.");
414 
415  // Print the QA records
416  if (verbose)
417  {
418  for (int i=0; i<num_qa_rec; i++)
419  {
420  libMesh::out << "QA Record: " << i << std::endl;
421  for (int j=0; j<4; j++)
422  libMesh::out << qa_record[i][j] << std::endl;
423  }
424  }
425 
426 
427  // Clean up dynamically-allocated memory
428  for (int i=0; i<num_qa_rec; i++)
429  for (int j=0; j<4; j++)
430  delete [] qa_record[i][j];
431 
432  delete [] qa_record;
433  }
434 }

1803 {

1805  return;

1807  // Loop over the element blocks and write the data one block at a time
1808  std::map<unsigned int, std::vector<unsigned int> > subdomain_map;
1809 
1810  // Ask the file how many element vars it has, store it in the num_elem_vars variable.
1811  ex_err = exII::ex_get_var_param(ex_id, "e", &num_elem_vars);
1812  EX_CHECK_ERR(ex_err, "Error reading number of elemental variables.");
1813 
1814  MeshBase::const_element_iterator mesh_it = mesh.active_elements_begin();
1815  const MeshBase::const_element_iterator end = mesh.active_elements_end();
1816 
1817  // loop through element and map between block and element vector
1818  for (; mesh_it!=end; ++mesh_it)
1819  {
1820  const Elem * elem = *mesh_it;
1821  subdomain_map[elem->subdomain_id()].push_back(elem->id());
1822  }
1823 
1824  // Use mesh.n_elem() to access into the values vector rather than
1825  // the number of elements the Exodus writer thinks the mesh has,
1826  // which may not include inactive elements.
1827  dof_id_type n_elem = mesh.n_elem();
1828 
1829  // For each variable, create a 'data' array which holds all the elemental variable
1830  // values *for a given block* on this processor, then write that data vector to file
1831  // before moving onto the next block.
1832  for (unsigned int i=0; i<static_cast<unsigned>(num_elem_vars); ++i)
1833  {
1834  // The size of the subdomain map is the number of blocks.
1835  std::map<unsigned int, std::vector<unsigned int> >::iterator it = subdomain_map.begin();
1836 
1837  for (unsigned int j=0; it!=subdomain_map.end(); ++it, ++j)
1838  {
1839  const std::vector<unsigned int> & elem_nums = (*it).second;
1840  const unsigned int num_elems_this_block =
1841  cast_int<unsigned int>(elem_nums.size());
1842  std::vector<Real> data(num_elems_this_block);
1843 
1844  for (unsigned int k=0; k<num_elems_this_block; ++k)
1845  data[k] = values[i*n_elem + elem_nums[k]];
1846 
1847  if (_single_precision)
1848  {
1849  std::vector<float> cast_data(data.begin(), data.end());
1850 
1851  ex_err = exII::ex_put_elem_var(ex_id,
1852  timestep,
1853  i+1,
1854  this->get_block_id(j),
1855  num_elems_this_block,
1856  &cast_data[0]);
1857  }
1858  else
1859  {
1860  ex_err = exII::ex_put_elem_var(ex_id,
1861  timestep,
1862  i+1,
1863  this->get_block_id(j),
1864  num_elems_this_block,
1865  &data[0]);
1866  }
1867  EX_CHECK_ERR(ex_err, "Error writing element values.");
1868  }
1869  }
1870

1872  EX_CHECK_ERR(ex_err, "Error flushing buffers to file.");
