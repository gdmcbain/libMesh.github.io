50 // abstract single predicate. Derived classes must implement the clone()
51 // function. Be careful using it since it allocates memory! The clone()
52 // function is necessary since the predicate class has pure virtual
53 // functions.
54 template <typename T>
55 struct predicate
56 {
57  virtual ~predicate() {}
58  virtual bool operator()(const T& it) const = 0;
59 
60 protected:
61  friend struct abstract_multi_predicate<T>;
62  virtual predicate* clone() const = 0;
63 };

243 // The facelocal_pid predicate returns true if the element
244 // pointed to is face-local to (is on or has a neighbor on the
245 // partition of) a given processor id.
246 template <typename T>
247 struct facelocal_pid : predicate<T>
248 {
249  // Constructor
250  facelocal_pid(processor_id_type p) : _pid(p) {}
251  virtual ~facelocal_pid() {}
252 
253  // op()
254  virtual bool operator()(const T& it) const libmesh_override
255  {
256  if ((*it)->processor_id() == _pid)
257  return true;
258  for (unsigned int n = 0; n != (*it)->n_neighbors(); ++n)
259  if ((*it)->neighbor(n) &&
260  (*it)->neighbor(n)->processor_id() == _pid)
261  return true;
262  return false;
263  }
264 
265 protected:
266  virtual predicate<T>* clone() const libmesh_override { return new facelocal_pid<T>(*this); }
267  const processor_id_type _pid;
268 };
