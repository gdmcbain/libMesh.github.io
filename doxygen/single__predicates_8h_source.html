48 // abstract single predicate. Derived classes must implement the clone()
49 // function. Be careful using it since it allocates memory! The clone()
50 // function is necessary since the predicate class has pure virtual
51 // functions.
52 template <typename T>
53 struct predicate
54 {
55  virtual ~predicate() {}
56  virtual bool operator()(const T& it) const = 0;
57 
58 
59 protected:
60  friend struct abstract_multi_predicate<T>;
61  virtual predicate* clone() const = 0;
62 
63 };

194 // The facelocal_pid predicate returns true if the element
195 // pointed to is face-local to (is on or has a neighbor on the
196 // partition of) a given processor id.
197 template <typename T>
198 struct facelocal_pid : predicate<T>
199 {
200  // Constructor
201  facelocal_pid(const processor_id_type p) : _pid(p) {}
202  virtual ~facelocal_pid() {}
203 
204  // op()
205  virtual bool operator()(const T& it) const {
206  if ((*it)->processor_id() == _pid)
207  return true;
208  for (unsigned int n = 0; n != (*it)->n_neighbors(); ++n)
209  if ((*it)->neighbor(n) &&
210  (*it)->neighbor(n)->processor_id() == _pid)
211  return true;
212  return false;
213  }
214 
215 protected:
216  virtual predicate<T>* clone() const { return new facelocal_pid<T>(*this); }
217  const processor_id_type _pid;
218 };
