48 void MeshTools::Modification::distort (MeshBase& mesh,
49  const Real factor,
50  const bool perturb_boundary)
51 {
52  libmesh_assert (mesh.n_nodes());
53  libmesh_assert (mesh.n_elem());
54  libmesh_assert ((factor >= 0.) && (factor <= 1.));
55 
56  START_LOG("distort()", "MeshTools::Modification");
57 
58 
59 
60  // First find nodes on the boundary and flag them
61  // so that we don't move them
62  // on_boundary holds false (not on boundary) and true (on boundary)
63  std::vector<bool> on_boundary (mesh.n_nodes(), false);
64 
65  if (!perturb_boundary) MeshTools::find_boundary_nodes (mesh, on_boundary);
66 
67  // Now calculate the minimum distance to
68  // neighboring nodes for each node.
69  // hmin holds these distances.
70  std::vector<float> hmin (mesh.n_nodes(),
71  std::numeric_limits<float>::max());
72 
73  MeshBase::element_iterator el = mesh.active_elements_begin();
74  const MeshBase::element_iterator end = mesh.active_elements_end();
75 
76  for (; el!=end; ++el)
77  for (unsigned int n=0; n<(*el)->n_nodes(); n++)
78  hmin[(*el)->node(n)] = std::min(hmin[(*el)->node(n)],
79  static_cast<float>((*el)->hmin()));
80 
81 
82  // Now actually move the nodes
83  {
84  const unsigned int seed = 123456;
85 
86  // seed the random number generator.
87  // We'll loop from 1 to n_nodes on every processor, even those
88  // that don't have a particular node, so that the pseudorandom
89  // numbers will be the same everywhere.
90  std::srand(seed);
91 
92  // If the node is on the boundary or
93  // the node is not used by any element (hmin[n]<1.e20)
94  // then we should not move it.
95  // [Note: Testing for (in)equality might be wrong
96  // (different types, namely float and double)]
97  for (unsigned int n=0; n<mesh.n_nodes(); n++)
98  if (!on_boundary[n] && (hmin[n] < 1.e20) )
99  {
100  // the direction, random but unit normalized
101 
102  Point dir( static_cast<Real>(std::rand())/static_cast<Real>(RAND_MAX),
103  (mesh.mesh_dimension() > 1) ?
104  static_cast<Real>(std::rand())/static_cast<Real>(RAND_MAX)
105  : 0.,
106  ((mesh.mesh_dimension() == 3) ?
107  static_cast<Real>(std::rand())/static_cast<Real>(RAND_MAX)
108  : 0.)
109  );
110 
111  dir(0) = (dir(0)-.5)*2.;
112  if (mesh.mesh_dimension() > 1)
113  dir(1) = (dir(1)-.5)*2.;
114  if (mesh.mesh_dimension() == 3)
115  dir(2) = (dir(2)-.5)*2.;
116 
117  dir = dir.unit();
118 
119  Node *node = mesh.node_ptr(n);
120  if (!node)
121  continue;
122 
123  (*node)(0) += dir(0)*factor*hmin[n];
124  if (mesh.mesh_dimension() > 1)
125  (*node)(1) += dir(1)*factor*hmin[n];
126  if (mesh.mesh_dimension() == 3)
127  (*node)(2) += dir(2)*factor*hmin[n];
128  }
129  }
130 
131 
132  // All done
133  STOP_LOG("distort()", "MeshTools::Modification");
134 }

212 void MeshTools::Modification::rotate (MeshBase& mesh,
213  const Real phi,
214  const Real theta,
215  const Real psi)
216 {
217 #if LIBMESH_DIM == 3
218  const Real p = -phi/180.*libMesh::pi;
219  const Real t = -theta/180.*libMesh::pi;
220  const Real s = -psi/180.*libMesh::pi;
221  const Real sp = std::sin(p), cp = std::cos(p);
222  const Real st = std::sin(t), ct = std::cos(t);
223  const Real ss = std::sin(s), cs = std::cos(s);
224 
225  // We follow the convention described at http://mathworld.wolfram.com/EulerAngles.html
226  // (equations 6-14 give the entries of the composite transformation matrix).
227  // The rotations are performed sequentially about the z, x, and z axes, in that order.
228  // A positive angle yields a counter-clockwise rotation about the axis in question.
229  const MeshBase::node_iterator nd_end = mesh.nodes_end();
230 
231  for (MeshBase::node_iterator nd = mesh.nodes_begin();
232  nd != nd_end; ++nd)
233  {
234  const Point pt = **nd;
235  const Real x = pt(0);
236  const Real y = pt(1);
237  const Real z = pt(2);
238  **nd = Point(( cp*cs-sp*ct*ss)*x + ( sp*cs+cp*ct*ss)*y + (st*ss)*z,
239  (-cp*ss-sp*ct*cs)*x + (-sp*ss+cp*ct*cs)*y + (st*cs)*z,
240  ( sp*st)*x + (-cp*st)*y + (ct)*z );
241  }
242 #else
243  libmesh_error_msg("MeshTools::Modification::rotate() requires libMesh to be compiled with LIBMESH_DIM==3");
244 #endif
245 }

248 void MeshTools::Modification::scale (MeshBase& mesh,
249  const Real xs,
250  const Real ys,
251  const Real zs)
252 {
253  const Real x_scale = xs;
254  Real y_scale = ys;
255  Real z_scale = zs;
256 
257  if (ys == 0.)
258  {
259  libmesh_assert_equal_to (zs, 0.);
260 
261  y_scale = z_scale = x_scale;
262  }
263 
264  // Scale the x coordinate in all dimensions
265  const MeshBase::node_iterator nd_end = mesh.nodes_end();
266 
267  for (MeshBase::node_iterator nd = mesh.nodes_begin();
268  nd != nd_end; ++nd)
269  (**nd)(0) *= x_scale;
270 
271 
272  // Only scale the y coordinate in 2 and 3D
273  if (mesh.spatial_dimension() < 2)
274  return;
275 
276  for (MeshBase::node_iterator nd = mesh.nodes_begin();
277  nd != nd_end; ++nd)
278  (**nd)(1) *= y_scale;
279 
280  // Only scale the z coordinate in 3D
281  if (mesh.spatial_dimension() < 3)
282  return;
283 
284  for (MeshBase::node_iterator nd = mesh.nodes_begin();
285  nd != nd_end; ++nd)
286  (**nd)(2) *= z_scale;
287 }

293 // UnstructuredMesh class member functions for mesh modification
294 void UnstructuredMesh::all_first_order ()
295 {
296  /*
297  * when the mesh is not prepared,
298  * at least renumber the nodes and
299  * elements, so that the node ids
300  * are correct
301  */
302  if (!this->_is_prepared)
303  this->renumber_nodes_and_elements ();
304 
305  START_LOG("all_first_order()", "Mesh");
306 
310  std::vector<bool> node_touched_by_me(this->max_node_id(), false);
311 
317  element_iterator endit = elements_end();
318  for (element_iterator it = elements_begin();
319  it != endit; ++it)
320  {
321  Elem* so_elem = *it;
322 
323  libmesh_assert(so_elem);
324 
325  /*
326  * build the first-order equivalent, add to
327  * the new_elements list.
328  */
329  Elem* lo_elem = Elem::build
330  (Elem::first_order_equivalent_type
331  (so_elem->type()), so_elem->parent()).release();
332 
333  for (unsigned int s=0; s != so_elem->n_sides(); ++s)
334  if (so_elem->neighbor(s) == remote_elem)
335  lo_elem->set_neighbor(s, const_cast<RemoteElem*>(remote_elem));
336 
337 #ifdef LIBMESH_ENABLE_AMR
338  /*
339  * Reset the parent links of any child elements
340  */
341  if (so_elem->has_children())
342  for (unsigned int c=0; c != so_elem->n_children(); ++c)
343  {
344  so_elem->child(c)->set_parent(lo_elem);
345  lo_elem->add_child(so_elem->child(c), c);
346  }
347 
348  /*
349  * Reset the child link of any parent element
350  */
351  if (so_elem->parent())
352  {
353  unsigned int c =
354  so_elem->parent()->which_child_am_i(so_elem);
355  lo_elem->parent()->replace_child(lo_elem, c);
356  }
357 
358  /*
359  * Copy as much data to the new element as makes sense
360  */
361  lo_elem->set_p_level(so_elem->p_level());
362  lo_elem->set_refinement_flag(so_elem->refinement_flag());
363  lo_elem->set_p_refinement_flag(so_elem->p_refinement_flag());
364 #endif
365 
366  libmesh_assert_equal_to (lo_elem->n_vertices(), so_elem->n_vertices());
367 
368  /*
369  * By definition the vertices of the linear and
370  * second order element are identically numbered.
371  * transfer these.
372  */
373  for (unsigned int v=0; v < so_elem->n_vertices(); v++)
374  {
375  lo_elem->set_node(v) = so_elem->get_node(v);
376  node_touched_by_me[lo_elem->node(v)] = true;
377  }
378 
385  libmesh_assert_equal_to (lo_elem->n_sides(), so_elem->n_sides());
386 
387  for (unsigned short s=0; s<so_elem->n_sides(); s++)
388  {
389  const std::vector<boundary_id_type> boundary_ids =
390  this->get_boundary_info().raw_boundary_ids (so_elem, s);
391 
392  this->get_boundary_info().add_side (lo_elem, s, boundary_ids);
393  }
394 
395  /*
396  * The new first-order element is ready.
397  * Inserting it into the mesh will replace and delete
398  * the second-order element.
399  */
400  lo_elem->set_id(so_elem->id());
401  lo_elem->processor_id() = so_elem->processor_id();
402  lo_elem->subdomain_id() = so_elem->subdomain_id();
403  this->insert_elem(lo_elem);
404  }
405 
406  const MeshBase::node_iterator nd_end = this->nodes_end();
407  MeshBase::node_iterator nd = this->nodes_begin();
408  while (nd != nd_end)
409  {
410  Node *the_node = *nd;
411  ++nd;
412  if (!node_touched_by_me[the_node->id()])
413  this->delete_node(the_node);
414  }
415 
416  STOP_LOG("all_first_order()", "Mesh");
417 
418  // On hanging nodes that used to also be second order nodes, we
419  // might now have an invalid nodal processor_id()
420  Partitioner::set_node_processor_ids(*this);
421 
422  // delete or renumber nodes, etc
423  this->prepare_for_use(/*skip_renumber =*/ false);
424 }

428 void UnstructuredMesh::all_second_order (const bool full_ordered)
429 {
430  // This function must be run on all processors at once
431  parallel_object_only();
432 
433  /*
434  * when the mesh is not prepared,
435  * at least renumber the nodes and
436  * elements, so that the node ids
437  * are correct
438  */
439  if (!this->_is_prepared)
440  this->renumber_nodes_and_elements ();
441 
442  /*
443  * If the mesh is empty
444  * then we have nothing to do
445  */
446  if (!this->n_elem())
447  return;
448 
449  /*
450  * If the mesh is already second order
451  * then we have nothing to do.
452  * We have to test for this in a round-about way to avoid
453  * a bug on distributed parallel meshes with more processors
454  * than elements.
455  */
456  bool already_second_order = false;
457  if (this->elements_begin() != this->elements_end() &&
458  (*(this->elements_begin()))->default_order() != FIRST)
459  already_second_order = true;
460  this->comm().max(already_second_order);
461  if (already_second_order)
462  return;
463 
464  START_LOG("all_second_order()", "Mesh");
465 
466  /*
467  * this map helps in identifying second order
468  * nodes. Namely, a second-order node:
469  * - edge node
470  * - face node
471  * - bubble node
472  * is uniquely defined through a set of adjacent
473  * vertices. This set of adjacent vertices is
474  * used to identify already added higher-order
475  * nodes. We are safe to use node id's since we
476  * make sure that these are correctly numbered.
477  */
478  std::map<std::vector<dof_id_type>, Node*> adj_vertices_to_so_nodes;
479 
480  /*
481  * for speed-up of the \p add_point() method, we
482  * can reserve memory. Guess the number of additional
483  * nodes for different dimensions
484  */
485  switch (this->mesh_dimension())
486  {
487  case 1:
488  /*
489  * in 1D, there can only be order-increase from Edge2
490  * to Edge3. Something like 1/2 of n_nodes() have
491  * to be added
492  */
493  this->reserve_nodes(static_cast<unsigned int>
494  (1.5*static_cast<double>(this->n_nodes())));
495  break;
496 
497  case 2:
498  /*
499  * in 2D, either refine from Tri3 to Tri6 (double the nodes)
500  * or from Quad4 to Quad8 (again, double) or Quad9 (2.25 that much)
501  */
502  this->reserve_nodes(static_cast<unsigned int>
503  (2*static_cast<double>(this->n_nodes())));
504  break;
505 
506 
507  case 3:
508  /*
509  * in 3D, either refine from Tet4 to Tet10 (factor = 2.5) up to
510  * Hex8 to Hex27 (something > 3). Since in 3D there _are_ already
511  * quite some nodes, and since we do not want to overburden the memory by
512  * a too conservative guess, use the lower bound
513  */
514  this->reserve_nodes(static_cast<unsigned int>
515  (2.5*static_cast<double>(this->n_nodes())));
516  break;
517 
518  default:
519  // Hm?
520  libmesh_error_msg("Unknown mesh dimension " << this->mesh_dimension());
521  }
522 
523 
524 
525  /*
526  * form a vector that will hold the node id's of
527  * the vertices that are adjacent to the son-th
528  * second-order node. Pull this outside of the
529  * loop so that silly compilers don't repeatedly
530  * create and destroy the vector.
531  */
532  std::vector<dof_id_type> adjacent_vertices_ids;
533 
540  const_element_iterator endit = elements_end();
541  for (const_element_iterator it = elements_begin();
542  it != endit; ++it)
543  {
544  // the linear-order element
545  const Elem* lo_elem = *it;
546 
547  libmesh_assert(lo_elem);
548 
549  // make sure it is linear order
550  if (lo_elem->default_order() != FIRST)
551  libmesh_error_msg("ERROR: This is not a linear element: type=" << lo_elem->type());
552 
553  // this does _not_ work for refined elements
554  libmesh_assert_equal_to (lo_elem->level (), 0);
555 
556  /*
557  * build the second-order equivalent, add to
558  * the new_elements list. Note that this here
559  * is the only point where \p full_ordered
560  * is necessary. The remaining code works well
561  * for either type of seconrd-order equivalent, e.g.
562  * Hex20 or Hex27, as equivalents for Hex8
563  */
564  Elem* so_elem =
565  Elem::build (Elem::second_order_equivalent_type(lo_elem->type(),
566  full_ordered) ).release();
567 
568  libmesh_assert_equal_to (lo_elem->n_vertices(), so_elem->n_vertices());
569 
570 
571  /*
572  * By definition the vertices of the linear and
573  * second order element are identically numbered.
574  * transfer these.
575  */
576  for (unsigned int v=0; v < lo_elem->n_vertices(); v++)
577  so_elem->set_node(v) = lo_elem->get_node(v);
578 
579  /*
580  * Now handle the additional mid-side nodes. This
581  * is simply handled through a map that remembers
582  * the already-added nodes. This map maps the global
583  * ids of the vertices (that uniquely define this
584  * higher-order node) to the new node.
585  * Notation: son = second-order node
586  */
587  const unsigned int son_begin = so_elem->n_vertices();
588  const unsigned int son_end = so_elem->n_nodes();
589 
590 
591  for (unsigned int son=son_begin; son<son_end; son++)
592  {
593  const unsigned int n_adjacent_vertices =
594  so_elem->n_second_order_adjacent_vertices(son);
595 
596  adjacent_vertices_ids.resize(n_adjacent_vertices);
597 
598  for (unsigned int v=0; v<n_adjacent_vertices; v++)
599  adjacent_vertices_ids[v] =
600  so_elem->node( so_elem->second_order_adjacent_vertex(son,v) );
601 
602  /*
603  * \p adjacent_vertices_ids is now in order of the current
604  * side. sort it, so that comparisons with the
605  * \p adjacent_vertices_ids created through other elements'
606  * sides can match
607  */
608  std::sort(adjacent_vertices_ids.begin(),
609  adjacent_vertices_ids.end());
610 
611 
612  // does this set of vertices already has a mid-node added?
613  std::pair<std::map<std::vector<dof_id_type>, Node*>::iterator,
614  std::map<std::vector<dof_id_type>, Node*>::iterator>
615  pos = adj_vertices_to_so_nodes.equal_range (adjacent_vertices_ids);
616 
617  // no, not added yet
618  if (pos.first == pos.second)
619  {
620  /*
621  * for this set of vertices, there is no
622  * second_order node yet. Add it.
623  *
624  * compute the location of the new node as
625  * the average over the adjacent vertices.
626  */
627  Point new_location = this->point(adjacent_vertices_ids[0]);
628  for (unsigned int v=1; v<n_adjacent_vertices; v++)
629  new_location += this->point(adjacent_vertices_ids[v]);
630 
631  new_location /= static_cast<Real>(n_adjacent_vertices);
632 
633  /* Add the new point to the mesh, giving it a globally
634  * well-defined processor id.
635  */
636  Node* so_node = this->add_point
637  (new_location, DofObject::invalid_id,
638  this->node(adjacent_vertices_ids[0]).processor_id());
639 
640  /*
641  * insert the new node with its defining vertex
642  * set into the map, and relocate pos to this
643  * new entry, so that the so_elem can use
644  * \p pos for inserting the node
645  */
646  adj_vertices_to_so_nodes.insert(pos.first,
647  std::make_pair(adjacent_vertices_ids,
648  so_node));
649 
650  so_elem->set_node(son) = so_node;
651  }
652  // yes, already added.
653  else
654  {
655  libmesh_assert(pos.first->second);
656 
657  so_elem->set_node(son) = pos.first->second;
658  }
659  }
660 
661 
673  libmesh_assert_equal_to (lo_elem->n_sides(), so_elem->n_sides());
674 
675  for (unsigned short s=0; s<lo_elem->n_sides(); s++)
676  {
677  const std::vector<boundary_id_type> boundary_ids =
678  this->get_boundary_info().raw_boundary_ids (lo_elem, s);
679 
680  this->get_boundary_info().add_side (so_elem, s, boundary_ids);
681 
682  if (lo_elem->neighbor(s) == remote_elem)
683  so_elem->set_neighbor(s, const_cast<RemoteElem*>(remote_elem));
684  }
685 
686  /*
687  * The new second-order element is ready.
688  * Inserting it into the mesh will replace and delete
689  * the first-order element.
690  */
691  so_elem->set_id(lo_elem->id());
692  so_elem->processor_id() = lo_elem->processor_id();
693  so_elem->subdomain_id() = lo_elem->subdomain_id();
694  this->insert_elem(so_elem);
695  }
696 
697  // we can clear the map
698  adj_vertices_to_so_nodes.clear();
699 
700 
701  STOP_LOG("all_second_order()", "Mesh");
702 
703  // In a ParallelMesh our ghost node processor ids may be bad and
704  // the ids of nodes touching remote elements may be inconsistent.
