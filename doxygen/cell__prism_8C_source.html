162 bool Prism::is_child_on_side(const unsigned int c,
163  const unsigned int s) const
164 {
165  libmesh_assert_less (c, this->n_children());
166  libmesh_assert_less (s, this->n_sides());
167 
168  for (unsigned int i = 0; i != 4; ++i)
169  if (Prism6::side_elems_map[s][i] == c)
170  return true;
171  return false;
