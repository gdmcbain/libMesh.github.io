829 Number MeshData::operator() (const Node* node,
830  const unsigned int i) const
831 {
832  libmesh_assert (_active || _compatibility_mode);
833  libmesh_assert (_node_data_closed);
834 
835  std::map<const Node*,
836  std::vector<Number> >::const_iterator pos = _node_data.find(node);
837 
838  if (pos == _node_data.end())
839  return libMesh::zero;
840 
841  // we only get here when pos != _node_data.end()
842  libmesh_assert_less (i, pos->second.size());
843  return pos->second[i];
844 }

1020 inline
1021 void MeshData::add_foreign_node_id (const Node* node,
1022  const unsigned int foreign_node_id)
1023 {
1024  if (_active)
1025  {
1026  libmesh_assert (!_node_id_map_closed);
1027  libmesh_assert(node);
1028  libmesh_assert (_node_id.find(node) == _node_id.end());
1029  libmesh_assert (_id_node.find(foreign_node_id) == _id_node.end());
1030 
1031  /*
1032  * _always_ insert in _id_node and _node_id. If we would
1033  * use the mesh.node(unsigned int) method or the node.id()
1034  * to get Node* and unsigned int, respectively, we would not
1035  * be safe any more when the mesh gets refined or re-numbered
1036  * within libMesh. And we could get in big trouble that would
1037  * be hard to find when importing data _after_ having refined...
1038  */
1039  _node_id.insert(std::make_pair(node, foreign_node_id));
1040  _id_node.insert(std::make_pair(foreign_node_id, node));
1041  }
1042 }
