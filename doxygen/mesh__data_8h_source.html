838 Number MeshData::operator() (const Node* node,
839  const unsigned int i) const
840 {
841  libmesh_assert (_active || _compatibility_mode);
842  libmesh_assert (_node_data_closed);
843 
844  std::map<const Node*,
845  std::vector<Number> >::const_iterator pos = _node_data.find(node);
846 
847  if (pos == _node_data.end())
848  return libMesh::zero;
849 
850  // we only get here when pos != _node_data.end()
851  libmesh_assert_less (i, pos->second.size());
852  return pos->second[i];
853 }

1029 inline
1030 void MeshData::add_foreign_node_id (const Node* node,
1031  const unsigned int foreign_node_id)
1032 {
1033  if (_active)
1034  {
1035  libmesh_assert (!_node_id_map_closed);
1036  libmesh_assert(node);
1037  libmesh_assert (_node_id.find(node) == _node_id.end());
1038  libmesh_assert (_id_node.find(foreign_node_id) == _id_node.end());
1039 
1040  /*
1041  * _always_ insert in _id_node and _node_id. If we would
1042  * use the mesh.node(unsigned int) method or the node.id()
1043  * to get Node* and unsigned int, respectively, we would not
1044  * be safe any more when the mesh gets refined or re-numbered
1045  * within libMesh. And we could get in big trouble that would
1046  * be hard to find when importing data _after_ having refined...
1047  */
1048  _node_id.insert(std::make_pair(node, foreign_node_id));
1049  _id_node.insert(std::make_pair(foreign_node_id, node));
1050  }
1051 }
