44 template <typename Output=Number, typename OutputGradient=Gradient>
45 class ParsedFunction : public FunctionBase<Output>
46 {
47 public:
48  explicit
49  ParsedFunction (const std::string& expression, const std::vector<std::string>* additional_vars=NULL,
50  const std::vector<Output>* initial_vals=NULL) :
51  _expression (expression),
52  _spacetime (LIBMESH_DIM+1 + (additional_vars ?
53  additional_vars->size() : 0)),
54  _valid_derivatives (true),
55  _additional_vars (additional_vars ? *additional_vars :
56  std::vector<std::string>()),
57  _initial_vals (initial_vals ? *initial_vals :
58  std::vector<Output>())
59  // Size the spacetime vector to account for space, time, and any additional
60  // variables passed
61  //_spacetime(LIBMESH_DIM+1 + (additional_vars ? additional_vars->size() : 0)),
62  {
63  std::string variables = "x";
64 #if LIBMESH_DIM > 1
65  variables += ",y";
66 #endif
67 #if LIBMESH_DIM > 2
68  variables += ",z";
69 #endif
70  variables += ",t";
71 
72  // If additional vars were passed, append them to the string
73  // that we send to the function parser. Also add them to the
74  // end of our spacetime vector
75  if (additional_vars)
76  {
77  for (unsigned int i=0; i < additional_vars->size(); ++i)
78  {
79  variables += "," + (*additional_vars)[i];
80  // Initialize extra variables to the vector passed in or zero
81  // Note: The initial_vals vector can be shorter than the additional_vars vector
82  _spacetime[LIBMESH_DIM+1 + i] = (initial_vals && i < initial_vals->size()) ? (*initial_vals)[i] : 0;
83  }
84  }
85 
86  size_t nextstart = 0, end = 0;
87 
88  while (end != std::string::npos)
89  {
90  // If we're past the end of the string, we can't make any more
91  // subparsers
92  if (nextstart >= expression.size())
93  break;
94 
95  // If we're at the start of a brace delimited section, then we
96  // parse just that section:
97  if (expression[nextstart] == '{')
98  {
99  nextstart++;
100  end = expression.find('}', nextstart);
101  }
102  // otherwise we parse the whole thing
103  else
104  end = std::string::npos;
105 
106  // We either want the whole end of the string (end == npos) or
107  // a substring in the middle.
108  std::string subexpression =
109  expression.substr(nextstart, (end == std::string::npos) ?
110  std::string::npos : end - nextstart);
111 
112  // fparser can crash on empty expressions
113  if (subexpression.empty())
114  libmesh_error_msg("ERROR: FunctionParser is unable to parse empty expression.\n");
115 
116  // Parse (and optimize if possible) the subexpression.
117  // Add some basic constants, to Real precision.
118  FunctionParserADBase<Output> fp;
119  fp.AddConstant("NaN", std::numeric_limits<Real>::quiet_NaN());
120  fp.AddConstant("pi", std::acos(Real(-1)));
121  fp.AddConstant("e", std::exp(Real(1)));
122  if (fp.Parse(subexpression, variables) != -1) // -1 for success
123  libmesh_error_msg("ERROR: FunctionParser is unable to parse expression: " << subexpression << '\n' << fp.ErrorMsg());
124 
125  // use of derivatives is optional. suppress error output on the console
126  // use the has_derivatives() method to check if AutoDiff was successful.
127  fp.silenceAutoDiffErrors();
128 
129  // generate derivatives through automatic differentiation
130  FunctionParserADBase<Output> dx_fp(fp);
131  if (dx_fp.AutoDiff("x") != -1) // -1 for success
132  _valid_derivatives = false;
133  dx_fp.Optimize();
134  dx_parsers.push_back(dx_fp);
135 #if LIBMESH_DIM > 1
136  FunctionParserADBase<Output> dy_fp(fp);
137  if (dy_fp.AutoDiff("y") != -1) // -1 for success
138  _valid_derivatives = false;
139  dy_fp.Optimize();
140  dy_parsers.push_back(dy_fp);
141 #endif
142 #if LIBMESH_DIM > 2
143  FunctionParserADBase<Output> dz_fp(fp);
144  if (dz_fp.AutoDiff("z") != -1) // -1 for success
145  _valid_derivatives = false;
146  dz_fp.Optimize();
147  dz_parsers.push_back(dz_fp);
148 #endif
149  FunctionParserADBase<Output> dt_fp(fp);
150  if (dt_fp.AutoDiff("t") != -1) // -1 for success
151  _valid_derivatives = false;
152  dt_fp.Optimize();
153  dt_parsers.push_back(dt_fp);
154 
155  // now optimise original function (after derivatives are taken)
156  fp.Optimize();
157  parsers.push_back(fp);
158 
159  // If at end, use nextstart=maxSize. Else start at next
160  // character.
161  nextstart = (end == std::string::npos) ?
162  std::string::npos : end + 1;
163  }
164 
165  this->_initialized = true;
166  }

274  // Evaluate the ith FunctionParser and check the result
275  inline Output eval(FunctionParserADBase<Output> & parser,
276  const std::string & libmesh_dbg_var(function_name),
277  unsigned int libmesh_dbg_var(component_idx))
278  {
279 #ifndef NDEBUG
280  Output result = parser.Eval(&_spacetime[0]);
281  int error_code = parser.EvalError();
282  if (error_code)
283  {
284  libMesh::err << "ERROR: FunctionParser is unable to evaluate component "
285  << component_idx
286  << " of expression '"
287  << function_name
288  << "' with arguments:\n";
289  for (unsigned int j=0; j<_spacetime.size(); ++j)
290  libMesh::err << '\t' << _spacetime[j] << '\n';
291  libMesh::err << '\n';
292 
293  // Currently no API to report error messages, we'll do it manually
294  std::string error_message = "Reason: ";
295 
296  switch (error_code)
297  {
298  case 1:
299  error_message += "Division by zero";
300  break;
301  case 2:
302  error_message += "Square Root error (negative value)";
303  break;
304  case 3:
305  error_message += "Log error (negative value)";
306  break;
307  case 4:
308  error_message += "Trigonometric error (asin or acos of illegal value)";
309  break;
310  case 5:
311  error_message += "Maximum recursion level reached";
312  break;
313  default:
314  error_message += "Unknown";
315  break;
316  }
317  libmesh_error_msg(error_message);
318  }
319 
320  return result;
321 #else
322  return parser.Eval(&_spacetime[0]);
323 #endif
324  }
