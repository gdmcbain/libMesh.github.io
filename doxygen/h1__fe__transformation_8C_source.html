24 template< typename OutputShape >
25 void H1FETransformation<OutputShape>::map_phi( const unsigned int dim,
26  const Elem* const elem,
27  const std::vector<Point>& qp,
28  const FEGenericBase<OutputShape>& fe,
29  std::vector<std::vector<OutputShape> >& phi ) const
30 {
31  switch(dim)
32  {
33  case 0:
34  {
35  for (unsigned int i=0; i<phi.size(); i++)
36  {
37  libmesh_assert_equal_to ( qp.size(), phi[i].size() );
38  for (unsigned int p=0; p<phi[i].size(); p++)
39  FEInterface::shape<OutputShape>(0, fe.get_fe_type(), elem, i, qp[p], phi[i][p]);
40  }
41  break;
42  }
43  case 1:
44  {
45  for (unsigned int i=0; i<phi.size(); i++)
46  {
47  libmesh_assert_equal_to ( qp.size(), phi[i].size() );
48  for (unsigned int p=0; p<phi[i].size(); p++)
49  FEInterface::shape<OutputShape>(1, fe.get_fe_type(), elem, i, qp[p], phi[i][p]);
50  }
51  break;
52  }
53  case 2:
54  {
55  for (unsigned int i=0; i<phi.size(); i++)
56  {
57  libmesh_assert_equal_to ( qp.size(), phi[i].size() );
58  for (unsigned int p=0; p<phi[i].size(); p++)
59  FEInterface::shape<OutputShape>(2, fe.get_fe_type(), elem, i, qp[p], phi[i][p]);
60  }
61  break;
62  }
63  case 3:
64  {
65  for (unsigned int i=0; i<phi.size(); i++)
66  {
67  libmesh_assert_equal_to ( qp.size(), phi[i].size() );
68  for (unsigned int p=0; p<phi[i].size(); p++)
69  FEInterface::shape<OutputShape>(3, fe.get_fe_type(), elem, i, qp[p], phi[i][p]);
70  }
71  break;
72  }
73 
74  default:
75  libmesh_error_msg("Invalid dim = " << dim);
76  }
77 }

80 template< typename OutputShape >
81 void H1FETransformation<OutputShape>::map_dphi( const unsigned int dim,
82  const Elem* const,
83  const std::vector<Point>&,
84  const FEGenericBase<OutputShape>& fe,
85  std::vector<std::vector<typename FEGenericBase<OutputShape>::OutputGradient> >& dphi,
86  std::vector<std::vector<OutputShape> >& dphidx,
87  std::vector<std::vector<OutputShape> >& dphidy,
88  std::vector<std::vector<OutputShape> >& dphidz ) const
89 {
90  switch(dim)
91  {
92  case 0: // No derivatives in 0D
93  {
94  for (unsigned int i=0; i<dphi.size(); i++)
95  for (unsigned int p=0; p<dphi[i].size(); p++)
96  {
97  dphi[i][p] = 0.;
98  }
99  break;
100  }
101 
102  case 1:
103  {
104  const std::vector<std::vector<OutputShape> >& dphidxi = fe.get_dphidxi();
105 
106  const std::vector<Real>& dxidx_map = fe.get_fe_map().get_dxidx();
107 #if LIBMESH_DIM>1
108  const std::vector<Real>& dxidy_map = fe.get_fe_map().get_dxidy();
109 #endif
110 #if LIBMESH_DIM>2
111  const std::vector<Real>& dxidz_map = fe.get_fe_map().get_dxidz();
112 #endif
113 
114  for (unsigned int i=0; i<dphi.size(); i++)
115  for (unsigned int p=0; p<dphi[i].size(); p++)
116  {
117  // dphi/dx = (dphi/dxi)*(dxi/dx)
118  dphi[i][p].slice(0) = dphidx[i][p] = dphidxi[i][p]*dxidx_map[p];
119 
120 #if LIBMESH_DIM>1
121  dphi[i][p].slice(1) = dphidy[i][p] = dphidxi[i][p]*dxidy_map[p];
122 #endif
123 #if LIBMESH_DIM>2
124  dphi[i][p].slice(2) = dphidz[i][p] = dphidxi[i][p]*dxidz_map[p];
125 #endif
126  }
127 
128  break;
129  }
130 
131  case 2:
132  {
133  const std::vector<std::vector<OutputShape> >& dphidxi = fe.get_dphidxi();
134  const std::vector<std::vector<OutputShape> >& dphideta = fe.get_dphideta();
135 
136  const std::vector<Real>& dxidx_map = fe.get_fe_map().get_dxidx();
137  const std::vector<Real>& dxidy_map = fe.get_fe_map().get_dxidy();
138 #if LIBMESH_DIM > 2
139  const std::vector<Real>& dxidz_map = fe.get_fe_map().get_dxidz();
140 #endif
141 
142  const std::vector<Real>& detadx_map = fe.get_fe_map().get_detadx();
143  const std::vector<Real>& detady_map = fe.get_fe_map().get_detady();
144 #if LIBMESH_DIM > 2
145  const std::vector<Real>& detadz_map = fe.get_fe_map().get_detadz();
146 #endif
147 
148  for (unsigned int i=0; i<dphi.size(); i++)
149  for (unsigned int p=0; p<dphi[i].size(); p++)
150  {
151  // dphi/dx = (dphi/dxi)*(dxi/dx) + (dphi/deta)*(deta/dx)
152  dphi[i][p].slice(0) = dphidx[i][p] = (dphidxi[i][p]*dxidx_map[p] +
153  dphideta[i][p]*detadx_map[p]);
154 
155  // dphi/dy = (dphi/dxi)*(dxi/dy) + (dphi/deta)*(deta/dy)
156  dphi[i][p].slice(1) = dphidy[i][p] = (dphidxi[i][p]*dxidy_map[p] +
157  dphideta[i][p]*detady_map[p]);
158 
159 #if LIBMESH_DIM > 2
160  // dphi/dz = (dphi/dxi)*(dxi/dz) + (dphi/deta)*(deta/dz)
161  dphi[i][p].slice(2) = dphidz[i][p] = (dphidxi[i][p]*dxidz_map[p] +
162  dphideta[i][p]*detadz_map[p]);
163 #endif
164  }
165 
166  break;
167  }
168 
169  case 3:
170  {
171  const std::vector<std::vector<OutputShape> >& dphidxi = fe.get_dphidxi();
172  const std::vector<std::vector<OutputShape> >& dphideta = fe.get_dphideta();
173  const std::vector<std::vector<OutputShape> >& dphidzeta = fe.get_dphidzeta();
174 
175  const std::vector<Real>& dxidx_map = fe.get_fe_map().get_dxidx();
176  const std::vector<Real>& dxidy_map = fe.get_fe_map().get_dxidy();
177  const std::vector<Real>& dxidz_map = fe.get_fe_map().get_dxidz();
178 
179  const std::vector<Real>& detadx_map = fe.get_fe_map().get_detadx();
180  const std::vector<Real>& detady_map = fe.get_fe_map().get_detady();
181  const std::vector<Real>& detadz_map = fe.get_fe_map().get_detadz();
182 
183  const std::vector<Real>& dzetadx_map = fe.get_fe_map().get_dzetadx();
184  const std::vector<Real>& dzetady_map = fe.get_fe_map().get_dzetady();
185  const std::vector<Real>& dzetadz_map = fe.get_fe_map().get_dzetadz();
186 
187  for (unsigned int i=0; i<dphi.size(); i++)
188  for (unsigned int p=0; p<dphi[i].size(); p++)
189  {
190  // dphi/dx = (dphi/dxi)*(dxi/dx) + (dphi/deta)*(deta/dx) + (dphi/dzeta)*(dzeta/dx);
191  dphi[i][p].slice(0) = dphidx[i][p] = (dphidxi[i][p]*dxidx_map[p] +
192  dphideta[i][p]*detadx_map[p] +
193  dphidzeta[i][p]*dzetadx_map[p]);
194 
195  // dphi/dy = (dphi/dxi)*(dxi/dy) + (dphi/deta)*(deta/dy) + (dphi/dzeta)*(dzeta/dy);
196  dphi[i][p].slice(1) = dphidy[i][p] = (dphidxi[i][p]*dxidy_map[p] +
197  dphideta[i][p]*detady_map[p] +
198  dphidzeta[i][p]*dzetady_map[p]);
199 
200  // dphi/dz = (dphi/dxi)*(dxi/dz) + (dphi/deta)*(deta/dz) + (dphi/dzeta)*(dzeta/dz);
201  dphi[i][p].slice(2) = dphidz[i][p] = (dphidxi[i][p]*dxidz_map[p] +
202  dphideta[i][p]*detadz_map[p] +
203  dphidzeta[i][p]*dzetadz_map[p]);
204  }
205  break;
206  }
207 
208  default:
209  libmesh_error_msg("Invalid dim = " << dim);
210  } // switch(dim)
211 }

227  libmesh_do_once(
228  if (!elem->has_affine_map())
229  {
230  libMesh::err << "WARNING: Second derivatives are not currently "

232  << std::endl;
233  }
234  );
235 
236 
237  switch (dim)
238  {
239  case 0: // No derivatives in 0D
240  {
241  for (unsigned int i=0; i<d2phi.size(); i++)
242  for (unsigned int p=0; p<d2phi[i].size(); p++)
243  {
244  d2phi[i][p] = 0.;
245  }
246 
247  break;
248  }
249

264  {

266  d2phidxi2[i][p]*dxidx_map[p]*dxidx_map[p];
267 #if LIBMESH_DIM>1
268  d2phi[i][p].slice(0).slice(1) =
269  d2phi[i][p].slice(1).slice(0) = d2phidxdy[i][p] =
270  d2phidxi2[i][p]*dxidx_map[p]*dxidy_map[p];

274 #endif

289  }

291  case 2:

294  const std::vector<std::vector<OutputShape> >& d2phidxideta = fe.get_d2phidxideta();

549  + (dphidzeta[i][p].slice(1))*dzetadz_map[p];

554

557  + (dphidzeta[i][p].slice(2))*dzetadx_map[p];
