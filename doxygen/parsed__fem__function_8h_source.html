51  ParsedFEMFunction (const System& sys,
52  const std::string& expression,
53  const std::vector<std::string>* additional_vars=NULL,
54  const std::vector<Output>* initial_vals=NULL)
55  : _sys(sys),
56  _expression(expression),
57  _n_vars(sys.n_vars())
58  {
59  std::string variables = "x";
60 #if LIBMESH_DIM > 1
61  variables += ",y";
62 #endif
63 #if LIBMESH_DIM > 2
64  variables += ",z";
65 #endif
66  variables += ",t";
67 
68  _spacetime.resize(LIBMESH_DIM+1 + _n_vars + (additional_vars ? additional_vars->size() : 0));
69 
70  for (unsigned int v=0; v != _n_vars; ++v)
71  {
72  variables += ',';
73  variables += sys.variable_name(v);
74  }
75 
76  // If additional vars were passed, append them to the string
77  // that we send to the function parser. Also add them to the
78  // end of our spacetime vector
79  if (additional_vars)
80  {
81  if (initial_vals)
82  std::copy(initial_vals->begin(), initial_vals->end(), std::back_inserter(_initial_vals));
83 
84  std::copy(additional_vars->begin(), additional_vars->end(), std::back_inserter(_additional_vars));
85 
86  for (unsigned int i=0; i < additional_vars->size(); ++i)
87  {
88  variables += "," + (*additional_vars)[i];
89  // Initialize extra variables to the vector passed in or zero
90  // Note: The initial_vals vector can be shorter than the additional_vars vector
91  _spacetime[LIBMESH_DIM+1+_n_vars + i] = (initial_vals && i < initial_vals->size()) ? (*initial_vals)[i] : 0;
92  }
93  }
94 
95  size_t nextstart = 0, end = 0;
96 
97  while (end != std::string::npos)
98  {
99  // If we're past the end of the string, we can't make any more
100  // subparsers
101  if (nextstart >= expression.size())
102  break;
103 
104  // If we're at the start of a brace delimited section, then we
105  // parse just that section:
106  if (expression[nextstart] == '{')
107  {
108  nextstart++;
109  end = expression.find('}', nextstart);
110  }
111  // otherwise we parse the whole thing
112  else
113  end = std::string::npos;
114 
115  // We either want the whole end of the string (end == npos) or
116  // a substring in the middle.
117  std::string subexpression =
118  expression.substr(nextstart, (end == std::string::npos) ?
119  std::string::npos : end - nextstart);
120 
121  // fparser can crash on empty expressions
122  libmesh_assert(!subexpression.empty());
123 
124 #ifdef LIBMESH_HAVE_FPARSER
125  // Parse (and optimize if possible) the subexpression.
126  // Add some basic constants, to Real precision.
127  FunctionParserBase<Output> fp;
128  fp.AddConstant("NaN", std::numeric_limits<Real>::quiet_NaN());
129  fp.AddConstant("pi", std::acos(Real(-1)));
130  fp.AddConstant("e", std::exp(Real(1)));
131  fp.Parse(subexpression, variables);
132  fp.Optimize();
133  parsers.push_back(fp);
134 #else
135  libmesh_error_msg("ERROR: This functionality requires fparser!");
136 #endif
137 
138  // If at end, use nextstart=maxSize. Else start at next
139  // character.
140  nextstart = (end == std::string::npos) ?
141  std::string::npos : end + 1;
142  }
143  }
