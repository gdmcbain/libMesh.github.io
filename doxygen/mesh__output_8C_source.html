30 void MeshOutput<MT>::write_equation_systems (const std::string& fname,
31  const EquationSystems& es,
32  const std::set<std::string>* system_names)
33 {
34  START_LOG("write_equation_systems()", "MeshOutput");
35 
36  // We may need to gather and/or renumber a ParallelMesh to output
37  // it, making that const qualifier in our constructor a dirty lie
38  MT& my_mesh = const_cast<MT&>(*_obj);
39 
40  // If we're asked to write data that's associated with a different
41  // mesh, output files full of garbage are the result.
42  libmesh_assert_equal_to(&es.get_mesh(), _obj);
43 
44  // A non-renumbered mesh may not have a contiguous numbering, and
45  // that needs to be fixed before we can build a solution vector.
46  if (my_mesh.max_elem_id() != my_mesh.n_elem() ||
47  my_mesh.max_node_id() != my_mesh.n_nodes())
48  {
49  // If we were allowed to renumber then we should have already
50  // been properly renumbered...
51  libmesh_assert(!my_mesh.allow_renumbering());
52 
53  libmesh_do_once(libMesh::out <<
54  "Warning: This MeshOutput subclass only supports meshes which are contiguously renumbered!"
55  << std::endl;);
56 
57  my_mesh.allow_renumbering(true);
58 
59  my_mesh.renumber_nodes_and_elements();
60 
61  // Not sure what good going back to false will do here, the
62  // renumbering horses have already left the barn...
63  my_mesh.allow_renumbering(false);
64  }
65 
66  MeshSerializer serialize(const_cast<MT&>(*_obj), !_is_parallel_format);
67 
68  // Build the nodal solution values & get the variable
69  // names from the EquationSystems object
70  std::vector<Number> soln;
71  std::vector<std::string> names;
72 
73  this->_build_variable_names_and_solution_vector(es, soln, names, system_names);
74  //es.build_variable_names (names);
75  //es.build_solution_vector (soln);
76 
77  this->write_nodal_data (fname, soln, names);
78 
79  STOP_LOG("write_equation_systems()", "MeshOutput");
80 }

86 _build_variable_names_and_solution_vector (const EquationSystems& es,
87  std::vector<Number>& soln,

90 {

96 
97  es.build_variable_names (names, NULL, system_names);
98  es.build_solution_vector (soln, system_names);
99 
100  // For now, if we're doing a parallel format we're going to broadcast the vector from processor 0
101  // to all of the processors to mimic what build_solution_vector used to do.
102  // this is TERRIBLE and WASTEFUL but it's only temporary until we redesign the output of build_solution_vector
103  // and the inputs to the I/O... both of which should actually be NumericVectors....
104  if(_is_parallel_format)
105  {
106  size_t size = soln.size();
107  _obj->comm().broadcast(size);
108 
109  if(_obj->comm().rank())
110  soln.resize(size);
111 
112  _obj->comm().broadcast(soln);
113  }
114 }
