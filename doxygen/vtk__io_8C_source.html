80  (LIBMESH_DETECTED_VTK_VERSION_MINOR == (minor) && \

170 void VTKIO::nodes_to_vtk()
171 {
172  const MeshBase& mesh = MeshOutput<MeshBase>::mesh();
173 
174  // containers for points and coordinates of points
175  vtkSmartPointer<vtkPoints> points = vtkSmartPointer<vtkPoints>::New();
176  vtkSmartPointer<vtkDoubleArray> pcoords = vtkSmartPointer<vtkDoubleArray>::New();
177  pcoords->SetNumberOfComponents(LIBMESH_DIM);
178  points->SetNumberOfPoints(mesh.n_local_nodes()); // it seems that it needs this to prevent a segfault
179 
180  unsigned int local_node_counter = 0;
181 
182  MeshBase::const_node_iterator nd = mesh.local_nodes_begin();
183  MeshBase::const_node_iterator nd_end = mesh.local_nodes_end();
184  for (; nd != nd_end; nd++, ++local_node_counter)
185  {
186  Node* node = (*nd);
187 
188  double pnt[LIBMESH_DIM];
189  for (unsigned int i=0; i<LIBMESH_DIM; ++i)
190  pnt[i] = (*node)(i);
191 
192  // Fill mapping between global and local node numbers
193  _local_node_map[node->id()] = local_node_counter;
194 
195  // add point
196  pcoords->InsertNextTupleValue(pnt);
197  }
198 
199  // add coordinates to points
200  points->SetData(pcoords);
201 
202  // add points to grid
203  _vtk_grid->SetPoints(points);
204 }

208 void VTKIO::cells_to_vtk()
209 {
210  const MeshBase& mesh = MeshOutput<MeshBase>::mesh();
211 
212  vtkSmartPointer<vtkCellArray> cells = vtkSmartPointer<vtkCellArray>::New();
213  vtkSmartPointer<vtkIdList> pts = vtkSmartPointer<vtkIdList>::New();
214 
215  std::vector<int> types(mesh.n_active_local_elem());
216  unsigned active_element_counter = 0;
217 
218  vtkSmartPointer<vtkIntArray> elem_id = vtkSmartPointer<vtkIntArray>::New();
219  elem_id->SetName("libmesh_elem_id");
220  elem_id->SetNumberOfComponents(1);
221 
222  vtkSmartPointer<vtkIntArray> subdomain_id = vtkSmartPointer<vtkIntArray>::New();
223  subdomain_id->SetName("subdomain_id");
224  subdomain_id->SetNumberOfComponents(1);
225 
226  MeshBase::const_element_iterator it = mesh.active_local_elements_begin();
227  const MeshBase::const_element_iterator end = mesh.active_local_elements_end();
228  for (; it != end; ++it, ++active_element_counter)
229  {
230  Elem *elem = *it;
231 
232  pts->SetNumberOfIds(elem->n_nodes());
233 
234  // get the connectivity for this element
235  std::vector<dof_id_type> conn;
236  elem->connectivity(0, VTK, conn);
237 
238  for (unsigned int i=0; i<conn.size(); ++i)
239  {
240  // If the node ID is not found in the _local_node_map, we'll
241  // add it to the _vtk_grid. NOTE[JWP]: none of the examples
242  // I have actually enters this section of code...
243  if (_local_node_map.find(conn[i]) == _local_node_map.end())
244  {
245  dof_id_type global_node_id = elem->node(i);
246 
247  const Node* the_node = mesh.node_ptr(global_node_id);
248 
249  // Error checking...
250  if (the_node == NULL)
251  libmesh_error_msg("Error getting pointer to node " << global_node_id << "!");
252 
253  // InsertNextPoint accepts either a double or float array of length 3.
254  Real pt[3] = {0., 0., 0.};
255  for (unsigned int d=0; d<LIBMESH_DIM; ++d)
256  pt[d] = (*the_node)(d);
257 
258  // Insert the point into the _vtk_grid
259  vtkIdType local = _vtk_grid->GetPoints()->InsertNextPoint(pt);
260 
261  // Update the _local_node_map with the ID returned by VTK
262  _local_node_map[global_node_id] =
263  cast_int<dof_id_type>(local);
264  }
265 
266  // Otherwise, the node ID was found in the _local_node_map, so
267  // insert it into the vtkIdList.
268  pts->InsertId(i, _local_node_map[conn[i]]);
269  }
270 
271  vtkIdType vtkcellid = cells->InsertNextCell(pts);
272  types[active_element_counter] =
273  cast_int<int>(this->get_elem_type(elem->type()));
274  elem_id->InsertTuple1(vtkcellid, elem->id());
275  subdomain_id->InsertTuple1(vtkcellid, elem->subdomain_id());
276  } // end loop over active elements
277 
278  _vtk_grid->SetCells(&types[0], cells);
279  _vtk_grid->GetCellData()->AddArray(elem_id);
280  _vtk_grid->GetCellData()->AddArray(subdomain_id);
281 }

296  const System& sys = es.get_system(i);
297  System::const_vectors_iterator v_end = sys.vectors_end();
298  System::const_vectors_iterator it = sys.vectors_begin();
299  for (; it!= v_end; ++it)
300  {
301  // for all vectors on this system
302  std::vector<Number> values;
303  // libMesh::out<<"it "<<it->first<<std::endl;
304 
305  it->second->localize_to_one(values, 0);
306  // libMesh::out<<"finish localize"<<std::endl;
307  vecs[it->first] = values;
308  }
309  }
310 
311  std::map<std::string, std::vector<Number> >::iterator it = vecs.begin();
312 
313  for (; it!=vecs.end(); ++it)
314  {
315  vtkSmartPointer<vtkDoubleArray> data = vtkSmartPointer<vtkDoubleArray>::New();
316  data->SetName(it->first.c_str());
317  libmesh_assert_equal_to (it->second.size(), es.get_mesh().n_nodes());
318  data->SetNumberOfValues(it->second.size());
319

321  {
322 #ifdef LIBMESH_USE_COMPLEX_NUMBERS

325  << std::endl);
326  data->SetValue(i, it->second[i].real());
327 #else
328  data->SetValue(i, it->second[i]);
329 #endif
330 
331  }
332  grid->GetPointData()->AddArray(data);
333  }
334  }

398 void VTKIO::read (const std::string& name)
399 {
400  // This is a serial-only process for now;
401  // the Mesh should be read on processor 0 and
402  // broadcast later
403  libmesh_assert_equal_to (MeshOutput<MeshBase>::mesh().processor_id(), 0);
404 
405  // Keep track of what kinds of elements this file contains
406  elems_of_dimension.clear();
407  elems_of_dimension.resize(4, false);
408 
409 #ifndef LIBMESH_HAVE_VTK
410  libmesh_error_msg("Cannot read VTK file: " << name \
411  << "\nYou must have VTK installed and correctly configured to read VTK meshes.");
412 
413 #else
414  // Use a typedef, because these names are just crazy
415  typedef vtkSmartPointer<vtkXMLUnstructuredGridReader> MyReader;
416  MyReader reader = MyReader::New();
417 
418  // Pass the filename along to the reader
419  reader->SetFileName( name.c_str() );
420 
421  // Force reading
422  reader->Update();
423 
424  // read in the grid
425  _vtk_grid = reader->GetOutput();
426  // _vtk_grid->Update(); // FIXME: Necessary?
427 
428  // Get a reference to the mesh
429  MeshBase& mesh = MeshInput<MeshBase>::mesh();
430 
431  // Clear out any pre-existing data from the Mesh
432  mesh.clear();
433 
434  // Get the number of points from the _vtk_grid object
435  const unsigned int vtk_num_points = static_cast<unsigned int>(_vtk_grid->GetNumberOfPoints());
436 
437  // always numbered nicely??, so we can loop like this
438  // I'm pretty sure it is numbered nicely
439  for (unsigned int i=0; i<vtk_num_points; ++i)
440  {
441  // add to the id map
442  // and add the actual point
443  double * pnt = _vtk_grid->GetPoint(static_cast<vtkIdType>(i));
444  Point xyz(pnt[0], pnt[1], pnt[2]);
445  Node* newnode = mesh.add_point(xyz, i);
446 
447  // Add node to the nodes vector &
448  // tell the MeshData object the foreign node id.
449  if (this->_mesh_data != NULL)
450  this->_mesh_data->add_foreign_node_id (newnode, i);
451  }
452 
453  // Get the number of cells from the _vtk_grid object
454  const unsigned int vtk_num_cells = static_cast<unsigned int>(_vtk_grid->GetNumberOfCells());
455 
456  for (unsigned int i=0; i<vtk_num_cells; ++i)
457  {
458  vtkCell* cell = _vtk_grid->GetCell(i);
459  Elem* elem = NULL;
460  switch (cell->GetCellType())
461  {
462  case VTK_LINE:
463  elem = new Edge2;
464  break;
465  case VTK_QUADRATIC_EDGE:
466  elem = new Edge3;
467  break;
468  case VTK_TRIANGLE:
469  elem = new Tri3();
470  break;
471  case VTK_QUADRATIC_TRIANGLE:
472  elem = new Tri6();
473  break;
474  case VTK_QUAD:
475  elem = new Quad4();
476  break;
477  case VTK_QUADRATIC_QUAD:
478  elem = new Quad8();
479  break;
480 #if VTK_MAJOR_VERSION > 5 || (VTK_MAJOR_VERSION == 5 && VTK_MINOR_VERSION > 0)
481  case VTK_BIQUADRATIC_QUAD:
482  elem = new Quad9();
483  break;
484 #endif
485  case VTK_TETRA:
486  elem = new Tet4();
487  break;
488  case VTK_QUADRATIC_TETRA:
489  elem = new Tet10();
490  break;
491  case VTK_WEDGE:
492  elem = new Prism6();
493  break;
494  case VTK_QUADRATIC_WEDGE:
495  elem = new Prism15();
496  break;
497  case VTK_BIQUADRATIC_QUADRATIC_WEDGE:
498  elem = new Prism18();
499  break;
500  case VTK_HEXAHEDRON:
501  elem = new Hex8();
502  break;
503  case VTK_QUADRATIC_HEXAHEDRON:
504  elem = new Hex20();
505  break;
506  case VTK_TRIQUADRATIC_HEXAHEDRON:
507  elem = new Hex27();
508  break;
509  case VTK_PYRAMID:
510  elem = new Pyramid5();
511  break;
512  default:
513  libmesh_error_msg("Element type not implemented in vtkinterface " << cell->GetCellType());
514  }
515 
516  // get the straightforward numbering from the VTK cells
517  for (unsigned int j=0; j<elem->n_nodes(); ++j)
518  elem->set_node(j) =
519  mesh.node_ptr(cast_int<dof_id_type>(cell->GetPointId(j)));
520 
521  // then get the connectivity
522  std::vector<dof_id_type> conn;
523  elem->connectivity(0, VTK, conn);
524 
525  // then reshuffle the nodes according to the connectivity, this
526  // two-time-assign would evade the definition of the vtk_mapping
527  for (unsigned int j=0; j<conn.size(); ++j)
528  elem->set_node(j) = mesh.node_ptr(conn[j]);
529 
530  elem->set_id(i);
531 
532  elems_of_dimension[elem->dim()] = true;
533 
534  mesh.add_elem(elem);
535  } // end loop over VTK cells
536 
537  // Set the mesh dimension to the largest encountered for an element
538  for (unsigned char i=0; i!=4; ++i)
539  if (elems_of_dimension[i])
540  mesh.set_mesh_dimension(i);
541 
542 #if LIBMESH_DIM < 3
543  if (mesh.mesh_dimension() > LIBMESH_DIM)
544  libmesh_error_msg("Cannot open dimension " \
545  << mesh.mesh_dimension() \
546  << " mesh file when configured without " \
547  << mesh.mesh_dimension() \
548  << "D support.");
549 #endif
550 
551 #endif // LIBMESH_HAVE_VTK
552 }

556 void VTKIO::write_nodal_data (const std::string& fname,
557 #ifdef LIBMESH_HAVE_VTK
558  const std::vector<Number>& soln,
559  const std::vector<std::string>& names
560 #else
561  const std::vector<Number>&,
562  const std::vector<std::string>&
563 #endif
564  )
565 {
566 #ifndef LIBMESH_HAVE_VTK
567 
568  libmesh_error_msg("Cannot write VTK file: " << fname \
569  << "\nYou must have VTK installed and correctly configured to read VTK meshes.");
570 
571 #else
572 
573  const MeshBase & mesh = MeshOutput<MeshBase>::mesh();
574 
575  // Is this really important? If so, it should be more than an assert...
576  // libmesh_assert(fname.substr(fname.rfind("."), fname.size()) == ".pvtu");
577 
578  // we only use Unstructured grids
579  _vtk_grid = vtkUnstructuredGrid::New();
580  vtkSmartPointer<vtkXMLPUnstructuredGridWriter> writer = vtkSmartPointer<vtkXMLPUnstructuredGridWriter>::New();
581 
582  // add nodes to the grid and update _local_node_map
583  _local_node_map.clear();
584  this->nodes_to_vtk();
585 
586  // add cells to the grid
587  this->cells_to_vtk();
588 
589  // add nodal solutions to the grid, if solutions are given
590  if (names.size() > 0)
591  {
592  std::size_t num_vars = names.size();
593  dof_id_type num_nodes = mesh.n_nodes();
594 
595  for (std::size_t variable=0; variable<num_vars; ++variable)
596  {
597  vtkSmartPointer<vtkDoubleArray> data = vtkSmartPointer<vtkDoubleArray>::New();
598  data->SetName(names[variable].c_str());
599 
600  // number of local and ghost nodes
601  data->SetNumberOfValues(_local_node_map.size());
602 
603  // loop over all nodes and get the solution for the current
604  // variable, if the node is in the current partition
605  for (dof_id_type k=0; k<num_nodes; ++k)
606  {
607  if (_local_node_map.find(k) == _local_node_map.end())
608  continue; // not a local node
609 
610  if (!soln.empty())
611  {
612 #ifdef LIBMESH_USE_COMPLEX_NUMBERS
613  libmesh_do_once (libMesh::err << "Only writing the real part for complex numbers!\n"
614  << "if you need this support contact " << LIBMESH_PACKAGE_BUGREPORT
615  << std::endl);
616  data->SetValue(_local_node_map[k], soln[k*num_vars + variable].real());
617 #else
618  data->SetValue(_local_node_map[k], soln[k*num_vars + variable]);
619 #endif
620  }
621  else
622  {
623  data->SetValue(_local_node_map[k], 0);
624  }
625  }
626  _vtk_grid->GetPointData()->AddArray(data);
627  }
628  }
629 
630  // Tell the writer how many partitions exist and on which processor
631  // we are currently
632  writer->SetNumberOfPieces(MeshOutput<MeshBase>::mesh().n_processors());
633  writer->SetStartPiece(MeshOutput<MeshBase>::mesh().processor_id());
634  writer->SetEndPiece(MeshOutput<MeshBase>::mesh().processor_id());
635 
636  // partitions overlap by one node
637  // FIXME: According to this document
638  // http://paraview.org/Wiki/images/5/51/SC07_tut107_ParaView_Handouts.pdf
639  // the ghosts are cells rather than nodes.
