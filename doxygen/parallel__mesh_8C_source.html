133 void ParallelMesh::update_parallel_id_counts()
134 {
135  // This function must be run on all processors at once
136  parallel_object_only();
137 
138  _n_elem = this->parallel_n_elem();
139  _n_nodes = this->parallel_n_nodes();
140  _max_node_id = this->parallel_max_node_id();
141  _max_elem_id = this->parallel_max_elem_id();
142 
143  if (_next_free_unpartitioned_elem_id < _max_elem_id)
144  _next_free_unpartitioned_elem_id =
145  ((_max_elem_id-1) / (this->n_processors() + 1) + 1) *
146  (this->n_processors() + 1) + this->n_processors();
147  if (_next_free_local_elem_id < _max_elem_id)
148  _next_free_local_elem_id =
149  ((_max_elem_id + this->n_processors() - 1) / (this->n_processors() + 1) + 1) *
150  (this->n_processors() + 1) + this->processor_id();
151 
152  if (_next_free_unpartitioned_node_id < _max_node_id)
153  _next_free_unpartitioned_node_id =
154  ((_max_node_id-1) / (this->n_processors() + 1) + 1) *

359 Elem* ParallelMesh::add_elem (Elem *e)
360 {
361  // Don't try to add NULLs!
362  libmesh_assert(e);
363 
364  // Trying to add an existing element is a no-op
365  if (e->valid_id() && _elements[e->id()] == e)
366  return e;
367 
368  const processor_id_type elem_procid = e->processor_id();
369 
370  if (!e->valid_id())
371  {
372  // We should only be creating new ids past the end of the range
373  // of existing ids
374  libmesh_assert_greater_equal(_next_free_unpartitioned_elem_id,
375  _max_elem_id);
376  libmesh_assert_greater_equal(_next_free_local_elem_id, _max_elem_id);
377 
378  // Use the unpartitioned ids for unpartitioned elems,
379  // in serial, and temporarily for ghost elems
380  dof_id_type *next_id = &_next_free_unpartitioned_elem_id;
381  if (elem_procid == this->processor_id() &&
382  !this->is_serial())
383  next_id = &_next_free_local_elem_id;
384  e->set_id (*next_id);
385  }
386 
387  {
388  // Advance next_ids up high enough that each is pointing to an
389  // unused id and any subsequent increments will still point us
390  // to unused ids
391  _max_elem_id = std::max(_max_elem_id,
392  static_cast<dof_id_type>(e->id()+1));
393 
394  if (_next_free_unpartitioned_elem_id < _max_elem_id)
395  _next_free_unpartitioned_elem_id =
396  ((_max_elem_id-1) / (this->n_processors() + 1) + 1) *
397  (this->n_processors() + 1) + this->n_processors();
398  if (_next_free_local_elem_id < _max_elem_id)
399  _next_free_local_elem_id =
400  ((_max_elem_id + this->n_processors() - 1) / (this->n_processors() + 1) + 1) *
401  (this->n_processors() + 1) + this->processor_id();
402 
403 #ifndef NDEBUG
404  // We need a const mapvector so we don't inadvertently create
405  // NULL entries when testing for non-NULL ones
406  const mapvector<Elem*,dof_id_type>& const_elements = _elements;
407 #endif
408  libmesh_assert(!const_elements[_next_free_unpartitioned_elem_id]);
409  libmesh_assert(!const_elements[_next_free_local_elem_id]);
410  }
411 
412  // Don't try to overwrite existing elems
413  libmesh_assert (!_elements[e->id()]);
414 
415  _elements[e->id()] = e;
416 
417  // Try to make the cached elem data more accurate
418  if (elem_procid == this->processor_id() ||
419  elem_procid == DofObject::invalid_processor_id)
420  _n_elem++;
421 
422 #ifdef LIBMESH_ENABLE_UNIQUE_ID
423  if (!e->valid_unique_id() && processor_id() == e->processor_id())
424  {
425  e->set_unique_id() = _next_unique_id;
426  _next_unique_id += this->n_processors();
427  }
428 #endif
429 
430  // Unpartitioned elems should be added on every processor
431  // And shouldn't be added in the same batch as ghost elems
432  // But we might be just adding on processor 0 to
433  // broadcast later
434  // #ifdef DEBUG
435  // if (elem_procid == DofObject::invalid_processor_id)
436  // {
437  // dof_id_type elem_id = e->id();
438  // this->comm().max(elem_id);
439  // libmesh_assert_equal_to (elem_id, e->id());
440  // }
441  // #endif
442 
443  return e;
444 }

659 void ParallelMesh::renumber_node(const dof_id_type old_id,
660  const dof_id_type new_id)
661 {
662  Node *nd = _nodes[old_id];
663  libmesh_assert (nd);
664  libmesh_assert_equal_to (nd->id(), old_id);
665 
666  nd->set_id(new_id);
667 
668  // If we have nodes shipped to this processor for NodeConstraints
669  // use, then those nodes will exist in _nodes, but may not be
670  // locatable via a TopologyMap due to the insufficiency of elements
671  // connecting to them. If local refinement then wants to create a
672  // *new* node in the same location, it will initally get a temporary
673  // id, and then make_node_ids_parallel_consistent() will try to move
674  // it to the canonical id. We need to account for this case to

686  libmesh_assert (!_nodes[new_id]);
687 #endif

880 template <typename T>
881 dof_id_type ParallelMesh::renumber_dof_objects
882 (mapvector<T*,dof_id_type> &objects)
883 {
884  // This function must be run on all processors at once
885  parallel_object_only();
886 
887  typedef typename mapvector<T*,dof_id_type>::veclike_iterator object_iterator;
888 
889  // In parallel we may not know what objects other processors have.
890  // Start by figuring out how many
891  dof_id_type unpartitioned_objects = 0;
892 
893  std::vector<dof_id_type>
894  ghost_objects_from_proc(this->n_processors(), 0);
895 
896  object_iterator it = objects.begin();
897  object_iterator end = objects.end();
898 
899  for (; it != end;)
900  {
901  T *obj = *it;
902 
903  // Remove any NULL container entries while we're here,
904  // being careful not to invalidate our iterator
905  if (!*it)
906  objects.erase(it++);
907  else
908  {
909  processor_id_type obj_procid = obj->processor_id();
910  if (obj_procid == DofObject::invalid_processor_id)
911  unpartitioned_objects++;
912  else
913  ghost_objects_from_proc[obj_procid]++;
914  ++it;
915  }
916  }
917 
918  std::vector<dof_id_type> objects_on_proc(this->n_processors(), 0);
919  this->comm().allgather(ghost_objects_from_proc[this->processor_id()],
920  objects_on_proc);
921 
922 #ifndef NDEBUG
923  libmesh_assert(this->comm().verify(unpartitioned_objects));
924  for (processor_id_type p=0; p != this->n_processors(); ++p)
925  libmesh_assert_less_equal (ghost_objects_from_proc[p], objects_on_proc[p]);
926 #endif
927 
928  // We'll renumber objects in blocks by processor id
929  std::vector<dof_id_type> first_object_on_proc(this->n_processors());
930  for (processor_id_type i=1; i != this->n_processors(); ++i)
931  first_object_on_proc[i] = first_object_on_proc[i-1] +
932  objects_on_proc[i-1];
933  dof_id_type next_id = first_object_on_proc[this->processor_id()];
934  dof_id_type first_free_id =
935  first_object_on_proc[this->n_processors()-1] +
936  objects_on_proc[this->n_processors()-1] +
937  unpartitioned_objects;
938 
939  // First set new local object ids and build request sets
940  // for non-local object ids
941 
942  // Request sets to send to each processor
943  std::vector<std::vector<dof_id_type> >
944  requested_ids(this->n_processors());
945 
946 #ifdef LIBMESH_ENABLE_UNIQUE_ID
947  std::vector<std::vector<unique_id_type> >
948  requested_unique_ids(this->n_processors());
949 #endif
950 
951  // We know how many objects live on each processor, so reseve() space for
952  // each.
953  for (processor_id_type p=0; p != this->n_processors(); ++p)
954  if (p != this->processor_id())
955  {
956  requested_ids[p].reserve(ghost_objects_from_proc[p]);
957 #ifdef LIBMESH_ENABLE_UNIQUE_ID
958  requested_unique_ids[p].reserve(ghost_objects_from_proc[p]);
959 #endif
960  }
961 
962  end = objects.end();
963  for (it = objects.begin(); it != end; ++it)
964  {
965  T *obj = *it;
966  if (obj->processor_id() == this->processor_id())
967  obj->set_id(next_id++);
968  else if (obj->processor_id() != DofObject::invalid_processor_id)
969  {
970  requested_ids[obj->processor_id()].push_back(obj->id());
971 #ifdef LIBMESH_ENABLE_UNIQUE_ID
972  // It's possible to have an invalid id for dofs not owned by this process.
973  // We'll assert that they match on the receiving end.
974  requested_unique_ids[obj->processor_id()].push_back(obj->valid_unique_id() ? obj-> unique_id() : DofObject::invalid_unique_id);
975 #endif
976  }
977  }
978 
979  // Next set ghost object ids from other processors
980  if (this->n_processors() > 1)
981  {
982  for (processor_id_type p=1; p != this->n_processors(); ++p)
983  {
984  // Trade my requests with processor procup and procdown
985  processor_id_type procup = cast_int<processor_id_type>
986  ((this->processor_id() + p) % this->n_processors());
987  processor_id_type procdown = cast_int<processor_id_type>
988  ((this->n_processors() + this->processor_id() - p) %
989  this->n_processors());
990  std::vector<dof_id_type> request_to_fill;
991  this->comm().send_receive(procup, requested_ids[procup],
992  procdown, request_to_fill);
993 
994 #ifdef LIBMESH_ENABLE_UNIQUE_ID
995  std::vector<unique_id_type> unique_request_to_fill;
996  this->comm().send_receive(procup, requested_unique_ids[procup],
997  procdown, unique_request_to_fill);
998  std::vector<unique_id_type> new_unique_ids(unique_request_to_fill.size());
999 #endif
1000 
1001  // Fill those requests
1002  std::vector<dof_id_type> new_ids(request_to_fill.size());
1003  for (std::size_t i=0; i != request_to_fill.size(); ++i)
1004  {
1005  T *obj = objects[request_to_fill[i]];
1006  libmesh_assert(obj);
1007  libmesh_assert_equal_to (obj->processor_id(), this->processor_id());
1008  new_ids[i] = obj->id();
1009 #ifdef LIBMESH_ENABLE_UNIQUE_ID
1010  new_unique_ids[i] = obj->valid_unique_id() ? obj->unique_id() : DofObject::invalid_unique_id;
1011 #endif
1012 
1013  libmesh_assert_greater_equal (new_ids[i],
1014  first_object_on_proc[this->processor_id()]);
1015  libmesh_assert_less (new_ids[i],
1016  first_object_on_proc[this->processor_id()] +
1017  objects_on_proc[this->processor_id()]);
1018  }
1019 
1020  // Trade back the results
1021  std::vector<dof_id_type> filled_request;
1022  this->comm().send_receive(procdown, new_ids,
1023  procup, filled_request);
1024 
1025 #ifdef LIBMESH_ENABLE_UNIQUE_ID
1026  std::vector<unique_id_type> unique_filled_request;
1027  this->comm().send_receive(procdown, new_unique_ids,
1028  procup, unique_filled_request);
1029 #endif
1030 
1031  // And copy the id changes we've now been informed of
1032  for (std::size_t i=0; i != filled_request.size(); ++i)
1033  {
1034  T *obj = objects[requested_ids[procup][i]];
1035  libmesh_assert (obj);
1036  libmesh_assert_equal_to (obj->processor_id(), procup);
1037  libmesh_assert_greater_equal (filled_request[i],
1038  first_object_on_proc[procup]);
1039  libmesh_assert_less (filled_request[i],
1040  first_object_on_proc[procup] +
1041  objects_on_proc[procup]);
1042  obj->set_id(filled_request[i]);
1043 
1044 #ifdef LIBMESH_ENABLE_UNIQUE_ID
1045  if (!obj->valid_unique_id() && unique_filled_request[i] != DofObject::invalid_unique_id)
1046  obj->set_unique_id() = unique_filled_request[i];
1047 #endif
1048  }
1049  }
1050  }
1051 
1052  // Next set unpartitioned object ids
1053  next_id = 0;
1054  for (processor_id_type i=0; i != this->n_processors(); ++i)
1055  next_id += objects_on_proc[i];
1056  for (it = objects.begin(); it != end; ++it)
1057  {
1058  T *obj = *it;
1059  if (obj->processor_id() == DofObject::invalid_processor_id)
1060  obj->set_id(next_id++);
1061  }
1062 
1063  // Finally shuffle around objects so that container indices
1064  // match ids
1065  end = objects.end();
1066  for (it = objects.begin(); it != end;)
1067  {
1068  T *obj = *it;
1069  if (obj) // don't try shuffling already-NULL entries
1070  {
1071  T *next = objects[obj->id()];
1072  // If we have to move this object
1073  if (next != obj)
1074  {
1075  // NULL out its original position for now
1076  // (our shuffling may put another object there shortly)
1077  *it = NULL;
1078 
1079  // There may already be another object with this id that
1080  // needs to be moved itself
1081  while (next)
1082  {
1083  // We shouldn't be trying to give two objects the
1084  // same id
1085  libmesh_assert_not_equal_to (next->id(), obj->id());
1086  objects[obj->id()] = obj;
1087  obj = next;
1088  next = objects[obj->id()];
1089  }
1090  objects[obj->id()] = obj;
1091  }
1092  }
1093  // Remove any container entries that were left as NULL,
1094  // being careful not to invalidate our iterator
1095  if (!*it)
1096  objects.erase(it++);
1097  else
1098  ++it;
1099  }
1100 
1101  return first_free_id;
1102 }
