502 template <typename Range, typename Body>
503 inline
504 void parallel_for (const Range &range, const Body &body)
505 {
506  Threads::BoolAcquire b(Threads::in_threads);
507 
508 #ifdef LIBMESH_ENABLE_PERFORMANCE_LOGGING
509  const bool logging_was_enabled = libMesh::perflog.logging_enabled();
510 
511  if (libMesh::n_threads() > 1)
512  libMesh::perflog.disable_logging();
513 #endif
514 
515  unsigned int n_threads = num_pthreads(range);
516 
517  std::vector<Range *> ranges(n_threads);
518  std::vector<RangeBody<const Range, const Body> > range_bodies(n_threads);
519  std::vector<pthread_t> threads(n_threads);
520 
521  // Create the ranges for each thread
522  unsigned int range_size = range.size() / n_threads;
523 
524  typename Range::const_iterator current_beginning = range.begin();
525 
526  for(unsigned int i=0; i<n_threads; i++)
527  {
528  unsigned int this_range_size = range_size;
529 
530  if(i+1 == n_threads)
531  this_range_size += range.size() % n_threads; // Give the last one the remaining work to do
532 
533  ranges[i] = new Range(range, current_beginning, current_beginning + this_range_size);
534 
535  current_beginning = current_beginning + this_range_size;
536  }
537 
538  // Create the RangeBody arguments
539  for(unsigned int i=0; i<n_threads; i++)
540  {
541  range_bodies[i].range = ranges[i];
542  range_bodies[i].body = &body;
543  }
544 
545  // Create the threads
546 #pragma omp parallel for schedule (static)
547  for(unsigned int i=0; i<n_threads; i++)
548  {
549 #if LIBMESH_HAVE_OPENMP
550  run_body<Range, Body>((void*)&range_bodies[i]);
551 #else // Just use Pthreads
552  spin_mutex::scoped_lock lock(_pthread_unique_id_mutex);
553  pthread_create(&threads[i], NULL, &run_body<Range, Body>, (void*)&range_bodies[i]);
554  _pthread_unique_ids[threads[i]] = i;
555 #endif
556  }
557 
558 #if !LIBMESH_HAVE_OPENMP
559  // Wait for them to finish
560 
561  // The use of 'int' instead of unsigned for the iteration variable
562  // is deliberate here. This is an OpenMP loop, and some older
563  // compilers warn when you don't use int for the loop index. The
564  // reason has to do with signed vs. unsigned integer overflow
565  // behavior and optimization.
566  // http://blog.llvm.org/2011/05/what-every-c-programmer-should-know.html
567  for (int i=0; i<static_cast<int>(n_threads); i++)
568  {
569  pthread_join(threads[i], NULL);
570  spin_mutex::scoped_lock lock(_pthread_unique_id_mutex);
571  _pthread_unique_ids.erase(threads[i]);
572  }
573 #endif
574 
575  // Clean up
576  for(unsigned int i=0; i<n_threads; i++)
577  delete ranges[i];
578 
579 #ifdef LIBMESH_ENABLE_PERFORMANCE_LOGGING
580  if (libMesh::n_threads() > 1 && logging_was_enabled)
581  libMesh::perflog.enable_logging();
582 #endif
583 }

602 template <typename Range, typename Body>
603 inline
604 void parallel_reduce (const Range &range, Body &body)
605 {
606  Threads::BoolAcquire b(Threads::in_threads);
607 
608 #ifdef LIBMESH_ENABLE_PERFORMANCE_LOGGING
609  const bool logging_was_enabled = libMesh::perflog.logging_enabled();
610 
611  if (libMesh::n_threads() > 1)
612  libMesh::perflog.disable_logging();
613 #endif
614 
615  unsigned int n_threads = num_pthreads(range);
616 
617  std::vector<Range *> ranges(n_threads);
618  std::vector<Body *> bodies(n_threads);
619  std::vector<RangeBody<Range, Body> > range_bodies(n_threads);
620 
621  // Create copies of the body for each thread
622  bodies[0] = &body; // Use the original body for the first one
623  for(unsigned int i=1; i<n_threads; i++)
624  bodies[i] = new Body(body, Threads::split());
625 
626  // Create the ranges for each thread
627  unsigned int range_size = range.size() / n_threads;
628 
629  typename Range::const_iterator current_beginning = range.begin();
630 
631  for(unsigned int i=0; i<n_threads; i++)
632  {
633  unsigned int this_range_size = range_size;
634 
635  if(i+1 == n_threads)
636  this_range_size += range.size() % n_threads; // Give the last one the remaining work to do
637 
638  ranges[i] = new Range(range, current_beginning, current_beginning + this_range_size);
639 
640  current_beginning = current_beginning + this_range_size;
641  }
642 
643  // Create the RangeBody arguments
644  for(unsigned int i=0; i<n_threads; i++)
645  {
646  range_bodies[i].range = ranges[i];
647  range_bodies[i].body = bodies[i];
648  }
649 
650  // Create the threads
651  std::vector<pthread_t> threads(n_threads);
652 
653 #pragma omp parallel for schedule (static)
654  // The use of 'int' instead of unsigned for the iteration variable
655  // is deliberate here. This is an OpenMP loop, and some older
656  // compilers warn when you don't use int for the loop index. The
657  // reason has to do with signed vs. unsigned integer overflow
658  // behavior and optimization.
659  // http://blog.llvm.org/2011/05/what-every-c-programmer-should-know.html
660  for (int i=0; i<static_cast<int>(n_threads); i++)
661  {
662 #if LIBMESH_HAVE_OPENMP
663  run_body<Range, Body>((void*)&range_bodies[i]);
664 #else // Just use Pthreads
665  spin_mutex::scoped_lock lock(_pthread_unique_id_mutex);
666  pthread_create(&threads[i], NULL, &run_body<Range, Body>, (void*)&range_bodies[i]);
667  _pthread_unique_ids[threads[i]] = i;
668 #endif
669  }
670 
671 #if !LIBMESH_HAVE_OPENMP
672  // Wait for them to finish
673  for(unsigned int i=0; i<n_threads; i++)
674  {
675  pthread_join(threads[i], NULL);
676  spin_mutex::scoped_lock lock(_pthread_unique_id_mutex);
677  _pthread_unique_ids.erase(threads[i]);
678  }
679 #endif
680 
681  // Join them all down to the original Body
682  for(unsigned int i=n_threads-1; i != 0; i--)
683  bodies[i-1]->join(*bodies[i]);
684 
685  // Clean up
686  for(unsigned int i=1; i<n_threads; i++)
687  delete bodies[i];
688  for(unsigned int i=0; i<n_threads; i++)
689  delete ranges[i];
690 
691 #ifdef LIBMESH_ENABLE_PERFORMANCE_LOGGING
692  if (libMesh::n_threads() > 1 && logging_was_enabled)
693  libMesh::perflog.enable_logging();
694 #endif
695 }
