80 void PointLocatorList::init ()
81 {
82  libmesh_assert (!this->_list);
83 
84  if (this->_initialized)
85  libMesh::err << "ERROR: Already initialized! Will ignore this call..." << std::endl;
86 
87  else
88  {
89  if (this->_master == NULL)
90  {
91  START_LOG("init(no master)", "PointLocatorList");
92 
93  // We are the master, so we have to build the list.
94  // First create it, then get a handy reference, and
95  // then try to speed up by reserving space...
96  this->_list = new std::vector<std::pair<Point, const Elem *> >;
97  std::vector<std::pair<Point, const Elem *> >& my_list = *(this->_list);
98 
99  my_list.clear();
100  my_list.reserve(this->_mesh.n_active_elem());
101 
102  // fill our list with the centroids and element
103  // pointers of the mesh. For this use the handy
104  // element iterators.
105  MeshBase::const_element_iterator el = _mesh.active_elements_begin();
106  const MeshBase::const_element_iterator end = _mesh.active_elements_end();
107 
108  for (; el!=end; ++el)
109  my_list.push_back(std::make_pair((*el)->centroid(), *el));
110 
111  STOP_LOG("init(no master)", "PointLocatorList");
112  }
113 
114  else
115  {
116  // We are _not_ the master. Let our _list point to
117  // the master's list. But for this we first transform
118  // the master in a state for which we are friends
119  // (this should also beware of a bad master pointer?).
120  // And make sure the master @e has a list!
121  const PointLocatorList* my_master =
122  cast_ptr<const PointLocatorList*>(this->_master);
123 
124  if (my_master->initialized())
125  this->_list = my_master->_list;
126  else
127  libmesh_error_msg("ERROR: Initialize master first, then servants!");
128  }
129  }
130 
131  // ready for take-off
132  this->_initialized = true;
133 }

137 const Elem* PointLocatorList::operator() (const Point& p, const std::set<subdomain_id_type> *allowed_subdomains) const
138 {
139  libmesh_assert (this->_initialized);
140 
141  START_LOG("operator()", "PointLocatorList");
142 
143  // Ask the list. This is quite expensive, since
144  // we loop through the whole list to try to find
145  // the @e nearest element.
146  // However, there is not much else to do: when
147  // we would use bounding boxes like in a tree,
148  // it may happen that a surface element is just
149  // in plane with a bounding box face, and quite
150  // close to it. But when a point comes, this
151  // point may belong to the bounding box (where the
152  // coplanar element does @e not belong to). Then
153  // we would search through the elements in this
154  // bounding box, while the other bounding box'es
155  // element is closer, but we simply don't consider
156  // it!
157  //
158  // We _can_, however, use size_sq() instead of size()
159  // here to avoid repeated calls to std::sqrt(), which is
160  // pretty expensive.
161  {
162  std::vector<std::pair<Point, const Elem *> >& my_list = *(this->_list);
163 
164  Real last_distance_sq = Point(my_list[0].first -p).size_sq();
165  const Elem * last_elem = NULL;
166  const std::size_t max_index = my_list.size();
167 
168  for (std::size_t n=1; n<max_index; n++)
169  {
170  // Only consider elements in the allowed_subdomains list, if it exists
171  if (!allowed_subdomains ||
172  allowed_subdomains->count(my_list[n].second->subdomain_id()))
173  {
174  const Real current_distance_sq = Point(my_list[n].first -p).size_sq();
175 
176  if (current_distance_sq < last_distance_sq)
177  {
178  last_distance_sq = current_distance_sq;
179  last_elem = my_list[n].second;
180  }
181  }
182  }
183 
184  // If we found an element, it should be active
185  libmesh_assert (!last_elem || last_elem->active());
186 
187  // If we found an element and have a restriction list, they better match
188  libmesh_assert (!last_elem || !allowed_subdomains || allowed_subdomains->count(last_elem->subdomain_id()));
189 
190  STOP_LOG("operator()", "PointLocatorList");
191 
192  // return the element
193  return last_elem;
194  }
195 }
