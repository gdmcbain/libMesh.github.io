187 void Quad4::connectivity(const unsigned int libmesh_dbg_var(sf),
188  const IOPackage iop,
189  std::vector<dof_id_type>& conn) const
190 {
191  libmesh_assert_less (sf, this->n_sub_elem());
192  libmesh_assert_not_equal_to (iop, INVALID_IO_PACKAGE);
193 
194  // Create storage.
195  conn.resize(4);
196 
197  switch (iop)
198  {
199  case TECPLOT:
200  {
201  conn[0] = this->node(0)+1;
202  conn[1] = this->node(1)+1;
203  conn[2] = this->node(2)+1;
204  conn[3] = this->node(3)+1;
205  return;
206  }
207 
208  case VTK:
209  {
210  conn[0] = this->node(0);
211  conn[1] = this->node(1);
212  conn[2] = this->node(2);
213  conn[3] = this->node(3);
214  return;
215  }
216 
217  default:
218  libmesh_error_msg("Unsupported IO package " << iop);
219  }

224 Real Quad4::volume () const
225 {
226  // The A,B,C,D naming scheme here corresponds exactly to the
227  // libmesh counter-clockwise numbering scheme.
228 
229  // 3 2 D C
230  // QUAD4: o-----------o o-----------o
231  // | | | |
232  // | | | |
233  // | | | |
234  // | | | |
235  // | | | |
236  // o-----------o o-----------o
237  // 0 1 A B
238 
239  // Vector pointing from A to C
240  Point AC ( this->point(2) - this->point(0) );
241 
242  // Vector pointing from A to B
243  Point AB ( this->point(1) - this->point(0) );
244 
245  // Vector pointing from A to D
246  Point AD ( this->point(3) - this->point(0) );
247 
248  // The diagonal vector minus the side vectors
249  Point AC_AB_AD (AC - AB - AD);
250 
251  // Check for quick return for planar QUAD4. This will
252  // be the most common case, occuring for all purely 2D meshes.
253  if (AC_AB_AD == Point(0.,0.,0.))
254  return AB.cross(AD).size();
255 
256  else
257  {
258  // Use 2x2 quadrature to approximate the surface area. (The
259  // true integral is too difficult to compute analytically.) The
260  // accuracy here is exactly the same as would be obtained via a
261  // call to Elem::volume(), however it is a bit more optimized to
262  // do it this way. The technique used is to integrate the magnitude
263  // of the normal vector over the whole area. See for example,
264  //
265  // Y. Zhang, C. Bajaj, G. Xu. Surface Smoothing and Quality
266  // Improvement of Quadrilateral/Hexahedral Meshes with Geometric
267  // Flow. The special issue of the Journal Communications in
268  // Numerical Methods in Engineering (CNME), submitted as an
269  // invited paper, 2006.
270  // http://www.ices.utexas.edu/~jessica/paper/quadhexgf/quadhex_geomflow_CNM.pdf
271 
272  // 4-point rule
273  const Real q[2] = {0.5 - std::sqrt(3.) / 6.,
274  0.5 + std::sqrt(3.) / 6.};
275 
276  Real vol=0.;
277  for (unsigned int i=0; i<2; ++i)
278  for (unsigned int j=0; j<2; ++j)
279  vol += (AB + q[i]*AC_AB_AD).cross(AD + q[j]*AC_AB_AD).size();
280 
281  return 0.25*vol;
282  }
283 }
