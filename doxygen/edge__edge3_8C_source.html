170 Real Edge3::volume () const
171 {
172  // Finding the (exact) length of a general quadratic element
173  // is a surprisingly complicated formula.
174  Point A = this->point(0) + this->point(1) - 2*this->point(2);
175  Point B = (this->point(1) - this->point(0))/2;
176 
177  const Real a = A.size_sq();
178  const Real b = 2.*(A*B);
179  const Real c = B.size_sq();
180 
181  // Degenerate straight line case
182  if (a < TOLERANCE*TOLERANCE*TOLERANCE)
183  return (this->point(1) - this->point(0)).size();
184 
185  const Real ba=b/a;
186  const Real ca=c/a;
187 
188  libmesh_assert (1.-ba+ca>0.);
189 
190  const Real s1 = std::sqrt(1. - ba + ca);
191  const Real s2 = std::sqrt(1. + ba + ca);
192 
193  return 0.5*std::sqrt(a)*((1.-0.5*ba)*s1 +
194  (1.+0.5*ba)*s2 +
195  (ca - 0.25*ba*ba)*std::log( (1.-0.5*ba+s1)/(-1.-0.5*ba+s2) )
196  );
197 }
