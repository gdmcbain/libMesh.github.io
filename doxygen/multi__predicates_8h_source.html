48 // This class represents a generic combination of more than one predicate.
49 // It is meant to be derived from to actually be used.
50 template <typename T>
51 struct abstract_multi_predicate : multi_predicate
52 {
53  // virtual destructor.
54  virtual ~abstract_multi_predicate()
55  {
56  // Clean-up vector
57  for (unsigned int i=0; i<_predicates.size(); ++i)
58  delete _predicates[i];
59  }
60 
61  // operator= (perform deep copy of entries in _predicates vector
62  abstract_multi_predicate& operator=(const abstract_multi_predicate& rhs)
63  {
64  // First clear out the predicates vector
65  for (unsigned int i=0; i<_predicates.size(); ++i)
66  delete _predicates[i];
67 
68  // Now copy over the information from the rhs.
69  this->deep_copy(rhs);
70 
71  return *this;
72  }
73 
74  // operator() checks all the predicates in the vector.
75  virtual bool operator()(const T& it) const
76  {
77  for (unsigned int i=0; i<_predicates.size(); ++i)
78  {
79  const predicate<T>* pred = _predicates[i];
80 
81  libmesh_assert (pred);
82 
83  if ( ! (*pred)(it) )
84  return false;
85  }
86 
87  return true;
88  }
89 
90 protected:
91  // Do not instantiate the base class.
92  abstract_multi_predicate() {}
93 
94  // Copy constructor.
95  abstract_multi_predicate(const abstract_multi_predicate& rhs)
96  {
97  this->deep_copy(rhs);
98  }
99 
100  // The deep_copy function is used by both the op= and
101  // copy constructors. This function uses the default (empty)
102  // copy constructor for the predicate class.
103  void deep_copy(const abstract_multi_predicate& rhs)
104  {
105  for (unsigned int i=0; i<rhs._predicates.size(); ++i)
106  _predicates.push_back(rhs._predicates[i]->clone());
107  }
108 
109  // Predicates to be evaluated.
110  std::vector<predicate<T>*> _predicates;
111 };
