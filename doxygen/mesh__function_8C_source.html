97 void MeshFunction::init (const Trees::BuildType /*point_locator_build_type*/)
98 {
99  // are indices of the desired variable(s) provided?
100  libmesh_assert_greater (this->_system_vars.size(), 0);
101 
102  // Don't do twice...
103  if (this->_initialized)
104  {
105  libmesh_assert(this->_point_locator);
106  return;
107  }
108 
109  /*
110  * set up the PointLocator: either someone else
111  * is the master (go and get the address of his
112  * point locator) or this object is the master
113  * (build the point locator on our own).
114  */
115  if (this->_master != NULL)
116  {
117  // we aren't the master
118  const MeshFunction* master =
119  cast_ptr<const MeshFunction*>(this->_master);
120 
121  if (master->_point_locator == NULL)
122  libmesh_error_msg("ERROR: When the master-servant concept is used, the master has to be initialized first!");
123 
124  else
125  {
126  this->_point_locator = master->_point_locator;
127  }
128  }
129  else
130  {
131  // we are the master: build the point locator
132 
133  // constant reference to the other mesh
134  const MeshBase& mesh = this->_eqn_systems.get_mesh();
135 
136  // build the point locator. Only \p TREE version available
137  //AutoPtr<PointLocatorBase> ap (PointLocatorBase::build (TREE, mesh));
138  //this->_point_locator = ap.release();
139  // this->_point_locator = new PointLocatorTree (mesh, point_locator_build_type);
140  this->_point_locator = mesh.sub_point_locator().release();
141 
142  // Point locator no longer needs to be initialized.
143  // this->_point_locator->init();
144  }
145 
146 
147  // ready for use
148  this->_initialized = true;
149 }

213 void MeshFunction::operator() (const Point& p,
214  const Real,
215  DenseVector<Number>& output)
216 {
217  libmesh_assert (this->initialized());
218 
219  /* Ensure that in the case of a master mesh function, the
220  out-of-mesh mode is enabled either for both or for none. This is
221  important because the out-of-mesh mode is also communicated to
222  the point locator. Since this is time consuming, enable it only
223  in debug mode. */
224 #ifdef DEBUG
225  if (this->_master != NULL)
226  {
227  const MeshFunction* master =
228  cast_ptr<const MeshFunction*>(this->_master);
229  if(_out_of_mesh_mode!=master->_out_of_mesh_mode)
230  libmesh_error_msg("ERROR: If you use out-of-mesh-mode in connection with master mesh " \
231  << "functions, you must enable out-of-mesh mode for both the master and the slave mesh function.");
232  }
233 #endif
234 
235  // locate the point in the other mesh
236  const Elem* element = this->_point_locator->operator()(p);
237 
238  // If we have an element, but it's not a local element, then we
239  // either need to have a serialized vector or we need to find a
240  // local element sharing the same point.
241  if (element &&
242  (element->processor_id() != this->processor_id()) &&
243  _vector.type() != SERIAL)
244  {
245  // look for a local element containing the point
246  std::set<const Elem*> point_neighbors;
247  element->find_point_neighbors(p, point_neighbors);
248  element = NULL;
249  std::set<const Elem*>::const_iterator it = point_neighbors.begin();
250  const std::set<const Elem*>::const_iterator end = point_neighbors.end();
251  for (; it != end; ++it)
252  {
253  const Elem* elem = *it;
254  if (elem->processor_id() == this->processor_id())
255  {
256  element = elem;
257  break;
258  }
259  }
260  }
261 
262  if (!element)
263  {
264  output = _out_of_mesh_value;
265  }
266  else
267  {
268  // resize the output vector to the number of output values
269  // that the user told us
270  output.resize (cast_int<unsigned int>
271  (this->_system_vars.size()));
272 
273 
274  {
275  const unsigned int dim = this->_eqn_systems.get_mesh().mesh_dimension();
276 
277 
278  /*
279  * Get local coordinates to feed these into compute_data().
280  * Note that the fe_type can safely be used from the 0-variable,
281  * since the inverse mapping is the same for all FEFamilies
282  */
283  const Point mapped_point (FEInterface::inverse_map (dim,
284  this->_dof_map.variable_type(0),
285  element,
286  p));
287 
288 
289  // loop over all vars
290  for (unsigned int index=0; index < this->_system_vars.size(); index++)
291  {
292  /*
293  * the data for this variable
294  */
295  const unsigned int var = _system_vars[index];
296  const FEType& fe_type = this->_dof_map.variable_type(var);
297 
302  {
303  FEComputeData data (this->_eqn_systems, mapped_point);
304 
305  FEInterface::compute_data (dim, fe_type, element, data);
306 
307  // where the solution values for the var-th variable are stored
308  std::vector<dof_id_type> dof_indices;
309  this->_dof_map.dof_indices (element, dof_indices, var);
310 
311  // interpolate the solution
312  {
313  Number value = 0.;
314 
315  for (unsigned int i=0; i<dof_indices.size(); i++)
316  value += this->_vector(dof_indices[i]) * data.shape[i];
317 
318  output(index) = value;
319  }
320 
321  }
322 
323  // next variable
324  }
325  }
326  }
327 
328  // all done
329  return;
330 }

438  // all done
