98 void MeshFunction::init (const Trees::BuildType /*point_locator_build_type*/)
99 {
100  // are indices of the desired variable(s) provided?
101  libmesh_assert_greater (this->_system_vars.size(), 0);
102 
103  // Don't do twice...
104  if (this->_initialized)
105  {
106  libmesh_assert(this->_point_locator);
107  return;
108  }
109 
110  /*
111  * set up the PointLocator: either someone else
112  * is the master (go and get the address of his
113  * point locator) or this object is the master
114  * (build the point locator on our own).
115  */
116  if (this->_master != NULL)
117  {
118  // we aren't the master
119  const MeshFunction* master =
120  cast_ptr<const MeshFunction*>(this->_master);
121 
122  if (master->_point_locator == NULL)
123  libmesh_error_msg("ERROR: When the master-servant concept is used, the master has to be initialized first!");
124 
125  else
126  {
127  this->_point_locator = master->_point_locator;
128  }
129  }
130  else
131  {
132  // we are the master: build the point locator
133 
134  // constant reference to the other mesh
135  const MeshBase& mesh = this->_eqn_systems.get_mesh();
136 
137  // build the point locator. Only \p TREE version available
138  //UniquePtr<PointLocatorBase> ap (PointLocatorBase::build (TREE, mesh));
139  //this->_point_locator = ap.release();
140  // this->_point_locator = new PointLocatorTree (mesh, point_locator_build_type);
141  this->_point_locator = mesh.sub_point_locator().release();
142 
143  // Point locator no longer needs to be initialized.
144  // this->_point_locator->init();
145  }
146 
147 
148  // ready for use
149  this->_initialized = true;
150 }

219 void MeshFunction::operator() (const Point& p,
220  const Real,
221  DenseVector<Number>& output,
222  const std::set<subdomain_id_type>* subdomain_ids)
223 {
224  libmesh_assert (this->initialized());
225 
226  const Elem* element = this->find_element(p,subdomain_ids);
227 
228  if (!element)
229  {
230  output = _out_of_mesh_value;
231  }
232  else
233  {
234  // resize the output vector to the number of output values
235  // that the user told us
236  output.resize (cast_int<unsigned int>
237  (this->_system_vars.size()));
238 
239 
240  {
241  const unsigned int dim = element->dim();
242 
243 
244  /*
245  * Get local coordinates to feed these into compute_data().
246  * Note that the fe_type can safely be used from the 0-variable,
247  * since the inverse mapping is the same for all FEFamilies
248  */
249  const Point mapped_point (FEInterface::inverse_map (dim,
250  this->_dof_map.variable_type(0),
251  element,
252  p));
253 
254 
255  // loop over all vars
256  for (unsigned int index=0; index < this->_system_vars.size(); index++)
257  {
258  /*
259  * the data for this variable
260  */
261  const unsigned int var = _system_vars[index];
262  const FEType& fe_type = this->_dof_map.variable_type(var);
263 
268  {
269  FEComputeData data (this->_eqn_systems, mapped_point);
270 
271  FEInterface::compute_data (dim, fe_type, element, data);
272 
273  // where the solution values for the var-th variable are stored
274  std::vector<dof_id_type> dof_indices;
275  this->_dof_map.dof_indices (element, dof_indices, var);
276 
277  // interpolate the solution
278  {
279  Number value = 0.;
280 
281  for (unsigned int i=0; i<dof_indices.size(); i++)
282  value += this->_vector(dof_indices[i]) * data.shape[i];
283 
284  output(index) = value;
285  }
286 
287  }
288 
289  // next variable
290  }
291  }
292  }
293 
294  // all done
295  return;
296 }

300 void MeshFunction::gradient (const Point& p,
301  const Real,
302  std::vector<Gradient>& output,
303  const std::set<subdomain_id_type>* subdomain_ids)
304 {
305  libmesh_assert (this->initialized());
306 
307  const Elem* element = this->find_element(p,subdomain_ids);
308 
309  if (!element)
310  {
311  output.resize(0);
312  }
313  else
314  {
315  // resize the output vector to the number of output values
316  // that the user told us
317  output.resize (this->_system_vars.size());
318 
319 
320  {
321  const unsigned int dim = element->dim();
322 
323 
324  /*
325  * Get local coordinates to feed these into compute_data().
326  * Note that the fe_type can safely be used from the 0-variable,
327  * since the inverse mapping is the same for all FEFamilies
328  */
329  const Point mapped_point (FEInterface::inverse_map (dim,
330  this->_dof_map.variable_type(0),
331  element,
332  p));
333 
334  std::vector<Point> point_list (1, mapped_point);
335 
336  // loop over all vars
337  for (unsigned int index=0; index < this->_system_vars.size(); index++)
338  {
339  /*
340  * the data for this variable
341  */
342  const unsigned int var = _system_vars[index];
343  const FEType& fe_type = this->_dof_map.variable_type(var);
344 
345  UniquePtr<FEBase> point_fe (FEBase::build(dim, fe_type));
346  const std::vector<std::vector<RealGradient> >& dphi = point_fe->get_dphi();
347  point_fe->reinit(element, &point_list);
348 
349  // where the solution values for the var-th variable are stored
350  std::vector<dof_id_type> dof_indices;
351  this->_dof_map.dof_indices (element, dof_indices, var);
352 
353  // interpolate the solution
354  Gradient grad(0.);
355 
356  for (unsigned int i=0; i<dof_indices.size(); i++)
357  grad.add_scaled(dphi[i][0], this->_vector(dof_indices[i]));
358 
359  output[index] = grad;
360  }
361  }
362  }
363 
364  // all done
365  return;
366 }

441 const Elem* MeshFunction::find_element( const Point& p,
442  const std::set<subdomain_id_type>* subdomain_ids ) const
443 {
444  /* Ensure that in the case of a master mesh function, the
445  out-of-mesh mode is enabled either for both or for none. This is
446  important because the out-of-mesh mode is also communicated to
447  the point locator. Since this is time consuming, enable it only
448  in debug mode. */
449 #ifdef DEBUG
450  if (this->_master != NULL)
451  {
452  const MeshFunction* master =
453  cast_ptr<const MeshFunction*>(this->_master);
454  if(_out_of_mesh_mode!=master->_out_of_mesh_mode)
455  libmesh_error_msg("ERROR: If you use out-of-mesh-mode in connection with master mesh " \
456  << "functions, you must enable out-of-mesh mode for both the master and the slave mesh function.");
457  }
458 #endif
459 
460  // locate the point in the other mesh
461  const Elem* element = this->_point_locator->operator()(p,subdomain_ids);
462 
463  // If we have an element, but it's not a local element, then we
464  // either need to have a serialized vector or we need to find a
465  // local element sharing the same point.
466  if (element &&
467  (element->processor_id() != this->processor_id()) &&
468  _vector.type() != SERIAL)
469  {
470  // look for a local element containing the point
471  std::set<const Elem*> point_neighbors;
472  element->find_point_neighbors(p, point_neighbors);
473  element = NULL;
474  std::set<const Elem*>::const_iterator it = point_neighbors.begin();
475  const std::set<const Elem*>::const_iterator end = point_neighbors.end();
476  for (; it != end; ++it)
477  {
478  const Elem* elem = *it;
479  if (elem->processor_id() == this->processor_id())
480  {
481  element = elem;
482  break;
483  }
484  }
485  }
486 
487  return element;
488 }
