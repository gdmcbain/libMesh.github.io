248 void System::init_data ()
249 {
250  MeshBase &mesh = this->get_mesh();
251 
252  // Add all variable groups to our underlying DofMap
253  for (unsigned int vg=0; vg<this->n_variable_groups(); vg++)
254  _dof_map->add_variable_group(this->variable_group(vg));
255 
256  // Distribute the degrees of freedom on the mesh
257  _dof_map->distribute_dofs (mesh);
258 
259 #ifdef LIBMESH_ENABLE_CONSTRAINTS
260 
261  // Recreate any hanging node constraints
262  _dof_map->create_dof_constraints(mesh);
263 
264  // Apply any user-defined constraints
265  this->user_constrain();
266 
267  // Expand any recursive constraints
268  _dof_map->process_constraints(mesh);
269 
270 #endif
271 
272  // And clean up the send_list before we first use it
273  _dof_map->prepare_send_list();
274 
275  // Resize the solution conformal to the current mesh
276  solution->init (this->n_dofs(), this->n_local_dofs(), false, PARALLEL);
277 
278  // Resize the current_local_solution for the current mesh
279 #ifdef LIBMESH_ENABLE_GHOSTED
280  current_local_solution->init (this->n_dofs(), this->n_local_dofs(),
281  _dof_map->get_send_list(), false,
282  GHOSTED);
283 #else
284  current_local_solution->init (this->n_dofs(), false, SERIAL);
285 #endif
286 
287  // from now on, no chance to add additional vectors
288  _can_add_vectors = false;
289 
290  // initialize & zero other vectors, if necessary
291  for (vectors_iterator pos = _vectors.begin(); pos != _vectors.end(); ++pos)
292  {
293  ParallelType type = _vector_types[pos->first];
294 
295  if (type == GHOSTED)
296  {
297 #ifdef LIBMESH_ENABLE_GHOSTED
298  pos->second->init (this->n_dofs(), this->n_local_dofs(),
299  _dof_map->get_send_list(), false,
300  GHOSTED);
301 #else
302  libmesh_error_msg("Cannot initialize ghosted vectors when they are not enabled.");
303 #endif
304  }
305  else if (type == SERIAL)
306  {
307  pos->second->init (this->n_dofs(), false, type);
308  }
309  else
310  {
311  libmesh_assert_equal_to(type, PARALLEL);
312  pos->second->init (this->n_dofs(), this->n_local_dofs(), false, type);
313  }
314  }

319 void System::restrict_vectors ()
320 {
321 #ifdef LIBMESH_ENABLE_AMR
322  // Restrict the _vectors on the coarsened cells
323  for (vectors_iterator pos = _vectors.begin(); pos != _vectors.end(); ++pos)
324  {
325  NumericVector<Number>* v = pos->second;
326 
327  if (_vector_projections[pos->first])
328  {
329  this->project_vector (*v, this->vector_is_adjoint(pos->first));
330  }
331  else
332  {
333  ParallelType type = _vector_types[pos->first];
334 
335  if(type == GHOSTED)
336  {
337 #ifdef LIBMESH_ENABLE_GHOSTED
338  pos->second->init (this->n_dofs(), this->n_local_dofs(),
339  _dof_map->get_send_list(), false,
340  GHOSTED);
341 #else
342  libmesh_error_msg("Cannot initialize ghosted vectors when they are not enabled.");
343 #endif
344  }
345  else
346  pos->second->init (this->n_dofs(), this->n_local_dofs(), false, type);
347  }
348  }
349 
350  const std::vector<dof_id_type>& send_list = _dof_map->get_send_list ();
351 
352  // Restrict the solution on the coarsened cells
353  if (_solution_projection)
354  this->project_vector (*solution);
355 
356 #ifdef LIBMESH_ENABLE_GHOSTED
357  current_local_solution->init(this->n_dofs(),
358  this->n_local_dofs(), send_list,
359  false, GHOSTED);
360 #else
361  current_local_solution->init(this->n_dofs());
362 #endif
363 
364  if (_solution_projection)
365  solution->localize (*current_local_solution, send_list);
366 
367 #endif // LIBMESH_ENABLE_AMR
368 }

527 bool System::compare (const System& other_system,
528  const Real threshold,
529  const bool verbose) const
530 {
531  // we do not care for matrices, but for vectors
532  libmesh_assert (!_can_add_vectors);
533  libmesh_assert (!other_system._can_add_vectors);
534 
535  if (verbose)
536  {
537  libMesh::out << " Systems \"" << _sys_name << "\"" << std::endl;
538  libMesh::out << " comparing matrices not supported." << std::endl;
539  libMesh::out << " comparing names...";
540  }
541 
542  // compare the name: 0 means identical
543  const int name_result = _sys_name.compare(other_system.name());
544  if (verbose)
545  {
546  if (name_result == 0)
547  libMesh::out << " identical." << std::endl;
548  else
549  libMesh::out << " names not identical." << std::endl;
550  libMesh::out << " comparing solution vector...";
551  }
552 
553 
554  // compare the solution: -1 means identical
555  const int solu_result = solution->compare (*other_system.solution.get(),
556  threshold);
557 
558  if (verbose)
559  {
560  if (solu_result == -1)
561  libMesh::out << " identical up to threshold." << std::endl;
562  else
563  libMesh::out << " first difference occured at index = "
564  << solu_result << "." << std::endl;
565  }
566 
567 
568  // safety check, whether we handle at least the same number
569  // of vectors
570  std::vector<int> ov_result;
571 
572  if (this->n_vectors() != other_system.n_vectors())
573  {
574  if (verbose)
575  {
576  libMesh::out << " Fatal difference. This system handles "
577  << this->n_vectors() << " add'l vectors," << std::endl
578  << " while the other system handles "
579  << other_system.n_vectors()
580  << " add'l vectors." << std::endl
581  << " Aborting comparison." << std::endl;
582  }
583  return false;
584  }
585  else if (this->n_vectors() == 0)
586  {
587  // there are no additional vectors...
588  ov_result.clear ();
589  }
590  else
591  {
592  // compare other vectors
593  for (const_vectors_iterator pos = _vectors.begin();
594  pos != _vectors.end(); ++pos)
595  {
596  if (verbose)
597  libMesh::out << " comparing vector \""
598  << pos->first << "\" ...";
599 
600  // assume they have the same name
601  const NumericVector<Number>& other_system_vector =
602  other_system.get_vector(pos->first);
603 
604  ov_result.push_back(pos->second->compare (other_system_vector,
605  threshold));
606 
607  if (verbose)
608  {
609  if (ov_result[ov_result.size()-1] == -1)
610  libMesh::out << " identical up to threshold." << std::endl;
611  else
612  libMesh::out << " first difference occured at" << std::endl
613  << " index = " << ov_result[ov_result.size()-1] << "." << std::endl;
614  }
615 
616  }
617 
618  } // finished comparing additional vectors
619 
620 
621  bool overall_result;
622 
623  // sum up the results
624  if ((name_result==0) && (solu_result==-1))
625  {
626  if (ov_result.size()==0)
627  overall_result = true;
628  else
629  {
630  bool ov_identical;
631  unsigned int n = 0;
632  do
633  {
634  ov_identical = (ov_result[n]==-1);
635  n++;
636  }
637  while (ov_identical && n<ov_result.size());
638  overall_result = ov_identical;
639  }
640  }
641  else
642  overall_result = false;
643 
644  if (verbose)
645  {
646  libMesh::out << " finished comparisons, ";
647  if (overall_result)
648  libMesh::out << "found no differences." << std::endl << std::endl;
649  else
650  libMesh::out << "found differences." << std::endl << std::endl;
651  }
652 
653  return overall_result;
654 }

1463  // I'm not sure how best to mix Hilbert norms on some variables (for
1464  // which we'll want to square then sum then square root) with norms
1465  // like L_inf (for which we'll just want to take an absolute value
1466  // and then sum).
1467  bool using_hilbert_norm = true,
1468  using_nonhilbert_norm = true;
1469 
1470  // Loop over all variables
1471  for (unsigned int var=0; var != this->n_vars(); ++var)
1472  {
1473  // Skip any variables we don't need to integrate
1474  Real norm_weight_sq = norm.weight_sq(var);
1475  if (norm_weight_sq == 0.0)
1476  continue;
1477  Real norm_weight = norm.weight(var);
1478 
1479  // Check for unimplemented norms (rather than just returning 0).
1480  FEMNormType norm_type = norm.type(var);
1481  if((norm_type==H1) ||
1482  (norm_type==H2) ||
1483  (norm_type==L2) ||
1484  (norm_type==H1_SEMINORM) ||
1485  (norm_type==H2_SEMINORM))
1486  {
1487  if (!using_hilbert_norm)
1488  libmesh_not_implemented();
1489  using_nonhilbert_norm = false;
1490  }
1491  else if ((norm_type==L1) ||
1492  (norm_type==L_INF) ||
1493  (norm_type==W1_INF_SEMINORM) ||
1494  (norm_type==W2_INF_SEMINORM))
1495  {
1496  if (!using_nonhilbert_norm)
1497  libmesh_not_implemented();
1498  using_hilbert_norm = false;
1499  }
1500  else
1501  libmesh_not_implemented();
1502 
1503  const FEType& fe_type = this->get_dof_map().variable_type(var);
1504  AutoPtr<QBase> qrule =
1505  fe_type.default_quadrature_rule (dim);
1506  AutoPtr<FEBase> fe
1507  (FEBase::build(dim, fe_type));
1508  fe->attach_quadrature_rule (qrule.get());
1509 
1510  const std::vector<Real>& JxW = fe->get_JxW();
1511  const std::vector<std::vector<Real> >* phi = NULL;
1512  if (norm_type == H1 ||
1513  norm_type == H2 ||
1514  norm_type == L2 ||
1515  norm_type == L1 ||
1516  norm_type == L_INF)
1517  phi = &(fe->get_phi());
1518 
1519  const std::vector<std::vector<RealGradient> >* dphi = NULL;
1520  if (norm_type == H1 ||
1521  norm_type == H2 ||
1522  norm_type == H1_SEMINORM ||
1523  norm_type == W1_INF_SEMINORM)
1524  dphi = &(fe->get_dphi());
1525 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
1526  const std::vector<std::vector<RealTensor> >* d2phi = NULL;
1527  if (norm_type == H2 ||
1528  norm_type == H2_SEMINORM ||
1529  norm_type == W2_INF_SEMINORM)
1530  d2phi = &(fe->get_d2phi());
1531 #endif
1532 
1533  std::vector<dof_id_type> dof_indices;
1534 
1535  // Begin the loop over the elements
1536  MeshBase::const_element_iterator el =
1537  this->get_mesh().active_local_elements_begin();
1538  const MeshBase::const_element_iterator end_el =
1539  this->get_mesh().active_local_elements_end();
1540 
1541  for ( ; el != end_el; ++el)
1542  {
1543  const Elem* elem = *el;
1544 
1545  fe->reinit (elem);
1546 
1547  this->get_dof_map().dof_indices (elem, dof_indices, var);
1548 
1549  const unsigned int n_qp = qrule->n_points();
1550 
1551  const unsigned int n_sf = cast_int<unsigned int>
1552  (dof_indices.size());
1553 
1554  // Begin the loop over the Quadrature points.
1555  for (unsigned int qp=0; qp<n_qp; qp++)
1556  {
1557  if (norm_type == L1)
1558  {
1559  Number u_h = 0.;
1560  for (unsigned int i=0; i != n_sf; ++i)
1561  u_h += (*phi)[i][qp] * (*local_v)(dof_indices[i]);
1562  v_norm += norm_weight *
1563  JxW[qp] * std::abs(u_h);
1564  }
1565 
1566  if (norm_type == L_INF)
1567  {
1568  Number u_h = 0.;
1569  for (unsigned int i=0; i != n_sf; ++i)
1570  u_h += (*phi)[i][qp] * (*local_v)(dof_indices[i]);
1571  v_norm = std::max(v_norm, norm_weight * std::abs(u_h));
1572  }
1573 
1574  if (norm_type == H1 ||
1575  norm_type == H2 ||
1576  norm_type == L2)
1577  {
1578  Number u_h = 0.;
1579  for (unsigned int i=0; i != n_sf; ++i)
1580  u_h += (*phi)[i][qp] * (*local_v)(dof_indices[i]);
1581  v_norm += norm_weight_sq *
1582  JxW[qp] * TensorTools::norm_sq(u_h);
1583  }
1584 
1585  if (norm_type == H1 ||
1586  norm_type == H2 ||
1587  norm_type == H1_SEMINORM)
1588  {
1589  Gradient grad_u_h;
1590  for (unsigned int i=0; i != n_sf; ++i)
1591  grad_u_h.add_scaled((*dphi)[i][qp], (*local_v)(dof_indices[i]));
1592  v_norm += norm_weight_sq *
1593  JxW[qp] * grad_u_h.size_sq();
1594  }
1595 
1596  if (norm_type == W1_INF_SEMINORM)
1597  {
1598  Gradient grad_u_h;
1599  for (unsigned int i=0; i != n_sf; ++i)
1600  grad_u_h.add_scaled((*dphi)[i][qp], (*local_v)(dof_indices[i]));
1601  v_norm = std::max(v_norm, norm_weight * grad_u_h.size());
1602  }
1603 
1604 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
1605  if (norm_type == H2 ||
1606  norm_type == H2_SEMINORM)
1607  {
1608  Tensor hess_u_h;
1609  for (unsigned int i=0; i != n_sf; ++i)
1610  hess_u_h.add_scaled((*d2phi)[i][qp], (*local_v)(dof_indices[i]));
1611  v_norm += norm_weight_sq *
1612  JxW[qp] * hess_u_h.size_sq();
1613  }
1614 
1615  if (norm_type == W2_INF_SEMINORM)
1616  {
1617  Tensor hess_u_h;
1618  for (unsigned int i=0; i != n_sf; ++i)
1619  hess_u_h.add_scaled((*d2phi)[i][qp], (*local_v)(dof_indices[i]));
1620  v_norm = std::max(v_norm, norm_weight * hess_u_h.size());
1621  }
1622 #endif
1623  }
1624  }
1625  }
1626 
1627  if (using_hilbert_norm)
1628  {
1629  this->comm().sum(v_norm);
1630  v_norm = std::sqrt(v_norm);
1631  }
1632  else
1633  {
1634  this->comm().max(v_norm);
1635  }
1636 
1637  STOP_LOG ("calculate_norm()", "System");
1638 
1639  return v_norm;
1640 }

1975 Number System::point_value(unsigned int var, const Point &p, const bool insist_on_success) const
1976 {
1977  // This function must be called on every processor; there's no
1978  // telling where in the partition p falls.
1979  parallel_object_only();
1980 
1981  // And every processor had better agree about which point we're
1982  // looking for
1983 #ifndef NDEBUG
1984  this->comm().verify(p);
1985 #endif // NDEBUG
1986 
1987  // Get a reference to the mesh object associated with the system object that calls this function
1988  const MeshBase &mesh = this->get_mesh();
1989 
1990  // Use an existing PointLocator or create a new one
1991  AutoPtr<PointLocatorBase> locator_ptr = mesh.sub_point_locator();
1992  PointLocatorBase& locator = *locator_ptr;
1993 
1994  if (!insist_on_success)
1995  locator.enable_out_of_mesh_mode();
1996 
1997  // Get a pointer to the element that contains P
1998  const Elem *e = locator(p);
1999 
2000  Number u = 0;
2001 
2002  if (e && e->processor_id() == this->processor_id())
2003  u = point_value(var, p, *e);
2004 
2005  // If I have an element containing p, then let's let everyone know
2006  processor_id_type lowest_owner =
2007  (e && (e->processor_id() == this->processor_id())) ?
2008  this->processor_id() : this->n_processors();
2009  this->comm().min(lowest_owner);
2010 
2011  // Everybody should get their value from a processor that was able
2012  // to compute it.
2013  // If nobody admits owning the point, we have a problem.
2014  if (lowest_owner != this->n_processors())
2015  this->comm().broadcast(u, lowest_owner);
2016  else
2017  libmesh_assert(!insist_on_success);
2018 
2019  return u;
2020 }
