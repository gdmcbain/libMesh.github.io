483 template <typename T>
484 inline
485 void LaspackVector<T>::init (const numeric_index_type n,
486  const numeric_index_type libmesh_dbg_var(n_local),
487  const bool fast,
488  const ParallelType)
489 {
490  // Laspack vectors only for serial cases,
491  // but can provide a "parallel" vector on one processor.
492  libmesh_assert_equal_to (n, n_local);
493 
494  this->_type = SERIAL;
495 
496  // Clear initialized vectors
497  if (this->initialized())
498  this->clear();
499 
500  // create a sequential vector
501 
502  static int cnt = 0;
503  char foo[80];
504  std::sprintf(foo, "Vec-%d", cnt++);
505 
506  V_Constr(&_vec, const_cast<char *>(foo), n, Normal, _LPTrue);
507 
508  this->_is_initialized = true;
509 #ifndef NDEBUG
510  this->_is_closed = true;
511 #endif
512 
513  // Optionally zero out all components
514  if (fast == false)
515  this->zero ();
516 
517  return;
518 }
