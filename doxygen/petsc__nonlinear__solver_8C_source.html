75  //---------------------------------------------------------------
76  // this function is called by PETSc to evaluate the residual at X
77  PetscErrorCode
78  __libmesh_petsc_snes_residual (SNES snes, Vec x, Vec r, void *ctx)
79  {
80  START_LOG("residual()", "PetscNonlinearSolver");
81 
82  PetscErrorCode ierr=0;
83 
84  libmesh_assert(x);
85  libmesh_assert(r);
86  libmesh_assert(ctx);
87 
88  // No way to safety-check this cast, since we got a void*...
89  PetscNonlinearSolver<Number>* solver =
90  static_cast<PetscNonlinearSolver<Number>*> (ctx);
91 
92  // Get the current iteration number from the snes object,
93  // store it in the PetscNonlinearSolver object for possible use
94  // by the user's residual function.
95  {
96  PetscInt n_iterations = 0;
97  ierr = SNESGetIterationNumber(snes, &n_iterations);
98  CHKERRABORT(solver->comm().get(),ierr);
99  solver->_current_nonlinear_iteration_number = cast_int<unsigned>(n_iterations);
100  }
101 
102  NonlinearImplicitSystem &sys = solver->system();
103 
104  PetscVector<Number>& X_sys = *cast_ptr<PetscVector<Number>*>(sys.solution.get());
105  PetscVector<Number>& R_sys = *cast_ptr<PetscVector<Number>*>(sys.rhs);
106  PetscVector<Number> X_global(x, sys.comm()), R(r, sys.comm());
107 
108  // Use the systems update() to get a good local version of the parallel solution
109  X_global.swap(X_sys);
110  R.swap(R_sys);
111 
112  sys.get_dof_map().enforce_constraints_exactly(sys);
113 
114  sys.update();
115 
116  //Swap back
117  X_global.swap(X_sys);
118  R.swap(R_sys);
119 
120  if (solver->_zero_out_residual)
121  R.zero();
122 
123  //-----------------------------------------------------------------------------
124  // if the user has provided both function pointers and objects only the pointer
125  // will be used, so catch that as an error
126  if (solver->residual && solver->residual_object)
127  libmesh_error_msg("ERROR: cannot specifiy both a function and object to compute the Residual!");
128 
129  if (solver->matvec && solver->residual_and_jacobian_object)
130  libmesh_error_msg("ERROR: cannot specifiy both a function and object to compute the combined Residual & Jacobian!");
131 
132  if (solver->residual != NULL)
133  solver->residual(*sys.current_local_solution.get(), R, sys);
134 
135  else if (solver->residual_object != NULL)
136  solver->residual_object->residual(*sys.current_local_solution.get(), R, sys);
137 
138  else if (solver->matvec != NULL)
139  solver->matvec (*sys.current_local_solution.get(), &R, NULL, sys);
140 
141  else if (solver->residual_and_jacobian_object != NULL)
142  solver->residual_and_jacobian_object->residual_and_jacobian (*sys.current_local_solution.get(), &R, NULL, sys);
143 
144  else
145  libmesh_error_msg("Error! Unable to compute residual and/or Jacobian!");
146 
147  R.close();
148  X_global.close();
149 
150  STOP_LOG("residual()", "PetscNonlinearSolver");
151 
152  return ierr;
153  }

157  //---------------------------------------------------------------
158  // this function is called by PETSc to evaluate the Jacobian at X
159 #if PETSC_RELEASE_LESS_THAN(3,5,0)
160  PetscErrorCode
161  __libmesh_petsc_snes_jacobian (SNES snes, Vec x, Mat *jac, Mat *pc, MatStructure *msflag, void *ctx)
162 #else
163  PetscErrorCode
164  __libmesh_petsc_snes_jacobian (SNES snes, Vec x, Mat jac, Mat pc, void *ctx)
165 #endif
166  {
167  START_LOG("jacobian()", "PetscNonlinearSolver");
168 
169  PetscErrorCode ierr=0;
170 
171  libmesh_assert(ctx);
172 
173  // No way to safety-check this cast, since we got a void*...
174  PetscNonlinearSolver<Number>* solver =
175  static_cast<PetscNonlinearSolver<Number>*> (ctx);
176 
177  // Get the current iteration number from the snes object,
178  // store it in the PetscNonlinearSolver object for possible use
179  // by the user's Jacobian function.
180  {
181  PetscInt n_iterations = 0;
182  ierr = SNESGetIterationNumber(snes, &n_iterations);
183  CHKERRABORT(solver->comm().get(),ierr);
184  solver->_current_nonlinear_iteration_number = cast_int<unsigned>(n_iterations);
185  }
186 
187  NonlinearImplicitSystem &sys = solver->system();
188 #if PETSC_RELEASE_LESS_THAN(3,5,0)
189  PetscMatrix<Number> PC(*pc, sys.comm());
190  PetscMatrix<Number> Jac(*jac, sys.comm());
191 #else
192  PetscMatrix<Number> PC(pc, sys.comm());
193  PetscMatrix<Number> Jac(jac, sys.comm());
194 #endif
195  PetscVector<Number>& X_sys = *cast_ptr<PetscVector<Number>*>(sys.solution.get());
196  PetscMatrix<Number>& Jac_sys = *cast_ptr<PetscMatrix<Number>*>(sys.matrix);
197  PetscVector<Number> X_global(x, sys.comm());
198 
199  // Set the dof maps
200  PC.attach_dof_map(sys.get_dof_map());
201  Jac.attach_dof_map(sys.get_dof_map());
202 
203  // Use the systems update() to get a good local version of the parallel solution
204  X_global.swap(X_sys);
205  Jac.swap(Jac_sys);
206 
207  sys.get_dof_map().enforce_constraints_exactly(sys);
208  sys.update();
209 
210  X_global.swap(X_sys);
211  Jac.swap(Jac_sys);
212 
213  if (solver->_zero_out_jacobian)
214  PC.zero();
215 
216  //-----------------------------------------------------------------------------
217  // if the user has provided both function pointers and objects only the pointer
218  // will be used, so catch that as an error
219  if (solver->jacobian && solver->jacobian_object)
220  libmesh_error_msg("ERROR: cannot specify both a function and object to compute the Jacobian!");
221 
222  if (solver->matvec && solver->residual_and_jacobian_object)
223  libmesh_error_msg("ERROR: cannot specify both a function and object to compute the combined Residual & Jacobian!");
224 
225  if (solver->jacobian != NULL)
226  solver->jacobian(*sys.current_local_solution.get(), PC, sys);
227 
228  else if (solver->jacobian_object != NULL)
229  solver->jacobian_object->jacobian(*sys.current_local_solution.get(), PC, sys);
230 
231  else if (solver->matvec != NULL)
232  solver->matvec(*sys.current_local_solution.get(), NULL, &PC, sys);
233 
234  else if (solver->residual_and_jacobian_object != NULL)
235  solver->residual_and_jacobian_object->residual_and_jacobian (*sys.current_local_solution.get(), NULL, &PC, sys);
236 
237  else
238  libmesh_error_msg("Error! Unable to compute residual and/or Jacobian!");
239 
240  PC.close();
241  Jac.close();
242  X_global.close();
243 #if PETSC_RELEASE_LESS_THAN(3,5,0)
244  *msflag = SAME_NONZERO_PATTERN;
245 #endif
246  STOP_LOG("jacobian()", "PetscNonlinearSolver");
247 
248  return ierr;
249  }

301 template <typename T>
302 void PetscNonlinearSolver<T>::init (const char* name)
303 {
304  // Initialize the data structures if not done so already.
305  if (!this->initialized())
306  {
307  this->_is_initialized = true;
308 
309  PetscErrorCode ierr=0;
310 
311 #if PETSC_VERSION_LESS_THAN(2,1,2)
312  // At least until Petsc 2.1.1, the SNESCreate had a different calling syntax.
313  // The second argument was of type SNESProblemType, and could have a value of
314  // either SNES_NONLINEAR_EQUATIONS or SNES_UNCONSTRAINED_MINIMIZATION.
315  ierr = SNESCreate(this->comm().get(), SNES_NONLINEAR_EQUATIONS, &_snes);
316  LIBMESH_CHKERRABORT(ierr);
317 
318 #else
319 
320  ierr = SNESCreate(this->comm().get(),&_snes);
321  LIBMESH_CHKERRABORT(ierr);
322 
323 #endif
324 
325  if (name)
326  {
327  ierr = SNESSetOptionsPrefix(_snes, name);
328  LIBMESH_CHKERRABORT(ierr);
329  }
330 
331  if (_default_monitor)
332  {
333 #if PETSC_VERSION_LESS_THAN(2,3,3)
334  ierr = SNESSetMonitor (_snes, __libmesh_petsc_snes_monitor,
335  this, PETSC_NULL);
336 #else
337  // API name change in PETSc 2.3.3
338  ierr = SNESMonitorSet (_snes, __libmesh_petsc_snes_monitor,
339  this, PETSC_NULL);
340 #endif
341  LIBMESH_CHKERRABORT(ierr);
342  }
343 
344 #if PETSC_VERSION_LESS_THAN(3,1,0)
345  // Cannot call SNESSetOptions before SNESSetFunction when using
346  // any matrix free options with PETSc 3.1.0+
347  ierr = SNESSetFromOptions(_snes);
348  LIBMESH_CHKERRABORT(ierr);
349 #endif
350 
351  if(this->_preconditioner)
352  {
353  KSP ksp;
354  ierr = SNESGetKSP (_snes, &ksp);
355  LIBMESH_CHKERRABORT(ierr);
356  PC pc;
357  ierr = KSPGetPC(ksp,&pc);
358  LIBMESH_CHKERRABORT(ierr);
359 
360  this->_preconditioner->init();
361 
362  PCSetType(pc, PCSHELL);
363  PCShellSetContext(pc,(void*)this->_preconditioner);
364 
365  //Re-Use the shell functions from petsc_linear_solver
366  PCShellSetSetUp(pc,__libmesh_petsc_preconditioner_setup);
367  PCShellSetApply(pc,__libmesh_petsc_preconditioner_apply);
368  }
369  }

372 #if !PETSC_VERSION_LESS_THAN(3,3,0)
373 template <typename T>
374 void
375 PetscNonlinearSolver<T>::build_mat_null_space(NonlinearImplicitSystem::ComputeVectorSubspace* computeSubspaceObject,
376  void (*computeSubspace)(std::vector<NumericVector<Number>*>&, sys_type&),
377  MatNullSpace *msp)
378 {
379  PetscErrorCode ierr;
380  std::vector<NumericVector<Number>* > sp;
381  if (computeSubspaceObject)
382  (*computeSubspaceObject)(sp, this->system());
383  else
384  (*computeSubspace)(sp, this->system());
385 
386  *msp = PETSC_NULL;
387  if (sp.size())
388  {
389  Vec *modes;
390  PetscScalar *dots;
391  PetscInt nmodes = cast_int<PetscInt>(sp.size());
392 
393 #if PETSC_RELEASE_LESS_THAN(3,5,0)
394  ierr = PetscMalloc2(nmodes,Vec,&modes,nmodes,PetscScalar,&dots);
395 #else
396  ierr = PetscMalloc2(nmodes,&modes,nmodes,&dots);
397 #endif
398  LIBMESH_CHKERRABORT(ierr);
399 
400  for (PetscInt i=0; i<nmodes; ++i)
401  {
402  PetscVector<T>* pv = cast_ptr<PetscVector<T>*>(sp[i]);
403  Vec v = pv->vec();
404 
405  ierr = VecDuplicate(v, modes+i);
406  LIBMESH_CHKERRABORT(ierr);
407 
408  ierr = VecCopy(v,modes[i]);
409  LIBMESH_CHKERRABORT(ierr);
410  }
411 
412  // Normalize.
413  ierr = VecNormalize(modes[0],PETSC_NULL);
414  LIBMESH_CHKERRABORT(ierr);
415 
416  for (PetscInt i=1; i<nmodes; i++)
417  {
418  // Orthonormalize vec[i] against vec[0:i-1]
419  ierr = VecMDot(modes[i],i,modes,dots);
420  LIBMESH_CHKERRABORT(ierr);
421 
422  for (PetscInt j=0; j<i; j++)
423  dots[j] *= -1.;
424 
425  ierr = VecMAXPY(modes[i],i,dots,modes);
426  LIBMESH_CHKERRABORT(ierr);
427 
428  ierr = VecNormalize(modes[i],PETSC_NULL);
429  LIBMESH_CHKERRABORT(ierr);
430  }
431 
432  ierr = MatNullSpaceCreate(this->comm().get(), PETSC_FALSE, nmodes, modes, msp);
433  LIBMESH_CHKERRABORT(ierr);
434 
435  for (PetscInt i=0; i<nmodes; ++i)
436  {
437  ierr = VecDestroy(modes+i);
438  LIBMESH_CHKERRABORT(ierr);
439  }
440 
441  ierr = PetscFree2(modes,dots);
442  LIBMESH_CHKERRABORT(ierr);
443  }
444 }
445 #endif

447 template <typename T>
448 std::pair<unsigned int, Real>
449 PetscNonlinearSolver<T>::solve (SparseMatrix<T>& jac_in, // System Jacobian Matrix
450  NumericVector<T>& x_in, // Solution vector
451  NumericVector<T>& r_in, // Residual vector
452  const double, // Stopping tolerance
453  const unsigned int)
454 {
455  START_LOG("solve()", "PetscNonlinearSolver");
456  this->init ();
457 
458  // Make sure the data passed in are really of Petsc types
459  PetscMatrix<T>* jac = cast_ptr<PetscMatrix<T>*>(&jac_in);
460  PetscVector<T>* x = cast_ptr<PetscVector<T>*>(&x_in);
461  PetscVector<T>* r = cast_ptr<PetscVector<T>*>(&r_in);
462 
463  PetscErrorCode ierr=0;
464  PetscInt n_iterations =0;
465  // Should actually be a PetscReal, but I don't know which version of PETSc first introduced PetscReal
466  Real final_residual_norm=0.;
467 
468  ierr = SNESSetFunction (_snes, r->vec(), __libmesh_petsc_snes_residual, this);
469  LIBMESH_CHKERRABORT(ierr);
470 
471  // Only set the jacobian function if we've been provided with something to call.
472  // This allows a user to set their own jacobian function if they want to
473  if (this->jacobian || this->jacobian_object || this->residual_and_jacobian_object)
474  {
475  ierr = SNESSetJacobian (_snes, jac->mat(), jac->mat(), __libmesh_petsc_snes_jacobian, this);
476  LIBMESH_CHKERRABORT(ierr);
477  }
478 #if !PETSC_VERSION_LESS_THAN(3,3,0)
479  // Only set the nullspace if we have a way of computing it and the result is non-empty.
480  if (this->nullspace || this->nullspace_object)
481  {
482  MatNullSpace msp;
483  this->build_mat_null_space(this->nullspace_object, this->nullspace, &msp);
484  if (msp)
485  {
486  ierr = MatSetNullSpace(jac->mat(), msp);
487  LIBMESH_CHKERRABORT(ierr);
488 
489  ierr = MatNullSpaceDestroy(&msp);
490  LIBMESH_CHKERRABORT(ierr);
491  }
492  }
493 
494  // Only set the nearnullspace if we have a way of computing it and the result is non-empty.
495  if (this->nearnullspace || this->nearnullspace_object)
496  {
497  MatNullSpace msp = PETSC_NULL;
498  this->build_mat_null_space(this->nearnullspace_object, this->nearnullspace, &msp);
499 
500  if(msp) {
501  ierr = MatSetNearNullSpace(jac->mat(), msp);
502  LIBMESH_CHKERRABORT(ierr);
503 
504  ierr = MatNullSpaceDestroy(&msp);
505  LIBMESH_CHKERRABORT(ierr);
506  }
507  }
508 #endif
509  // Have the Krylov subspace method use our good initial guess rather than 0
510  KSP ksp;
511  ierr = SNESGetKSP (_snes, &ksp);
512  LIBMESH_CHKERRABORT(ierr);
513 
514  // Set the tolerances for the iterative solver. Use the user-supplied
515  // tolerance for the relative residual & leave the others at default values
516  ierr = KSPSetTolerances (ksp, this->initial_linear_tolerance, PETSC_DEFAULT,
517  PETSC_DEFAULT, this->max_linear_iterations);
518  LIBMESH_CHKERRABORT(ierr);
519 
520  // Set the tolerances for the non-linear solver.
521  ierr = SNESSetTolerances(_snes, this->absolute_residual_tolerance, this->relative_residual_tolerance,
522  this->relative_step_tolerance, this->max_nonlinear_iterations, this->max_function_evaluations);
523  LIBMESH_CHKERRABORT(ierr);
524 
525  //Pull in command-line options
526  KSPSetFromOptions(ksp);
527  SNESSetFromOptions(_snes);
528 
529  if (this->user_presolve)
530  this->user_presolve(this->system());
531 
532  //Set the preconditioning matrix
533  if(this->_preconditioner)
534  {
535  this->_preconditioner->set_matrix(jac_in);
536  this->_preconditioner->init();
537  }
538 
539  // ierr = KSPSetInitialGuessNonzero (ksp, PETSC_TRUE);
540  // LIBMESH_CHKERRABORT(ierr);
541 
542  // Older versions (at least up to 2.1.5) of SNESSolve took 3 arguments,
543  // the last one being a pointer to an int to hold the number of iterations required.
544 # if PETSC_VERSION_LESS_THAN(2,2,0)
545 
546  ierr = SNESSolve (_snes, x->vec(), &n_iterations);
547  LIBMESH_CHKERRABORT(ierr);
548 
549  // 2.2.x style
550 #elif PETSC_VERSION_LESS_THAN(2,3,0)
551 
552  ierr = SNESSolve (_snes, x->vec());
553  LIBMESH_CHKERRABORT(ierr);
554 
555  // 2.3.x & newer style
556 #else
557 
558  ierr = SNESSolve (_snes, PETSC_NULL, x->vec());
559  LIBMESH_CHKERRABORT(ierr);
560 
561  ierr = SNESGetIterationNumber(_snes,&n_iterations);
562  LIBMESH_CHKERRABORT(ierr);
563 
564  ierr = SNESGetLinearSolveIterations(_snes, &_n_linear_iterations);
565  LIBMESH_CHKERRABORT(ierr);
566 
567 #endif
568 
569  // SNESGetFunction has been around forever and should work on all
570  // versions of PETSc. This is also now the recommended approach
571  // according to the documentation for the PETSc 3.5.1 release:
572  // http://www.mcs.anl.gov/petsc/documentation/changes/35.html
573  Vec f;
574  ierr = SNESGetFunction(_snes, &f, 0, 0);
575  LIBMESH_CHKERRABORT(ierr);
576  ierr = VecNorm(f, NORM_2, &final_residual_norm);
577  LIBMESH_CHKERRABORT(ierr);
578 
579  // Get and store the reason for convergence
580  SNESGetConvergedReason(_snes, &_reason);
581 
582  //Based on Petsc 2.3.3 documentation all diverged reasons are negative
583  this->converged = (_reason >= 0);
584 
585  this->clear();
586 
587  STOP_LOG("solve()", "PetscNonlinearSolver");
588 
589  // return the # of its. and the final residual norm.
590  return std::make_pair(n_iterations, final_residual_norm);
591 }
