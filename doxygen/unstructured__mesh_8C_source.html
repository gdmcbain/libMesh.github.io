104 void UnstructuredMesh::copy_nodes_and_elements
105 (const UnstructuredMesh& other_mesh, const bool skip_find_neighbors)
106 {
107  // We're assuming our subclass data needs no copy
108  libmesh_assert_equal_to (_n_parts, other_mesh._n_parts);
109  libmesh_assert_equal_to (_dim, other_mesh._dim);
110  libmesh_assert_equal_to (_is_prepared, other_mesh._is_prepared);
111 
112  // We're assuming the other mesh has proper element number ordering,
113  // so that we add parents before their children.
114 #ifdef DEBUG
115  MeshTools::libmesh_assert_valid_amr_elem_ids(other_mesh);
116 #endif
117 
118  //Copy in Nodes
119  {
120  //Preallocate Memory if necessary
121  this->reserve_nodes(other_mesh.n_nodes());
122 
123  const_node_iterator it = other_mesh.nodes_begin();
124  const_node_iterator end = other_mesh.nodes_end();
125 
126  for (; it != end; ++it)
127  {
128  const Node *oldn = *it;
129 
130  // Add new nodes in old node Point locations
131  /*Node *newn =*/ this->add_point(*oldn, oldn->id(), oldn->processor_id());
132 
133  // And start them off in the same subdomain
134  // newn->processor_id() = oldn->processor_id();
135  }
136  }
137 
138  //Copy in Elements
139  {
140  //Preallocate Memory if necessary
141  this->reserve_elem(other_mesh.n_elem());
142 
143  // Declare a map linking old and new elements, needed to copy the neighbor lists
144  std::map<const Elem*, Elem*> old_elems_to_new_elems;
145 
146  // Loop over the elements
147  MeshBase::const_element_iterator it = other_mesh.elements_begin();
148  const MeshBase::const_element_iterator end = other_mesh.elements_end();
149 
150  // FIXME: Where do we set element IDs??
151  for (; it != end; ++it)
152  {
153  //Look at the old element
154  const Elem *old = *it;
155  //Build a new element
156  Elem *newparent = old->parent() ?
157  this->elem(old->parent()->id()) : NULL;
158  AutoPtr<Elem> ap = Elem::build(old->type(), newparent);
159  Elem * el = ap.release();
160 
161  el->subdomain_id() = old->subdomain_id();
162 
163  for (unsigned int s=0; s != old->n_sides(); ++s)
164  if (old->neighbor(s) == remote_elem)
165  el->set_neighbor(s, const_cast<RemoteElem*>(remote_elem));
166 
167 #ifdef LIBMESH_ENABLE_AMR
168  if (old->has_children())
169  for (unsigned int c=0; c != old->n_children(); ++c)
170  if (old->child(c) == remote_elem)
171  el->add_child(const_cast<RemoteElem*>(remote_elem), c);
172 
173  //Create the parent's child pointers if necessary
174  if (newparent)
175  {
176  unsigned int oldc = old->parent()->which_child_am_i(old);
177  newparent->add_child(el, oldc);
178  }
179 
180  // Copy the refinement flags
181  el->set_refinement_flag(old->refinement_flag());
182  el->set_p_refinement_flag(old->p_refinement_flag());
183 #endif // #ifdef LIBMESH_ENABLE_AMR
184 
185  //Assign all the nodes
186  for(unsigned int i=0;i<el->n_nodes();i++)
187  el->set_node(i) = &this->node(old->node(i));
188 
189  // And start it off in the same subdomain
190  el->processor_id() = old->processor_id();
191 
192  // Give it the same id
193  el->set_id(old->id());
194 
195  //Hold onto it
196  if(!skip_find_neighbors)
197  {
198  this->add_elem(el);
199  }
200  else
201  {
202  Elem* new_el = this->add_elem(el);
203  old_elems_to_new_elems[old] = new_el;
204  }
205 
206  // Add the link between the original element and this copy to the map
207  if(skip_find_neighbors)
208  old_elems_to_new_elems[old] = el;
209  }
210 
211  // Loop (again) over the elements to fill in the neighbors
212  if(skip_find_neighbors)
213  {
214  it = other_mesh.elements_begin();
215  for (; it != end; ++it)
216  {
217  Elem* old_elem = *it;
218  Elem* new_elem = old_elems_to_new_elems[old_elem];
219  for (unsigned int s=0; s != old_elem->n_neighbors(); ++s)
220  {
221  const Elem* old_neighbor = old_elem->neighbor(s);
222  Elem* new_neighbor = old_elems_to_new_elems[old_neighbor];
223  new_elem->set_neighbor(s, new_neighbor);
224  }
225  }
226  }
227  }
228 
229  //Finally prepare the new Mesh for use. Keep the same numbering and
230  //partitioning but also the same renumbering and partitioning
231  //policies as our source mesh.
232  this->allow_renumbering(false);
233  this->skip_partitioning(true);
234  this->prepare_for_use(false, skip_find_neighbors);
235  this->allow_renumbering(other_mesh.allow_renumbering());
236  this->skip_partitioning(other_mesh.skip_partitioning());
237 }

252 void UnstructuredMesh::find_neighbors (const bool reset_remote_elements,
253  const bool reset_current_list)
254 {
255  // We might actually want to run this on an empty mesh
256  // (e.g. the boundary mesh for a nonexistant bcid!)
257  // libmesh_assert_not_equal_to (this->n_nodes(), 0);
258  // libmesh_assert_not_equal_to (this->n_elem(), 0);
259 
260  // This function must be run on all processors at once
261  parallel_object_only();
262 
263  START_LOG("find_neighbors()", "Mesh");
264 
265  const element_iterator el_end = this->elements_end();
266 
267  //TODO:[BSK] This should be removed later?!
268  if (reset_current_list)
269  for (element_iterator el = this->elements_begin(); el != el_end; ++el)
270  {
271  Elem* e = *el;
272  for (unsigned int s=0; s<e->n_neighbors(); s++)
273  if (e->neighbor(s) != remote_elem ||
274  reset_remote_elements)
275  e->set_neighbor(s,NULL);
276  }
277 
278  // Find neighboring elements by first finding elements
279  // with identical side keys and then check to see if they
280  // are neighbors
281  {
282  // data structures -- Use the hash_multimap if available
283  typedef unsigned int key_type;
284  typedef std::pair<Elem*, unsigned char> val_type;
285  typedef std::pair<key_type, val_type> key_val_pair;
286 
287  typedef LIBMESH_BEST_UNORDERED_MULTIMAP<key_type, val_type> map_type;
288 
289  // A map from side keys to corresponding elements & side numbers
290  map_type side_to_elem_map;
291 
292 
293 
294  for (element_iterator el = this->elements_begin(); el != el_end; ++el)
295  {
296  Elem* element = *el;
297 
298  for (unsigned char ms=0; ms<element->n_neighbors(); ms++)
299  {
300  next_side:
301  // If we haven't yet found a neighbor on this side, try.
302  // Even if we think our neighbor is remote, that
303  // information may be out of date.
304  if (element->neighbor(ms) == NULL ||
305  element->neighbor(ms) == remote_elem)
306  {
307  // Get the key for the side of this element
308  const unsigned int key = element->key(ms);
309 
310  // Look for elements that have an identical side key
311  std::pair <map_type::iterator, map_type::iterator>
312  bounds = side_to_elem_map.equal_range(key);
313 
314  // May be multiple keys, check all the possible
315  // elements which _might_ be neighbors.
316  if (bounds.first != bounds.second)
317  {
318  // Get the side for this element
319  const AutoPtr<Elem> my_side(element->side(ms));
320 
321  // Look at all the entries with an equivalent key
322  while (bounds.first != bounds.second)
323  {
324  // Get the potential element
325  Elem* neighbor = bounds.first->second.first;
326 
327  // Get the side for the neighboring element
328  const unsigned int ns = bounds.first->second.second;
329  const AutoPtr<Elem> their_side(neighbor->side(ns));
330  //libmesh_assert(my_side.get());
331  //libmesh_assert(their_side.get());
332 
333  // If found a match with my side
334  //
335  // We need special tests here for 1D:
336  // since parents and children have an equal
337  // side (i.e. a node), we need to check
338  // ns != ms, and we also check level() to
339  // avoid setting our neighbor pointer to
340  // any of our neighbor's descendants
341  if( (*my_side == *their_side) &&
342  (element->level() == neighbor->level()) &&
343  ((_dim != 1) || (ns != ms)) )
344  {
345  // So share a side. Is this a mixed pair
346  // of subactive and active/ancestor
347  // elements?
348  // If not, then we're neighbors.
349  // If so, then the subactive's neighbor is
350 
351  if (element->subactive() ==
352  neighbor->subactive())
353  {
354  // an element is only subactive if it has
355  // been coarsened but not deleted
356  element->set_neighbor (ms,neighbor);
357  neighbor->set_neighbor(ns,element);
358  }
359  else if (element->subactive())
360  {
361  element->set_neighbor(ms,neighbor);
362  }
363  else if (neighbor->subactive())
364  {
365  neighbor->set_neighbor(ns,element);
366  }
367  side_to_elem_map.erase (bounds.first);
368 
369  // get out of this nested crap
370  goto next_side;
371  }
372 
373  ++bounds.first;
374  }
375  }
376 
377  // didn't find a match...
378  // Build the map entry for this element
379  key_val_pair kvp;
380 
381  kvp.first = key;
382  kvp.second.first = element;
383  kvp.second.second = ms;
384 
385  // use the lower bound as a hint for
386  // where to put it.
387 #if defined(LIBMESH_HAVE_UNORDERED_MAP) || defined(LIBMESH_HAVE_TR1_UNORDERED_MAP) || defined(LIBMESH_HAVE_HASH_MAP) || defined(LIBMESH_HAVE_EXT_HASH_MAP)
388  side_to_elem_map.insert (kvp);
389 #else
390  side_to_elem_map.insert (bounds.first,kvp);
391 #endif
392  }
393  }
394  }
395  }
396 
397 #ifdef LIBMESH_ENABLE_AMR
398 
419  const unsigned int n_levels = MeshTools::n_levels(*this);
420  for (unsigned int level = 1; level < n_levels; ++level)
421  {
422  element_iterator end = this->level_elements_end(level);
423  for (element_iterator el = this->level_elements_begin(level);
424  el != end; ++el)
425  {
426  Elem* current_elem = *el;
427  libmesh_assert(current_elem);
428  Elem* parent = current_elem->parent();
429  libmesh_assert(parent);
430  const unsigned int my_child_num = parent->which_child_am_i(current_elem);
431 
432  for (unsigned int s=0; s < current_elem->n_neighbors(); s++)
433  {
434  if (current_elem->neighbor(s) == NULL ||
435  (current_elem->neighbor(s) == remote_elem &&
436  parent->is_child_on_side(my_child_num, s)))
437  {
438  Elem *neigh = parent->neighbor(s);
439 
440  // If neigh was refined and had non-subactive children
441  // made remote earlier, then a non-subactive elem should
442  // actually have one of those remote children as a
443  // neighbor
444  if (neigh && (neigh->ancestor()) && (!current_elem->subactive()))
445  {
446 #ifdef DEBUG
447  // Let's make sure that "had children made remote"
448  // situation is actually the case
449  libmesh_assert(neigh->has_children());
450  bool neigh_has_remote_children = false;
451  for (unsigned int c = 0; c != neigh->n_children(); ++c)
452  {
453  if (neigh->child(c) == remote_elem)
454  neigh_has_remote_children = true;
455  }
456  libmesh_assert(neigh_has_remote_children);
457 
458  // And let's double-check that we don't have
459  // a remote_elem neighboring a local element
460  libmesh_assert_not_equal_to (current_elem->processor_id(),
461  this->processor_id());
462 #endif // DEBUG
463  neigh = const_cast<RemoteElem*>(remote_elem);
464  }
465 
466  current_elem->set_neighbor(s, neigh);
467 #ifdef DEBUG
468  if (neigh != NULL && neigh != remote_elem)
469  // We ignore subactive elements here because
470  // we don't care about neighbors of subactive element.
471  if ((!neigh->active()) && (!current_elem->subactive()))
472  {
473  libMesh::err << "On processor " << this->processor_id()
474  << std::endl;
475  libMesh::err << "Bad element ID = " << current_elem->id()
476  << ", Side " << s << ", Bad neighbor ID = " << neigh->id() << std::endl;
477  libMesh::err << "Bad element proc_ID = " << current_elem->processor_id()
478  << ", Bad neighbor proc_ID = " << neigh->processor_id() << std::endl;
479  libMesh::err << "Bad element size = " << current_elem->hmin()
480  << ", Bad neighbor size = " << neigh->hmin() << std::endl;
481  libMesh::err << "Bad element center = " << current_elem->centroid()
482  << ", Bad neighbor center = " << neigh->centroid() << std::endl;
483  libMesh::err << "ERROR: "
484  << (current_elem->active()?"Active":"Ancestor")
485  << " Element at level "
486  << current_elem->level() << std::endl;
487  libMesh::err << "with "
488  << (parent->active()?"active":
489  (parent->subactive()?"subactive":"ancestor"))
490  << " parent share "
491  << (neigh->subactive()?"subactive":"ancestor")
492  << " neighbor at level " << neigh->level()
493  << std::endl;
494  GMVIO(*this).write ("bad_mesh.gmv");
495  libmesh_error_msg("Problematic mesh written to bad_mesh.gmv.");
496  }
497 #endif // DEBUG
498  }
499  }
500  }
501  }
502 
503 #endif // AMR
504 
505 
506 #ifdef DEBUG
507  MeshTools::libmesh_assert_valid_neighbors(*this);
508 #endif
509 
510  STOP_LOG("find_neighbors()", "Mesh");
511 }

515 void UnstructuredMesh::read (const std::string& name,
516  MeshData* mesh_data,
517  bool skip_renumber_nodes_and_elements)
518 {
519  // See if the file exists. Perform this check on all processors
520  // so that the code is terminated properly in the case that the
521  // file does not exist.
522 
523  // For Nemesis files, the name we try to read will have suffixes
524  // identifying processor rank
525  if (name.rfind(".nem") + 4 == name.size() ||
526  name.rfind(".n") + 2 == name.size())
527  {
528  std::ostringstream full_name;
529 
530  // Find the length of a string which represents the highest processor ID
531  full_name << (this->n_processors());
532  int field_width = cast_int<int>(full_name.str().size());
533 
534  // reset the string stream
535  full_name.str("");
536 
537  // And build up the full filename
538  full_name << name
539  << '.' << this->n_processors()
540  << '.' << std::setfill('0') << std::setw(field_width) << this->processor_id();
541 
542  std::ifstream in (full_name.str().c_str());
543 
544  if (!in.good())
545  libmesh_error_msg("ERROR: cannot locate specified file:\n\t" << full_name.str());
546  }
547  else if(name.rfind(".cp")) {} // Do error checking in the reader
548  else
549  {
550  std::ifstream in (name.c_str());
551 
552  if (!in.good())
553  libmesh_error_msg("ERROR: cannot locate specified file:\n\t" << name);
554  }
555 
556  // Set the skip_renumber_nodes_and_elements flag on all processors.
557  // This ensures that renumber_nodes_and_elements is *not* called
558  // during prepare_for_use() for certain types of mesh files.
559  // This is required in cases where there is an associated solution
560  // file which expects a certain ordering of the nodes.
561  if(name.rfind(".gmv")+4==name.size())
562  {
563  skip_renumber_nodes_and_elements = true;
564  }
565 
566  // Look for parallel formats first
567  if (is_parallel_file_format(name))
568  {
569  // no need to handle bz2 files here -- the Xdr class does that.
570  if ((name.rfind(".xda") < name.size()) ||
571  (name.rfind(".xdr") < name.size()))
572  {
573  XdrIO xdr_io(*this);
574 
575  // .xda* ==> bzip2/gzip/ASCII flavors
576  if (name.rfind(".xda") < name.size())
577  {
578  xdr_io.binary() = false;
579  xdr_io.read (name);
580  }
581  else // .xdr* ==> true binary XDR file
582  {
583  xdr_io.binary() = true;
584  xdr_io.read (name);
585  }
586 
587  // The xdr_io object gets constructed with legacy() == false.
588  // if legacy() == true then it means that a legacy file was detected and
589  // thus processor 0 performed the read. We therefore need to broadcast the
590  // mesh. Further, for this flavor of mesh solution data ordering is tied
591  // to the node ordering, so we better not reorder the nodes!
592  if (xdr_io.legacy())
593  {
594  this->allow_renumbering(false);
595  MeshCommunication().broadcast(*this);
596  }
597 
598  // libHilbert-enabled libMesh builds should construct files
599  // with a canonical node ordering, which libHilbert-enabled
600  // builds will be able to read in again regardless of any
601  // renumbering. So in that case we're free to renumber.
602  // However, if either the writer or the reader of this file
603  // don't have libHilbert, then we'll have to skip
604  // renumbering because we need the numbering to remain
605  // consistent with any solution file we read in next.
606 #ifdef LIBMESH_HAVE_LIBHILBERT
607  // if (!xdr_io.libhilbert_ordering())
608  // skip_renumber_nodes_and_elements = true;
609 #else
610  this->allow_renumbering(false);
611 #endif
612  }
613  else if (name.rfind(".nem") < name.size() ||
614  name.rfind(".n") < name.size())
615  Nemesis_IO(*this).read (name);
616  else if (name.rfind(".cp") < name.size())
617  {
618  if(name.rfind(".cpa") < name.size())
619  CheckpointIO(*this, false).read(name);
620  else
621  CheckpointIO(*this, true).read(name);
622  }
623  }
624 
625  // Serial mesh formats
626  else
627  {
628  START_LOG("read()", "Mesh");
629 
630  // Read the file based on extension. Only processor 0
631  // needs to read the mesh. It will then broadcast it and
632  // the other processors will pick it up
633  if (this->processor_id() == 0)
634  {
635  std::ostringstream pid_suffix;
636  pid_suffix << '_' << getpid();
637  // Nasty hack for reading/writing zipped files
638  std::string new_name = name;
639  if (name.size() - name.rfind(".bz2") == 4)
640  {
641 #ifdef LIBMESH_HAVE_BZIP
642  new_name.erase(new_name.end() - 4, new_name.end());
643  new_name += pid_suffix.str();
644  std::string system_string = "bunzip2 -f -k -c ";
645  system_string += name + " > " + new_name;
646  START_LOG("system(bunzip2)", "Mesh");
647  if (std::system(system_string.c_str()))
648  libmesh_file_error(system_string);
649  STOP_LOG("system(bunzip2)", "Mesh");
650 #else
651  libmesh_error_msg("ERROR: need bzip2/bunzip2 to open .bz2 file " << name);
652 #endif
653  }
654  else if (name.size() - name.rfind(".xz") == 3)
655  {
656 #ifdef LIBMESH_HAVE_XZ
657  new_name.erase(new_name.end() - 3, new_name.end());
658  new_name += pid_suffix.str();
659  std::string system_string = "xz -f -d -k -c ";
660  system_string += name + " > " + new_name;
661  START_LOG("system(xz -d)", "XdrIO");
662  if (std::system(system_string.c_str()))
663  libmesh_file_error(system_string);
664  STOP_LOG("system(xz -d)", "XdrIO");
665 #else

1004 void UnstructuredMesh::create_submesh (UnstructuredMesh& new_mesh,
1005  const_element_iterator& it,
1006  const const_element_iterator& it_end) const
1007 {
1008  // Just in case the subdomain_mesh already has some information
1009  // in it, get rid of it.
1010  new_mesh.clear();
1011 
1012  // Fail if (*this == new_mesh), we cannot create a submesh inside ourself!
1013  // This may happen if the user accidently passes the original mesh into
1014  // this function! We will check this by making sure we did not just
1015  // clear ourself.
1016  libmesh_assert_not_equal_to (this->n_nodes(), 0);
1017  libmesh_assert_not_equal_to (this->n_elem(), 0);
1018 
1019  // How the nodes on this mesh will be renumbered to nodes
1020  // on the new_mesh.
1021  std::vector<dof_id_type> new_node_numbers (this->n_nodes());
1022 
1023  std::fill (new_node_numbers.begin(),
1024  new_node_numbers.end(),
1025  DofObject::invalid_id);
1026 
1027 
1028 
1029  // the number of nodes on the new mesh, will be incremented
1030  dof_id_type n_new_nodes = 0;
1031  dof_id_type n_new_elem = 0;
1032 
1033  for (; it != it_end; ++it)
1034  {
1035  // increment the new element counter
1036  n_new_elem++;
1037 
1038  const Elem* old_elem = *it;
1039 
1040  // Add an equivalent element type to the new_mesh
1041  Elem* new_elem =
1042  new_mesh.add_elem (Elem::build(old_elem->type()).release());
1043 
1044  libmesh_assert(new_elem);
1045 
1046  // Loop over the nodes on this element.
1047  for (unsigned int n=0; n<old_elem->n_nodes(); n++)
1048  {
1049  libmesh_assert_less (old_elem->node(n), new_node_numbers.size());
1050 
1051  if (new_node_numbers[old_elem->node(n)] == DofObject::invalid_id)
1052  {
1053  new_node_numbers[old_elem->node(n)] = n_new_nodes;
1054 
1055  // Add this node to the new mesh
1056  new_mesh.add_point (old_elem->point(n));
1057 
1058  // Increment the new node counter
1059  n_new_nodes++;
1060  }
1061 
1062  // Define this element's connectivity on the new mesh
1063  libmesh_assert_less (new_node_numbers[old_elem->node(n)], new_mesh.n_nodes());
1064 
1065  new_elem->set_node(n) = new_mesh.node_ptr (new_node_numbers[old_elem->node(n)]);
1066  }
1067 
1068  // Copy ids for this element
1069  new_elem->subdomain_id() = old_elem->subdomain_id();
1070  new_elem->processor_id() = old_elem->processor_id();
1071 
1072  // Maybe add boundary conditions for this element
1073  for (unsigned short s=0; s<old_elem->n_sides(); s++)
1074  // We're supporting boundary ids on internal sides now
1075  //if (old_elem->neighbor(s) == NULL)
1076  {
1077  const std::vector<boundary_id_type>& bc_ids =
1078  this->get_boundary_info().boundary_ids(old_elem, s);
1079  for (std::vector<boundary_id_type>::const_iterator id_it=bc_ids.begin(); id_it!=bc_ids.end(); ++id_it)
1080  {
1081  const boundary_id_type bc_id = *id_it;
1082  if (bc_id != this->get_boundary_info().invalid_id)
1083  new_mesh.get_boundary_info().add_side (new_elem, s,
1084  bc_id);
1085  }
1086  }
1087  } // end loop over elements
1088 
1089 
1090  // Prepare the new_mesh for use
1091  new_mesh.prepare_for_use(/*skip_renumber =*/false);
1092 
1093 }
