58 void DifferentiableSystem::clear ()
59 {
60  // If we had an attached Physics object, delete it.
61  if (this->_diff_physics != this)
62  {
63  delete this->_diff_physics;
64  this->_diff_physics = this;
65  }
66  // If we had no attached Physics object, clear our own Physics data
67  else
68  this->clear_physics();
69 
70  // If we had an attached QoI object, delete it.
71  if (this->diff_qoi != this)
72  {
73  delete this->diff_qoi;
74  this->diff_qoi = this;
75  }
76  // If we had no attached QoI object, clear our own QoI data
77  else
78  this->clear_qoi();
79 
80  use_fixed_solution = false;
81 }

97 void DifferentiableSystem::init_data ()
98 {
