63 void DifferentiableSystem::clear ()
64 {
65  // If we had an attached Physics object, delete it.
66  if (this->_diff_physics != this)
67  {
68  delete this->_diff_physics;
69  this->_diff_physics = this;
70  }
71  // If we had no attached Physics object, clear our own Physics data
72  else
73  this->clear_physics();
74 
75  // If we had an attached QoI object, delete it.
76  if (this->diff_qoi != this)
77  {
78  delete this->diff_qoi;
79  this->diff_qoi = this;
80  }
81  // If we had no attached QoI object, clear our own QoI data
82  else
83  this->clear_qoi();
84 
85  use_fixed_solution = false;
86 }

102 void DifferentiableSystem::init_data ()
