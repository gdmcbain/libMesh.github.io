741 template <typename T>
742 inline
743 void EpetraVector<T>::init (const numeric_index_type n,
744  const numeric_index_type n_local,
745  const bool fast,
746  const ParallelType type)
747 {
748  // We default to allocating n_local local storage
749  numeric_index_type my_n_local = n_local;
750 
751  if (type == AUTOMATIC)
752  {
753  if (n == n_local)
754  this->_type = SERIAL;
755  else
756  this->_type = PARALLEL;
757  }
758  else if (type == GHOSTED)
759  {
760  // We don't yet support GHOSTED Epetra vectors, so to get the
761  // same functionality we need a SERIAL vector with local
762  // storage allocated for every entry.
763  this->_type = SERIAL;
764  my_n_local = n;
765  }
766  else
767  this->_type = type;
768 
769  libmesh_assert ((this->_type==SERIAL && n==my_n_local) ||
770  this->_type==PARALLEL);
771 
772  _map = new Epetra_Map(static_cast<int>(n),
773  my_n_local,
774  0,
775  Epetra_MpiComm (this->comm().get()));
776 
777  _vec = new Epetra_Vector(*_map);
778 
779  myFirstID_ = _vec->Map().MinMyGID();
780  myNumIDs_ = _vec->Map().NumMyElements();
781 
782  //Currently we impose the restriction that NumVectors==1, so we won't
783  //need the LDA argument when calling ExtractView. Hence the "dummy" arg.
784  int dummy;
785  _vec->ExtractView(&myCoefs_, &dummy);
786 
787  this->_is_initialized = true;
788  this->_is_closed = true;
789  this->last_edit = 0;
790 
791  if (fast == false)
792  this->zero ();
793 }

821 template <typename T>
822 inline
823 void EpetraVector<T>::close ()
824 {
825  libmesh_assert (this->initialized());
826 
827  // Are we adding or inserting?
828  unsigned char global_last_edit = last_edit;
829  this->comm().max(global_last_edit);
830  libmesh_assert(!last_edit || last_edit == global_last_edit);
831 
832  if (global_last_edit == 1)
833  this->GlobalAssemble(Insert);
834  else if (global_last_edit == 2)
835  this->GlobalAssemble(Add);
836  else
837  libmesh_assert(!global_last_edit);
838 
839  this->_is_closed = true;
840  this->last_edit = 0;
841 }
