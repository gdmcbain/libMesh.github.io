42 struct CompareGlobalIdxMappings
43 {
44  // strict weak ordering for a.first -> a.second mapping. since we can only map to one
45  // value only order the first entry
46  bool operator()(const std::pair<unsigned int, unsigned int> &a,
47  const std::pair<unsigned int, unsigned int> &b) const
48  { return a.first < b.first; }
49 
50  // strict weak ordering for a.first -> a.second mapping. lookups will
51  // be in terms of a single integer, which is why we need this method.
52  bool operator()(const std::pair<unsigned int, unsigned int> &a,
53  const unsigned int b) const
54  { return a.first < b; }
55 };

135 #if defined(LIBMESH_HAVE_EXODUS_API) && defined(LIBMESH_HAVE_NEMESIS_API)
136 void Nemesis_IO::read (const std::string& base_filename)
137 {
138  // On one processor, Nemesis and ExodusII should be equivalent, so
139  // let's cowardly defer to that implementation...
140  if (this->n_processors() == 1)
141  {
142  // We can do this in one line but if the verbose flag was set in this
143  // object, it will no longer be set... thus no extra print-outs for serial runs.
144  // ExodusII_IO(this->mesh()).read (base_filename); // ambiguous when Nemesis_IO is multiply-inherited
145 
146  MeshBase& mesh = MeshInput<MeshBase>::mesh();
147  ExodusII_IO(mesh).read (base_filename);
148  return;
149  }
150 
151  START_LOG ("read()","Nemesis_IO");
152 
153  // This function must be run on all processors at once
154  parallel_object_only();
155 
156  if (_verbose)
157  {
158  libMesh::out << "[" << this->processor_id() << "] ";
159  libMesh::out << "Reading Nemesis file on processor: " << this->processor_id() << std::endl;
160  }
161 
162  // Construct the Nemesis filename based on the number of processors and the
163  // current processor ID.
164  std::string nemesis_filename = nemhelper->construct_nemesis_filename(base_filename);
165 
166  if (_verbose)
167  libMesh::out << "Opening file: " << nemesis_filename << std::endl;
168 
169  // Open the Exodus file in EX_READ mode
170  nemhelper->open(nemesis_filename.c_str(), /*read_only=*/true);
171 
172  // Get a reference to the mesh. We need to be specific
173  // since Nemesis_IO is multiply-inherited
174  // MeshBase& mesh = this->mesh();
175  MeshBase& mesh = MeshInput<MeshBase>::mesh();
176 
177  // We're reading a file on each processor, so our mesh is
178  // partitioned into that many parts as it's created
179  this->set_n_partitions(this->n_processors());
180 
181  // Local information: Read the following information from the standard Exodus header
182  // title[0]
183  // num_dim
184  // num_nodes
185  // num_elem
186  // num_elem_blk
187  // num_node_sets
188  // num_side_sets
189  nemhelper->read_header();
190  nemhelper->print_header();
191 
192  // Get global information: number of nodes, elems, blocks, nodesets and sidesets
193  nemhelper->get_init_global();
194 
195  // Get "load balance" information. This includes the number of internal & border
196  // nodes and elements as well as the number of communication maps.
197  nemhelper->get_loadbal_param();
198 
199  // Do some error checking
200  if (nemhelper->num_external_nodes)
201  libmesh_error_msg("ERROR: there should be no external nodes in an element-based partitioning!");
202 
203  libmesh_assert_equal_to (nemhelper->num_nodes,
204  (nemhelper->num_internal_nodes +
205  nemhelper->num_border_nodes));
206 
207  libmesh_assert_equal_to (nemhelper->num_elem,
208  (nemhelper->num_internal_elems +
209  nemhelper->num_border_elems));
210 
211  libmesh_assert_less_equal (nemhelper->num_nodes, nemhelper->num_nodes_global);
212  libmesh_assert_less_equal (nemhelper->num_elem, nemhelper->num_elems_global);
213 
214  // Read nodes from the exodus file: this fills the nemhelper->x,y,z arrays.
215  nemhelper->read_nodes();
216 
217  // Reads the nemhelper->node_num_map array, node_num_map[i] is the global node number for
218  // local node number i.
219  nemhelper->read_node_num_map();
220 
221  // The get_cmap_params() function reads in the:
222  // node_cmap_ids[],
223  // node_cmap_node_cnts[],
224  // elem_cmap_ids[],
225  // elem_cmap_elem_cnts[],
226  nemhelper->get_cmap_params();
227 
228  // Read the IDs of the interior, boundary, and external nodes. This function
229  // fills the vectors:
230  // node_mapi[],
231  // node_mapb[],
232  // node_mape[]
233  nemhelper->get_node_map();
234 
235  // Read each node communication map for this processor. This function
236  // fills the vectors of vectors named:
237  // node_cmap_node_ids[][]
238  // node_cmap_proc_ids[][]
239  nemhelper->get_node_cmap();
240 
241  libmesh_assert_equal_to (to_uint(nemhelper->num_node_cmaps), nemhelper->node_cmap_node_cnts.size());
242  libmesh_assert_equal_to (to_uint(nemhelper->num_node_cmaps), nemhelper->node_cmap_node_ids.size());
243  libmesh_assert_equal_to (to_uint(nemhelper->num_node_cmaps), nemhelper->node_cmap_proc_ids.size());
244 
245 #ifndef NDEBUG
246  // We expect the communication maps to be symmetric - e.g. if processor i thinks it
247  // communicates with processor j, then processor j should also be expecting to
248  // communicate with i. We can assert that here easily enough with an alltoall,
249  // but let's only do it when not in optimized mode to limit unnecessary communication.
250  {
251  std::vector<unsigned char> pid_send_partner (this->n_processors(), 0);
252 
253  // strictly speaking, we should expect to communicate with ourself...
254  pid_send_partner[this->processor_id()] = 1;
255 
256  // mark each processor id we reference with a node cmap
257  for (unsigned int cmap=0; cmap<to_uint(nemhelper->num_node_cmaps); cmap++)
258  {
259  libmesh_assert_less (to_uint(nemhelper->node_cmap_ids[cmap]), this->n_processors());
260 
261  pid_send_partner[nemhelper->node_cmap_ids[cmap]] = 1;
262  }
263 
264  // Copy the send pairing so we can catch the receive paring and
265  // test for equality
266  const std::vector<unsigned char> pid_recv_partner (pid_send_partner);
267 
268  this->comm().alltoall (pid_send_partner);
269 
270  libmesh_assert (pid_send_partner == pid_recv_partner);
271  }
272 #endif
273 
274  // We now have enough information to infer node ownership. We start by assuming
275  // we own all the nodes on this processor. We will then interrogate the
276  // node cmaps and see if a lower-rank processor is associated with any of
277  // our nodes. If so, then that processor owns the node, not us...
278  std::vector<processor_id_type> node_ownership (nemhelper->num_internal_nodes +
279  nemhelper->num_border_nodes,
280  this->processor_id());
281 
282  // a map from processor id to cmap number, to be used later
283  std::map<unsigned int, unsigned int> pid_to_cmap_map;
284 
285  // For each node_cmap...
286  for (unsigned int cmap=0; cmap<to_uint(nemhelper->num_node_cmaps); cmap++)
287  {
288  // Good time for error checking...
289  libmesh_assert_equal_to (to_uint(nemhelper->node_cmap_node_cnts[cmap]),
290  nemhelper->node_cmap_node_ids[cmap].size());
291 
292  libmesh_assert_equal_to (to_uint(nemhelper->node_cmap_node_cnts[cmap]),
293  nemhelper->node_cmap_proc_ids[cmap].size());
294 
295  // In all the samples I have seen, node_cmap_ids[cmap] is the processor
296  // rank of the remote processor...
297  const int adjcnt_pid_idx = nemhelper->node_cmap_ids[cmap];
298 
299  libmesh_assert_less (adjcnt_pid_idx, this->n_processors());
300  libmesh_assert_not_equal_to (adjcnt_pid_idx, this->processor_id());
301 
302  // We only expect one cmap per adjacent processor
303  libmesh_assert (!pid_to_cmap_map.count(adjcnt_pid_idx));
304 
305  pid_to_cmap_map[adjcnt_pid_idx] = cmap;
306 
307  // ...and each node in that cmap...
308  for (unsigned int idx=0; idx<to_uint(nemhelper->node_cmap_node_cnts[cmap]); idx++)
309  {
310  // Are the node_cmap_ids and node_cmap_proc_ids really redundant?
311  libmesh_assert_equal_to (adjcnt_pid_idx, nemhelper->node_cmap_proc_ids[cmap][idx]);
312 
313  // we are expecting the exodus node numbering to be 1-based...
314  const unsigned int local_node_idx = nemhelper->node_cmap_node_ids[cmap][idx]-1;
315 
316  libmesh_assert_less (local_node_idx, node_ownership.size());
317 
318  // if the adjacent processor is lower rank than the current
319  // owner for this node, then it will get the node...
320  node_ownership[local_node_idx] =
321  std::min(node_ownership[local_node_idx],
322  cast_int<processor_id_type>(adjcnt_pid_idx));
323  }
324  } // We now should have established proper node ownership.
325 
326  // now that ownership is established, we can figure out how many nodes we
327  // will be responsible for numbering.
328  unsigned int num_nodes_i_must_number = 0;
329 
330  for (unsigned int idx=0; idx<node_ownership.size(); idx++)
331  if (node_ownership[idx] == this->processor_id())
332  num_nodes_i_must_number++;
333 
334  // more error checking...
335  libmesh_assert_greater_equal (num_nodes_i_must_number, to_uint(nemhelper->num_internal_nodes));
336  libmesh_assert (num_nodes_i_must_number <= to_uint(nemhelper->num_internal_nodes +
337  nemhelper->num_border_nodes));
338  if (_verbose)
339  libMesh::out << "[" << this->processor_id() << "] "
340  << "num_nodes_i_must_number="
341  << num_nodes_i_must_number
342  << std::endl;
343 
344  // The call to get_loadbal_param() gets 7 pieces of information. We allgather
345  // these now across all processors to determine some global numberings. We should
346  // also gather the number of nodes each processor thinks it will number so that
347  // we can (i) determine our offset, and (ii) do some error checking.
348  std::vector<int> all_loadbal_data ( 8 );
349  all_loadbal_data[0] = nemhelper->num_internal_nodes;
350  all_loadbal_data[1] = nemhelper->num_border_nodes;
351  all_loadbal_data[2] = nemhelper->num_external_nodes;
352  all_loadbal_data[3] = nemhelper->num_internal_elems;
353  all_loadbal_data[4] = nemhelper->num_border_elems;
354  all_loadbal_data[5] = nemhelper->num_node_cmaps;
355  all_loadbal_data[6] = nemhelper->num_elem_cmaps;
356  all_loadbal_data[7] = num_nodes_i_must_number;
357 
358  this->comm().allgather (all_loadbal_data, /* identical_buffer_sizes = */ true);
359 
360  // OK, we are now in a position to request new global indices for all the nodes
361  // we do not own
362 
363  // Let's get a unique message tag to use for send()/receive()
364  Parallel::MessageTag nodes_tag = mesh.comm().get_unique_tag(12345);
365 
366  std::vector<std::vector<int> >
367  needed_node_idxs (nemhelper->num_node_cmaps); // the indices we will ask for
368 
369  std::vector<Parallel::Request>
370  needed_nodes_requests (nemhelper->num_node_cmaps);
371 
372  for (unsigned int cmap=0; cmap<to_uint(nemhelper->num_node_cmaps); cmap++)
373  {
374  // We know we will need no more indices than there are nodes
375  // in this cmap, but that number is an upper bound in general
376  // since the neighboring processor associated with the cmap
377  // may not actually own it
378  needed_node_idxs[cmap].reserve (nemhelper->node_cmap_node_cnts[cmap]);
379 
380  const unsigned int adjcnt_pid_idx = nemhelper->node_cmap_ids[cmap];
381 
382  // ...and each node in that cmap...
383  for (unsigned int idx=0; idx<to_uint(nemhelper->node_cmap_node_cnts[cmap]); idx++)
384  {
385  const unsigned int
386  local_node_idx = nemhelper->node_cmap_node_ids[cmap][idx]-1,
387  owning_pid_idx = node_ownership[local_node_idx];
388 
389  // add it to the request list for its owning processor.
390  if (owning_pid_idx == adjcnt_pid_idx)
391  {
392  const unsigned int
393  global_node_idx = nemhelper->node_num_map[local_node_idx]-1;
394  needed_node_idxs[cmap].push_back(global_node_idx);
395  }
396  }
397  // now post the send for this cmap
398  this->comm().send (adjcnt_pid_idx, // destination
399  needed_node_idxs[cmap], // send buffer
400  needed_nodes_requests[cmap], // request
401  nodes_tag);
402  } // all communication requests for getting updated global indices for border
403  // nodes have been initiated
404 
405  // Figure out how many nodes each processor thinks it will number and make sure
406  // that it adds up to the global number of nodes. Also, set up global node
407  // index offsets for each processor.
408  std::vector<unsigned int>
409  all_num_nodes_i_must_number (this->n_processors());
410 
411  for (unsigned int pid=0; pid<this->n_processors(); pid++)
412  all_num_nodes_i_must_number[pid] = all_loadbal_data[8*pid + 7];
413 
414  // The sum of all the entries in this vector should sum to the number of global nodes
415  libmesh_assert (std::accumulate(all_num_nodes_i_must_number.begin(),
416  all_num_nodes_i_must_number.end(),
417  0) == nemhelper->num_nodes_global);
418 
419  unsigned int my_next_node = 0;
420  for (unsigned int pid=0; pid<this->processor_id(); pid++)
421  my_next_node += all_num_nodes_i_must_number[pid];
422 
423  const unsigned int my_node_offset = my_next_node;
424 
425  if (_verbose)
426  libMesh::out << "[" << this->processor_id() << "] "
427  << "my_node_offset="
428  << my_node_offset
429  << std::endl;
430 
431  // Add internal nodes to the ParallelMesh, using the node ID offset we
432  // computed and the current processor's ID.
433  for (unsigned int i=0; i<to_uint(nemhelper->num_internal_nodes); ++i)
434  {
435  const unsigned int local_node_idx = nemhelper->node_mapi[i]-1;
436 #ifndef NDEBUG
437  const unsigned int owning_pid_idx = node_ownership[local_node_idx];
438 #endif
439 
440  // an internal node we do not own? huh??
441  libmesh_assert_equal_to (owning_pid_idx, this->processor_id());
442  libmesh_assert_less (my_next_node, to_uint(nemhelper->num_nodes_global));
443 
444  // "Catch" the node pointer after addition, make sure the
445  // ID matches the requested value.
446  Node* added_node =
447  mesh.add_point (Point(nemhelper->x[local_node_idx],
448  nemhelper->y[local_node_idx],
449  nemhelper->z[local_node_idx]),
450  my_next_node,
451  this->processor_id());
452 
453  // Make sure the node we added has the ID we thought it would
454  if (added_node->id() != my_next_node)
455  {
456  libMesh::err << "Error, node added with ID " << added_node->id()
457  << ", but we wanted ID " << my_next_node << std::endl;
458  }
459 
460  // update the local->global index map, when we are done
461  // it will be 0-based.
462  nemhelper->node_num_map[local_node_idx] = my_next_node++;
463  }
464 
465  // Now, for the boundary nodes... We may very well own some of them,
466  // but there may be others for which we have requested the new global
467  // id. We expect to be asked for the ids of the ones we own, so
468  // we need to create a map from the old global id to the new one
469  // we are about to create.
470  typedef std::vector<std::pair<unsigned int, unsigned int> > global_idx_mapping_type;
471  global_idx_mapping_type old_global_to_new_global_map;
472  old_global_to_new_global_map.reserve (num_nodes_i_must_number // total # i will have
473  - (my_next_node // amount i have thus far
474  - my_node_offset)); // this should be exact!
475  CompareGlobalIdxMappings global_idx_mapping_comp;
476 
477  for (unsigned int i=0; i<to_uint(nemhelper->num_border_nodes); ++i)
478  {
479  const unsigned int
480  local_node_idx = nemhelper->node_mapb[i]-1,
481  owning_pid_idx = node_ownership[local_node_idx];
482 
483  // if we own it...
484  if (owning_pid_idx == this->processor_id())
485  {
486  const unsigned int
487  global_node_idx = nemhelper->node_num_map[local_node_idx]-1;
488 
489  // we will number it, and create a mapping from its old global index to
490  // the new global index, for lookup purposes when neighbors come calling
491  old_global_to_new_global_map.push_back(std::make_pair(global_node_idx,
492  my_next_node));
493 
494  // "Catch" the node pointer after addition, make sure the
495  // ID matches the requested value.
496  Node* added_node =
497  mesh.add_point (Point(nemhelper->x[local_node_idx],
498  nemhelper->y[local_node_idx],
499  nemhelper->z[local_node_idx]),
500  my_next_node,
501  this->processor_id());
502 
503  // Make sure the node we added has the ID we thought it would
504  if (added_node->id() != my_next_node)
505  {
506  libMesh::err << "Error, node added with ID " << added_node->id()
507  << ", but we wanted ID " << my_next_node << std::endl;
508  }
509 
510  // update the local->global index map, when we are done
511  // it will be 0-based.
512  nemhelper->node_num_map[local_node_idx] = my_next_node++;
513  }
514  }
515  // That should cover numbering all the nodes which belong to us...
516  libmesh_assert_equal_to (num_nodes_i_must_number, (my_next_node - my_node_offset));
517 
518  // Let's sort the mapping so we can efficiently answer requests
519  std::sort (old_global_to_new_global_map.begin(),
520  old_global_to_new_global_map.end(),
521  global_idx_mapping_comp);
522 
523  // and it had better be unique...
524  libmesh_assert (std::unique (old_global_to_new_global_map.begin(),
525  old_global_to_new_global_map.end(),
526  global_idx_mapping_equality) ==
527  old_global_to_new_global_map.end());
528 
529  // We can now catch incoming requests and process them. for efficiency
530  // let's do whatever is available next
531  std::map<unsigned int, std::vector<int> > requested_node_idxs; // the indices asked of us
532 
533  std::vector<Parallel::Request> requested_nodes_requests(nemhelper->num_node_cmaps);
534 
535  // We know we will receive the request from a given processor before
536  // we receive its reply to our request. However, we may receive
537  // a request and a response from one processor before getting
538  // a request from another processor. So what we are doing here
539  // is processing whatever message comes next, while recognizing
540  // we will receive a request from a processor before receiving
541  // its reply
542  std::vector<bool> processed_cmap (nemhelper->num_node_cmaps, false);
543 
544  for (unsigned int comm_step=0; comm_step<2*to_uint(nemhelper->num_node_cmaps); comm_step++)
545  {
546  // query the first message which is available
547  const Parallel::Status
548  status (this->comm().probe (Parallel::any_source,
549  nodes_tag));
550  const unsigned int
551  requesting_pid_idx = status.source(),
552  source_pid_idx = status.source();
553 
554  // this had better be from a processor we are expecting...
555  libmesh_assert (pid_to_cmap_map.count(requesting_pid_idx));
556 
557  // the local cmap which corresponds to the source processor
558  const unsigned int cmap = pid_to_cmap_map[source_pid_idx];
559 
560  if (!processed_cmap[cmap])
561  {
562  processed_cmap[cmap] = true;
563 
564  // we should only get one request per paired processor
565  libmesh_assert (!requested_node_idxs.count(requesting_pid_idx));
566 
567  // get a reference to the request buffer for this processor to
568  // avoid repeated map lookups
