102 bool Prism15::has_affine_map() const
103 {
104  // Make sure z edges are affine
105  Point v = this->point(3) - this->point(0);
106  if (!v.relative_fuzzy_equals(this->point(4) - this->point(1)) ||
107  !v.relative_fuzzy_equals(this->point(5) - this->point(2)))
108  return false;
109  // Make sure edges are straight
110  v /= 2;
111  if (!v.relative_fuzzy_equals(this->point(9) - this->point(0)) ||
112  !v.relative_fuzzy_equals(this->point(10) - this->point(1)) ||
113  !v.relative_fuzzy_equals(this->point(11) - this->point(2)))
114  return false;
115  v = (this->point(1) - this->point(0))/2;
116  if (!v.relative_fuzzy_equals(this->point(6) - this->point(0)) ||
117  !v.relative_fuzzy_equals(this->point(12) - this->point(3)))
118  return false;
119  v = (this->point(2) - this->point(0))/2;
120  if (!v.relative_fuzzy_equals(this->point(8) - this->point(0)) ||
121  !v.relative_fuzzy_equals(this->point(14) - this->point(3)))
122  return false;
123  v = (this->point(2) - this->point(1))/2;
124  if (!v.relative_fuzzy_equals(this->point(7) - this->point(1)) ||
125  !v.relative_fuzzy_equals(this->point(13) - this->point(4)))
126  return false;
127  return true;
128 }

132 AutoPtr<Elem> Prism15::build_side (const unsigned int i,
133  bool proxy) const
134 {
135  libmesh_assert_less (i, this->n_sides());
136 
137  if (proxy)
138  {
139  switch (i)
140  {
141  case 0: // the triangular face at z=-1
142  case 4:
143  {
144  AutoPtr<Elem> face(new Side<Tri6,Prism15>(this,i));
145  return face;
146  }
147 
148  case 1:
149  case 2:
150  case 3:
151  {
152  AutoPtr<Elem> face(new Side<Quad8,Prism15>(this,i));
153  return face;
154  }
155 
156  default:
157  libmesh_error_msg("Invalid side i = " << i);
158  }
159  }
160 
161  else
162  {
163  // Create NULL pointer to be initialized, returned later.
164  AutoPtr<Elem> face(NULL);
165 
166  switch (i)
167  {
168  case 0: // the triangular face at z=-1
169  {
170  face.reset(new Tri6);
171 
172  face->set_node(0) = this->get_node(0);
173  face->set_node(1) = this->get_node(2);
174  face->set_node(2) = this->get_node(1);
175  face->set_node(3) = this->get_node(8);
176  face->set_node(4) = this->get_node(7);
177  face->set_node(5) = this->get_node(6);
178 
179  break;
180  }
181  case 1: // the quad face at y=0
182  {
183  face.reset(new Quad8);
184 
185  face->set_node(0) = this->get_node(0);
186  face->set_node(1) = this->get_node(1);
187  face->set_node(2) = this->get_node(4);
188  face->set_node(3) = this->get_node(3);
189  face->set_node(4) = this->get_node(6);
190  face->set_node(5) = this->get_node(10);
191  face->set_node(6) = this->get_node(12);
192  face->set_node(7) = this->get_node(9);
193 
194  break;
195  }
196  case 2: // the other quad face
197  {
198  face.reset(new Quad8);
199 
200  face->set_node(0) = this->get_node(1);
201  face->set_node(1) = this->get_node(2);
202  face->set_node(2) = this->get_node(5);
203  face->set_node(3) = this->get_node(4);
204  face->set_node(4) = this->get_node(7);
205  face->set_node(5) = this->get_node(11);
206  face->set_node(6) = this->get_node(13);

208 
209  break;
210  }
211  case 3: // the quad face at x=0
212  {
213  face.reset(new Quad8);
214 
215  face->set_node(0) = this->get_node(2);
216  face->set_node(1) = this->get_node(0);
217  face->set_node(2) = this->get_node(3);

242

261 void Prism15::connectivity(const unsigned int libmesh_dbg_var(sc),
262  const IOPackage iop,
263  std::vector<dof_id_type>& conn) const
264 {
265  libmesh_assert(_nodes);
266  libmesh_assert_less (sc, this->n_sub_elem());
267  libmesh_assert_not_equal_to (iop, INVALID_IO_PACKAGE);
268 
269  switch (iop)
270  {
271  case TECPLOT:
272  {
273  conn.resize(8);
274  conn[0] = this->node(0)+1;
275  conn[1] = this->node(1)+1;
276  conn[2] = this->node(2)+1;
277  conn[3] = this->node(2)+1;
278  conn[4] = this->node(3)+1;
279  conn[5] = this->node(4)+1;
280  conn[6] = this->node(5)+1;
281  conn[7] = this->node(5)+1;
282  return;
283  }
284 
285  case VTK:
286  {
287  /*
288  conn.resize(6);
289  conn[0] = this->node(0);
290  conn[1] = this->node(2);
291  conn[2] = this->node(1);
292  conn[3] = this->node(3);
293  conn[4] = this->node(5);
294  conn[5] = this->node(4);
295  */
296 
297  // VTK's VTK_QUADRATIC_WEDGE first 9 nodes match, then their
298  // middle and top layers of mid-edge nodes are reversed from
299  // LibMesh's.
300  conn.resize(15);
301  for (unsigned i=0; i<9; ++i)
302  conn[i] = this->node(i);
303 
304  // top "ring" of mid-edge nodes
305  conn[9] = this->node(12);
306  conn[10] = this->node(13);
307  conn[11] = this->node(14);
308 
309  // middle "ring" of mid-edge nodes
310  conn[12] = this->node(9);
311  conn[13] = this->node(10);
312  conn[14] = this->node(11);
313 
314 
315  return;
316  }
317 
318  default:
319  libmesh_error_msg("Unsupported IO package " << iop);
320  }
321 }
