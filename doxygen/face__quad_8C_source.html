87 bool Quad::is_child_on_side(const unsigned int c,
88  const unsigned int s) const
89 {
90  libmesh_assert_less (c, this->n_children());
91  libmesh_assert_less (s, this->n_sides());
92 
93  // A quad's children and nodes don't share the same ordering:
94  // child 2 and 3 are swapped;
95  unsigned int n = (c < 2) ? c : 5-c;
96  return (n == s || n == (s+1)%4);
97 }

140 Real Quad::quality (const ElemQuality q) const
141 {
142  switch (q)
143  {
144 
149  case DISTORTION:
150  case DIAGONAL:
151  case STRETCH:
152  {
153  // Diagonal between node 0 and node 2
154  const Real d02 = this->length(0,2);
155 
156  // Diagonal between node 1 and node 3
157  const Real d13 = this->length(1,3);
158 
159  // Find the biggest and smallest diagonals
160  if ( (d02 > 0.) && (d13 >0.) )
161  if (d02 < d13) return d02 / d13;
162  else return d13 / d02;
163  else
164  return 0.;
165  break;
166  }
167 
168  default:
169  return Elem::quality(q);
170  }
171 
177  return Elem::quality(q);
178 }
