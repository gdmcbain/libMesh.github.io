56 {
57  bool operator()(const Elem *a,

60  libmesh_assert (a);

62  const unsigned int
63  al = a->level(), bl = b->level();
64  const dof_id_type

80  // Find the neighbor ids for each requested element
81  void gather_data (const std::vector<dof_id_type>& ids,
82  std::vector<datum>& neighbors)
83  {
84  neighbors.resize(ids.size());
85 
86  for (std::size_t i=0; i != ids.size(); ++i)
87  {
88  // Look for this element in the mesh
89  // We'd better find every element we're asked for
90  const Elem *elem = mesh.elem(ids[i]);
91 
92  // Return the element's neighbors
93  const unsigned int n_neigh = elem->n_neighbors();
94  neighbors[i].resize(n_neigh);
95  for (unsigned int n = 0; n != n_neigh; ++n)
96  {
97  const Elem *neigh = elem->neighbor(n);
98  if (neigh)
99  {
100  libmesh_assert_not_equal_to(neigh, remote_elem);
101  neighbors[i][n] = neigh->id();
102  }
103  else
104  neighbors[i][n] = DofObject::invalid_id;
105  }
106  }
107  }

109  void act_on_data (const std::vector<dof_id_type>& ids,
110  std::vector<datum>& neighbors)
111  {
112  for (std::size_t i=0; i != ids.size(); ++i)
113  {
114  Elem *elem = mesh.elem(ids[i]);
115 
116  datum &new_neigh = neighbors[i];
117 
118  const unsigned int n_neigh = elem->n_neighbors();
119  libmesh_assert_equal_to (n_neigh, new_neigh.size());
120 
121  for (unsigned int n = 0; n != n_neigh; ++n)
122  {
123  const dof_id_type new_neigh_id = new_neigh[n];
124  const Elem *old_neigh = elem->neighbor(n);
125  if (old_neigh && old_neigh != remote_elem)
126  {
127  libmesh_assert_equal_to(old_neigh->id(), new_neigh_id);
128  }
129  else if (new_neigh_id == DofObject::invalid_id)
130  {
131  libmesh_assert (!old_neigh);
132  }
133  else
134  {
135  Elem *neigh = mesh.query_elem(new_neigh_id);
136  if (neigh)
137  elem->set_neighbor(n, neigh);
138  else
139  elem->set_neighbor(n, const_cast<RemoteElem*>(remote_elem));
140  }
141  }
142  }
143  }
144 };

171 void MeshCommunication::redistribute (ParallelMesh &mesh) const
172 {
173  // This method will be called after a new partitioning has been
174  // assigned to the elements. This partitioning was defined in
175  // terms of the active elements, and "trickled down" to the
176  // parents and nodes as to be consistent.
177  //
178  // The point is that the entire concept of local elements is
179  // kinda shaky in this method. Elements which were previously
180  // local may now be assigned to other processors, so we need to
181  // send those off. Similarly, we need to accept elements from
182  // other processors.
183  //
184  // The approach is as follows:
185  // (1) send all elements we have stored to their proper homes
186  // (2) receive elements from all processors, watching for duplicates
187  // (3) deleting all nonlocal elements elements
188  // (4) obtaining required ghost elements from neighboring processors
189  libmesh_parallel_only(mesh.comm());
190  libmesh_assert (!mesh.is_serial());
191  libmesh_assert (MeshTools::n_elem(mesh.unpartitioned_elements_begin(),
192  mesh.unpartitioned_elements_end()) == 0);
193 
194  START_LOG("redistribute()","MeshCommunication");
195 
196  // Get a few unique message tags to use in communications; we'll
197  // default to some numbers around pi*1000
198  Parallel::MessageTag
199  nodestag = mesh.comm().get_unique_tag(3141),
200  elemstag = mesh.comm().get_unique_tag(3142);
201 
202  // Figure out how many nodes and elements we have which are assigned to each
203  // processor. send_n_nodes_and_elem_per_proc contains the number of nodes/elements
204  // we will be sending to each processor, recv_n_nodes_and_elem_per_proc contains
205  // the number of nodes/elements we will be receiving from each processor.
206  // Format:
207  // send_n_nodes_and_elem_per_proc[2*pid+0] = number of nodes to send to pid
208  // send_n_nodes_and_elem_per_proc[2*pid+1] = number of elements to send to pid
209  std::vector<dof_id_type> send_n_nodes_and_elem_per_proc(2*mesh.n_processors(), 0);
210 
211  std::vector<Parallel::Request>
212  node_send_requests, element_send_requests;
213 
214  for (processor_id_type pid=0; pid<mesh.n_processors(); pid++)
215  if (pid != mesh.processor_id()) // don't send to ourselves!!
216  {
217  // Build up a list of nodes and elements to send to processor pid.
218  // We will certainly send all the elements assigned to this processor,
219  // but we will also ship off any other elements which touch
220  // their nodes.
221  std::set<const Node*> connected_nodes;
222  {
223  MeshBase::const_element_iterator elem_it = mesh.pid_elements_begin(pid);
224  const MeshBase::const_element_iterator elem_end = mesh.pid_elements_end(pid);
225 
226  for (; elem_it!=elem_end; ++elem_it)
227  {
228  const Elem* elem = *elem_it;
229 
230  for (unsigned int n=0; n<elem->n_nodes(); n++)
231  connected_nodes.insert (elem->get_node(n));
232  }
233  }
234 
235  std::set<const Elem*, CompareElemIdsByLevel> elements_to_send;
236  {
237  MeshBase::const_element_iterator elem_it = mesh.elements_begin();
238  const MeshBase::const_element_iterator elem_end = mesh.elements_end();
239 
240  for (; elem_it!=elem_end; ++elem_it)
241  {
242  const Elem* elem = *elem_it;
243 
244  for (unsigned int n=0; n<elem->n_nodes(); n++)
245  if (connected_nodes.count(elem->get_node(n)))
246  {
247  elements_to_send.insert (elem);
248 
249  // The remote processor needs all its semilocal
250  // elements' ancestors, and it's possible that
251  // they might not all be connected to the remote
252  // processor's nodes.
253  for (const Elem* parent = elem->parent(); parent;
254  parent = parent->parent())
255  elements_to_send.insert(parent);
256  }
257  }
258  }
259 
260  connected_nodes.clear();
261  {
262  std::set<const Elem*, CompareElemIdsByLevel>::iterator
263  elem_it = elements_to_send.begin(),
264  elem_end = elements_to_send.end();
265 
266  for (; elem_it!=elem_end; ++elem_it)
267  {
268  const Elem* elem = *elem_it;
269 
270  for (unsigned int n=0; n<elem->n_nodes(); n++)
271  connected_nodes.insert(elem->get_node(n));
272  }
273  }
274 
275  // the number of nodes we will ship to pid
276  send_n_nodes_and_elem_per_proc[2*pid+0] =
277  cast_int<dof_id_type>(connected_nodes.size());
278 
279  // send any nodes off to the destination processor
280  if (!connected_nodes.empty())
281  {
282  node_send_requests.push_back(Parallel::request());
283 
284  mesh.comm().send_packed_range (pid,
285  &mesh,
286  connected_nodes.begin(),
287  connected_nodes.end(),
288  node_send_requests.back(),
289  nodestag);
290  }
291 
292  // the number of elements we will send to this processor
293  send_n_nodes_and_elem_per_proc[2*pid+1] =
294  cast_int<dof_id_type>(elements_to_send.size());
295 
296  if (!elements_to_send.empty())
297  {
298  // send the elements off to the destination processor
299  element_send_requests.push_back(Parallel::request());
300 
301  mesh.comm().send_packed_range (pid,
302  &mesh,
303  elements_to_send.begin(),
304  elements_to_send.end(),
305  element_send_requests.back(),
306  elemstag);
307  }
308  }
309 
310  std::vector<dof_id_type> recv_n_nodes_and_elem_per_proc(send_n_nodes_and_elem_per_proc);
311 
312  mesh.comm().alltoall (recv_n_nodes_and_elem_per_proc);
313 
314  // In general we will only need to communicate with a subset of the other processors.
315  // I can't immediately think of a case where we will send elements but not nodes, but
316  // these are only bools and we have the information anyway...
317  std::vector<bool>
318  send_node_pair(mesh.n_processors(),false), send_elem_pair(mesh.n_processors(),false),
319  recv_node_pair(mesh.n_processors(),false), recv_elem_pair(mesh.n_processors(),false);
320 
321  unsigned int
322  n_send_node_pairs=0, n_send_elem_pairs=0,
323  n_recv_node_pairs=0, n_recv_elem_pairs=0;
324 
325  for (processor_id_type pid=0; pid<mesh.n_processors(); pid++)
326  {
327  if (send_n_nodes_and_elem_per_proc[2*pid+0]) // we have nodes to send
328  {
329  send_node_pair[pid] = true;
330  n_send_node_pairs++;
331  }
332 
333  if (send_n_nodes_and_elem_per_proc[2*pid+1]) // we have elements to send
334  {
335  send_elem_pair[pid] = true;
336  n_send_elem_pairs++;
337  }
338 
339  if (recv_n_nodes_and_elem_per_proc[2*pid+0]) // we have nodes to receive
340  {
341  recv_node_pair[pid] = true;
342  n_recv_node_pairs++;
343  }
344 
345  if (recv_n_nodes_and_elem_per_proc[2*pid+1]) // we have elements to receive
346  {
347  recv_elem_pair[pid] = true;
348  n_recv_elem_pairs++;
349  }
350  }
351  libmesh_assert_equal_to (n_send_node_pairs, node_send_requests.size());
352  libmesh_assert_equal_to (n_send_elem_pairs, element_send_requests.size());
353 
354  // Receive nodes.
355  // We now know how many processors will be sending us nodes.
356  for (unsigned int node_comm_step=0; node_comm_step<n_recv_node_pairs; node_comm_step++)
357  // but we don't necessarily want to impose an ordering, so
358  // just process whatever message is available next.
359  mesh.comm().receive_packed_range (Parallel::any_source,
360  &mesh,
361  mesh_inserter_iterator<Node>(mesh),
362  nodestag);
363 
364  // Receive elements.
365  // Similarly we know how many processors are sending us elements,
366  // but we don't really care in what order we receive them.
367  for (unsigned int elem_comm_step=0; elem_comm_step<n_recv_elem_pairs; elem_comm_step++)
368  mesh.comm().receive_packed_range (Parallel::any_source,
369  &mesh,
370  mesh_inserter_iterator<Elem>(mesh),
371  elemstag);
372 
373  // Wait for all sends to complete
374  Parallel::wait (node_send_requests);
375  Parallel::wait (element_send_requests);
376 
377  // Check on the redistribution consistency
378 #ifdef DEBUG
379  MeshTools::libmesh_assert_equal_n_systems(mesh);
380 
381  MeshTools::libmesh_assert_valid_refinement_tree(mesh);
382 #endif
383 
384  STOP_LOG("redistribute()","MeshCommunication");
385 }
386 #endif // LIBMESH_HAVE_MPI

399 void MeshCommunication::gather_neighboring_elements (ParallelMesh &mesh) const
400 {
401  // Don't need to do anything if there is
402  // only one processor.
403  if (mesh.n_processors() == 1)
404  return;
405 
406  // This function must be run on all processors at once
407  libmesh_parallel_only(mesh.comm());
408 
409  START_LOG("gather_neighboring_elements()","MeshCommunication");
410 
411  //------------------------------------------------------------------
412  // The purpose of this function is to provide neighbor data structure
413  // consistency for a parallel, distributed mesh. In libMesh we require
414  // that each local element have access to a full set of valid face
415  // neighbors. In some cases this requires us to store "ghost elements" -
416  // elements that belong to other processors but we store to provide
417  // data structure consistency. Also, it is assumed that any element
418  // with a NULL neighbor resides on a physical domain boundary. So,
419  // even our "ghost elements" must have non-NULL neighbors. To handle
420  // this the concept of "RemoteElem" is used - a special construct which
421  // is used to denote that an element has a face neighbor, but we do
422  // not actually store detailed information about that neighbor. This
423  // is required to prevent data structure explosion.
424  //
425  // So when this method is called we should have only local elements.
426  // These local elements will then find neighbors among the local
427  // element set. After this is completed, any element with a NULL
428  // neighbor has either (i) a face on the physical boundary of the mesh,
429  // or (ii) a neighboring element which lives on a remote processor.
430  // To handle case (ii), we communicate the global node indices connected
431  // to all such faces to our neighboring processors. They then send us
432  // all their elements with a NULL neighbor that are connected to any
433  // of the nodes in our list.
434  //------------------------------------------------------------------
435 
436  // Let's begin with finding consistent neighbor data information
437  // for all the elements we currently have. We'll use a clean
438  // slate here - clear any existing information, including RemoteElem's.
439  mesh.find_neighbors (/* reset_remote_elements = */ true,
440  /* reset_current_list = */ true);
441 
442  // Get a unique message tag to use in communications; we'll default
443  // to some numbers around pi*10000
444  Parallel::MessageTag
445  element_neighbors_tag = mesh.comm().get_unique_tag(31416);
446 
447  // Now any element with a NULL neighbor either
448  // (i) lives on the physical domain boundary, or
449  // (ii) lives on an inter-processor boundary.
450  // We will now gather all the elements from adjacent processors
451  // which are of the same state, which should address all the type (ii)
452  // elements.
453 
454  // A list of all the processors which *may* contain neighboring elements.
455  // (for development simplicity, just make this the identity map)
456  std::vector<processor_id_type> adjacent_processors;
457  for (processor_id_type pid=0; pid<mesh.n_processors(); pid++)
458  if (pid != mesh.processor_id())
459  adjacent_processors.push_back (pid);
460 
461 
462  const processor_id_type n_adjacent_processors =
463  cast_int<processor_id_type>(adjacent_processors.size());
464 
465  //-------------------------------------------------------------------------
466  // Let's build a list of all nodes which live on NULL-neighbor sides.
467  // For simplicity, we will use a set to build the list, then transfer
468  // it to a vector for communication.
469  std::vector<dof_id_type> my_interface_node_list;
470  std::vector<const Elem*> my_interface_elements;
471  {
472  std::set<dof_id_type> my_interface_node_set;
473 
474  // since parent nodes are a subset of children nodes, this should be sufficient
475  MeshBase::const_element_iterator it = mesh.active_local_elements_begin();
476  const MeshBase::const_element_iterator it_end = mesh.active_local_elements_end();
477 
478  for (; it != it_end; ++it)
479  {
480  const Elem * const elem = *it;
481  libmesh_assert(elem);
482 
483  if (elem->on_boundary()) // denotes *any* side has a NULL neighbor
484  {
485  my_interface_elements.push_back(elem); // add the element, but only once, even
486  // if there are multiple NULL neighbors
487  for (unsigned int s=0; s<elem->n_sides(); s++)
488  if (elem->neighbor(s) == NULL)
489  {
490  UniquePtr<Elem> side(elem->build_side(s));
491 
492  for (unsigned int n=0; n<side->n_vertices(); n++)
493  my_interface_node_set.insert (side->node(n));
494  }
495  }
496  }
497 
498  my_interface_node_list.reserve (my_interface_node_set.size());
499  my_interface_node_list.insert (my_interface_node_list.end(),
500  my_interface_node_set.begin(),
501  my_interface_node_set.end());
502  }
503 
504  // we will now send my_interface_node_list to all of the adjacent processors.
505  // note that for the time being we will copy the list to a unique buffer for
506  // each processor so that we can use a nonblocking send and not access the
507  // buffer again until the send completes. it is my understanding that the
508  // MPI 2.1 standard seeks to remove this restriction as unnecessary, so in
509  // the future we should change this to send the same buffer to each of the
510  // adjacent processors. - BSK 11/17/2008
511  std::vector<std::vector<dof_id_type> >
512  my_interface_node_xfer_buffers (n_adjacent_processors, my_interface_node_list);
513  std::map<processor_id_type, unsigned char> n_comm_steps;
514 
515  std::vector<Parallel::Request> send_requests (3*n_adjacent_processors);
516  unsigned int current_request = 0;
517 
518  for (unsigned int comm_step=0; comm_step<n_adjacent_processors; comm_step++)
519  {
520  n_comm_steps[adjacent_processors[comm_step]]=1;
521  mesh.comm().send (adjacent_processors[comm_step],
522  my_interface_node_xfer_buffers[comm_step],
523  send_requests[current_request++],
524  element_neighbors_tag);
525  }
526 
527  //-------------------------------------------------------------------------
528  // processor pairings are symmetric - I expect to receive an interface node
529  // list from each processor in adjacent_processors as well!
530  // now we will catch an incoming node list for each of our adjacent processors.
531  //
532  // we are done with the adjacent_processors list - note that it is in general
533  // a superset of the processors we truly share elements with. so let's
534  // clear the superset list, and we will fill it with the true list.
535  adjacent_processors.clear();
536 
537  std::vector<dof_id_type> common_interface_node_list;
538 
539  // we expect two classess of messages -
540  // (1) incoming interface node lists, to which we will reply with our elements
541  // touching nodes in the list, and
542  // (2) replies from the requests we sent off previously.
543  // (2.a) - nodes
544  // (2.b) - elements
545  // so we expect 3 communications from each adjacent processor.
546  // by structuring the communication in this way we hopefully impose no
547  // order on the handling of the arriving messages. in particular, we
548  // should be able to handle the case where we receive a request and
549  // all replies from processor A before even receiving a request from
550  // processor B.
551 
552  for (unsigned int comm_step=0; comm_step<3*n_adjacent_processors; comm_step++)
553  {
554  //------------------------------------------------------------------
555  // catch incoming node list
556  Parallel::Status
557  status(mesh.comm().probe (Parallel::any_source,
558  element_neighbors_tag));
559  const processor_id_type
560  source_pid_idx = cast_int<processor_id_type>(status.source()),
561  dest_pid_idx = source_pid_idx;
562 
563  //------------------------------------------------------------------
564  // first time - incoming request
565  if (n_comm_steps[source_pid_idx] == 1)
566  {
567  n_comm_steps[source_pid_idx]++;
568 
569  mesh.comm().receive (source_pid_idx,
570  common_interface_node_list,
571  element_neighbors_tag);
572  const std::size_t
573  their_interface_node_list_size = common_interface_node_list.size();
574 
575  // we now have the interface node list from processor source_pid_idx.
576  // now we can find all of our elements which touch any of these nodes
577  // and send copies back to this processor. however, we can make our
578  // search more efficient by first excluding all the nodes in
579  // their list which are not also contained in
580  // my_interface_node_list. we can do this in place as a set
581  // intersection.
582  common_interface_node_list.erase
583  (std::set_intersection (my_interface_node_list.begin(),
584  my_interface_node_list.end(),
585  common_interface_node_list.begin(),
586  common_interface_node_list.end(),
587  common_interface_node_list.begin()),
588  common_interface_node_list.end());
589 
590  if (false)
591  libMesh::out << "[" << mesh.processor_id() << "] "
592  << "my_interface_node_list.size()=" << my_interface_node_list.size()
593  << ", [" << source_pid_idx << "] "
594  << "their_interface_node_list.size()=" << their_interface_node_list_size
595  << ", common_interface_node_list.size()=" << common_interface_node_list.size()
596  << std::endl;
597 
598  // Now we need to see which of our elements touch the nodes in the list.
599  // We will certainly send all the active elements which intersect source_pid_idx,
600  // but we will also ship off the other elements in the same family tree
601  // as the active ones for data structure consistency.
602  //
603  // FIXME - shipping full family trees is unnecessary and inefficient.
604  //
605  // We also ship any nodes connected to these elements. Note
606  // some of these nodes and elements may be replicated from
607  // other processors, but that is OK.
608  std::set<const Elem*, CompareElemIdsByLevel> elements_to_send;
609  std::set<const Node*> connected_nodes;
610 
611  // Check for quick return?
612  if (common_interface_node_list.empty())
613  {
614  // let's try to be smart here - if we have no nodes in common,
615  // we cannot share elements. so post the messages expected
616  // from us here and go on about our business.
617  // note that even though these are nonblocking sends
618  // they should complete essentially instantly, because
619  // in all cases the send buffers are empty
620  mesh.comm().send_packed_range (dest_pid_idx,
621  &mesh,
622  connected_nodes.begin(),
623  connected_nodes.end(),
624  send_requests[current_request++],
625  element_neighbors_tag);
626 
627  mesh.comm().send_packed_range (dest_pid_idx,
628  &mesh,
629  elements_to_send.begin(),
630  elements_to_send.end(),
631  send_requests[current_request++],
632  element_neighbors_tag);
633 
634  continue;
635  }
636  // otherwise, this really *is* an adjacent processor.
637  adjacent_processors.push_back(source_pid_idx);
638 
639  std::vector<const Elem*> family_tree;
640 
641  for (dof_id_type e=0, n_shared_nodes=0; e<my_interface_elements.size(); e++, n_shared_nodes=0)
642  {
643  const Elem * elem = my_interface_elements[e];
644 
645  for (unsigned int n=0; n<elem->n_vertices(); n++)
646  if (std::binary_search (common_interface_node_list.begin(),
647  common_interface_node_list.end(),
648  elem->node(n)))
649  {
650  n_shared_nodes++;
651 
652  // TBD - how many nodes do we need to share
653  // before we care? certainly 2, but 1? not
654  // sure, so let's play it safe...
655  if (n_shared_nodes > 0) break;
656  }
657 
658  if (n_shared_nodes) // share at least one node?
659  {
660  elem = elem->top_parent();
661 
662  // avoid a lot of duplicated effort -- if we already have elem
663  // in the set its entire family tree is already in the set.
664  if (!elements_to_send.count(elem))
665  {
666 #ifdef LIBMESH_ENABLE_AMR
667  elem->family_tree(family_tree);
668 #else
669  family_tree.clear();
670  family_tree.push_back(elem);
671 #endif
672  for (unsigned int leaf=0; leaf<family_tree.size(); leaf++)
673  {
674  elem = family_tree[leaf];
675  elements_to_send.insert (elem);
676 
677  for (unsigned int n=0; n<elem->n_nodes(); n++)
678  connected_nodes.insert (elem->get_node(n));
679  }
680  }
681  }
682  }
683 
684  // The elements_to_send and connected_nodes sets now contain all
685  // the elements and nodes we need to send to this processor.
686  // All that remains is to pack up the objects (along with
687  // any boundary conditions) and send the messages off.
688  {
689  libmesh_assert (connected_nodes.empty() || !elements_to_send.empty());
690  libmesh_assert (!connected_nodes.empty() || elements_to_send.empty());
691 
692  // send the nodes off to the destination processor
693  mesh.comm().send_packed_range (dest_pid_idx,
694  &mesh,
695  connected_nodes.begin(),
696  connected_nodes.end(),
697  send_requests[current_request++],
698  element_neighbors_tag);
699 
700  // send the elements off to the destination processor
701  mesh.comm().send_packed_range (dest_pid_idx,
702  &mesh,
703  elements_to_send.begin(),
704  elements_to_send.end(),
705  send_requests[current_request++],
706  element_neighbors_tag);
707  }
708  }
709  //------------------------------------------------------------------
710  // second time - reply of nodes
711  else if (n_comm_steps[source_pid_idx] == 2)
712  {
713  n_comm_steps[source_pid_idx]++;
714 
715  mesh.comm().receive_packed_range (source_pid_idx,
716  &mesh,
717  mesh_inserter_iterator<Node>(mesh),
718  element_neighbors_tag);
719  }
720  //------------------------------------------------------------------
721  // third time - reply of elements
722  else if (n_comm_steps[source_pid_idx] == 3)
723  {
724  n_comm_steps[source_pid_idx]++;
725 
726  mesh.comm().receive_packed_range (source_pid_idx,
727  &mesh,
728  mesh_inserter_iterator<Elem>(mesh),
729  element_neighbors_tag);
730  }
731  //------------------------------------------------------------------
732  // fourth time - shouldn't happen
733  else
734  {
735  libMesh::err << "ERROR: unexpected number of replies: "
736  << n_comm_steps[source_pid_idx]
737  << std::endl;
738  }
739  } // done catching & processing replies associated with tag ~ 100,000pi
740 
741  // allow any pending requests to complete
742  Parallel::wait (send_requests);
743 
744  // We can now find neighbor information for the interfaces between
745  // local elements and ghost elements.
746  mesh.find_neighbors (/* reset_remote_elements = */ true,
747  /* reset_current_list = */ false);
748 
749  // Ghost elements may not have correct remote_elem neighbor links,
750  // and we may not be able to locally infer correct neighbor links to
751  // remote elements. So we synchronize ghost element neighbor links.
752  SyncNeighbors nsync(mesh);
753 
754  Parallel::sync_dofobject_data_by_id
755  (mesh.comm(), mesh.elements_begin(), mesh.elements_end(), nsync);
756 
757  STOP_LOG("gather_neighboring_elements()","MeshCommunication");
758 }

841

844  return;

848 
849  START_LOG("(all)gather()","MeshCommunication");

1096 // ------------------------------------------------------------
1097 void MeshCommunication::delete_remote_elements(ParallelMesh& mesh, const std::set<Elem *> & extra_ghost_elem_ids) const
1098 {
1099  // The mesh should know it's about to be parallelized
1100  libmesh_assert (!mesh.is_serial());
1101 
1102  START_LOG("delete_remote_elements()", "MeshCommunication");
1103 
1104 #ifdef DEBUG
1105  // We expect maximum ids to be in sync so we can use them to size
1106  // vectors
1107  mesh.comm().verify(mesh.max_node_id());
1108  mesh.comm().verify(mesh.max_elem_id());
1109  const dof_id_type par_max_node_id = mesh.parallel_max_node_id();
1110  const dof_id_type par_max_elem_id = mesh.parallel_max_elem_id();
1111  libmesh_assert_equal_to (par_max_node_id, mesh.max_node_id());
1112  libmesh_assert_equal_to (par_max_elem_id, mesh.max_elem_id());
1113 #endif
1114 
1115  // FIXME - should these be "unsorted_set"s? O(N) is O(N)...
1116  std::vector<bool> local_nodes(mesh.max_node_id(), false);
1117  std::vector<bool> semilocal_nodes(mesh.max_node_id(), false);
1118  std::vector<bool> semilocal_elems(mesh.max_elem_id(), false);
1119 
1120  // We don't want to delete any element that shares a node
1121  // with or is an ancestor of a local element.
1122 
1123  // Using the local_nodes vector rather than e.g. point_neighbors()
1124  // gives us correct results even in corner cases, such as where two
1125  // elements meet only at a corner. ;-) Links between boundary and
1126  // interior elements on mixed dimensional meshes also give us
1127  // correct ghosting in this way.
1128 
1129  // We also preserve neighbors of local elements - in most cases this
1130  // is redundant with the node check, but for non-conforming Tet4
1131  // meshes and non-level-one-conforming 2D+3D meshes it is possible
1132  // for an element and its coarse neighbor to not share any vertices.
1133 
1134  MeshBase::const_element_iterator l_elem_it = mesh.local_elements_begin(),
1135  l_end = mesh.local_elements_end();
1136  for (; l_elem_it != l_end; ++l_elem_it)
1137  {
1138  const Elem *elem = *l_elem_it;
1139  for (unsigned int n=0; n != elem->n_nodes(); ++n)
1140  {
1141  dof_id_type nodeid = elem->node(n);
1142  libmesh_assert_less (nodeid, local_nodes.size());
1143  local_nodes[nodeid] = true;
1144  }
1145 
1146  for (unsigned int s=0; s != elem->n_sides(); ++s)
1147  {
1148  const Elem *neighbor = elem->neighbor(s);
1149  if (neighbor)
1150  semilocal_elems[neighbor->id()] = true;
1151  }
1152 
1153  while (elem)
1154  {
1155  dof_id_type elemid = elem->id();
1156  libmesh_assert_less (elemid, semilocal_elems.size());
1157  semilocal_elems[elemid] = true;
1158 
1159  for (unsigned int n=0; n != elem->n_nodes(); ++n)
1160  semilocal_nodes[elem->node(n)] = true;
1161 
1162  const Elem *parent = elem->parent();
1163  // Don't proceed from a boundary mesh to an interior mesh
1164  if (parent && parent->dim() != elem->dim())
1165  break;
1166 
1167  elem = parent;
1168  }
1169  }
1170 
1171  // We don't want to delete any element that shares a node
1172  // with or is an ancestor of an unpartitioned element either.
1173  MeshBase::const_element_iterator
1174  u_elem_it = mesh.unpartitioned_elements_begin(),
1175  u_end = mesh.unpartitioned_elements_end();
1176 
1177  for (; u_elem_it != u_end; ++u_elem_it)
1178  {
1179  const Elem *elem = *u_elem_it;
1180  for (unsigned int n=0; n != elem->n_nodes(); ++n)
1181  local_nodes[elem->node(n)] = true;
1182  while (elem)
1183  {
1184  semilocal_elems[elem->id()] = true;
1185 
1186  for (unsigned int n=0; n != elem->n_nodes(); ++n)
1187  semilocal_nodes[elem->node(n)] = true;
1188 
1189  const Elem *parent = elem->parent();
1190  // Don't proceed from a boundary mesh to an interior mesh
1191  if (parent && parent->dim() != elem->dim())
1192  break;
1193 
1194  elem = parent;
1195  }
1196  }
1197 
1198  // Flag all the elements that share nodes with
1199  // local and unpartitioned elements, along with their ancestors
1200  MeshBase::element_iterator nl_elem_it = mesh.not_local_elements_begin(),
1201  nl_end = mesh.not_local_elements_end();
1202  for (; nl_elem_it != nl_end; ++nl_elem_it)
1203  {
1204  const Elem *elem = *nl_elem_it;
1205  for (unsigned int n=0; n != elem->n_nodes(); ++n)
1206  if (local_nodes[elem->node(n)])
1207  {
1208  while (elem)
1209  {
1210  semilocal_elems[elem->id()] = true;
1211 
1212  for (unsigned int nn=0; nn != elem->n_nodes(); ++nn)
1213  semilocal_nodes[elem->node(nn)] = true;
1214 
1215  const Elem *parent = elem->parent();
1216  // Don't proceed from a boundary mesh to an interior mesh
1217  if (parent && parent->dim() != elem->dim())
1218  break;
1219 
1220  elem = parent;
1221  }
1222  break;
1223  }
1224  }
1225 
1226  // Don't delete elements that we were explicitly told not to
1227  for(std::set<Elem *>::iterator it = extra_ghost_elem_ids.begin();
1228  it != extra_ghost_elem_ids.end();
1229  ++it)
1230  {
1231  const Elem *elem = *it;
1232  semilocal_elems[elem->id()] = true;
1233  for (unsigned int n=0; n != elem->n_nodes(); ++n)
1234  semilocal_nodes[elem->node(n)] = true;
1235  }
1236 
1237  // Delete all the elements we have no reason to save,
1238  // starting with the most refined so that the mesh
1239  // is valid at all intermediate steps
1240  unsigned int n_levels = MeshTools::n_levels(mesh);
1241 
1242  for (int l = n_levels - 1; l >= 0; --l)
1243  {
1244  MeshBase::element_iterator lev_elem_it = mesh.level_elements_begin(l),
1245  lev_end = mesh.level_elements_end(l);
1246  for (; lev_elem_it != lev_end; ++lev_elem_it)
1247  {
1248  Elem *elem = *lev_elem_it;
1249  libmesh_assert (elem);
1250  // Make sure we don't leave any invalid pointers
1251  if (!semilocal_elems[elem->id()])
1252  elem->make_links_to_me_remote();
1253 
1254  // Subactive neighbor pointers aren't preservable here
1255  if (elem->subactive())
1256  for (unsigned int s=0; s != elem->n_sides(); ++s)
1257  elem->set_neighbor(s, NULL);
1258 
1259  // delete_elem doesn't currently invalidate element
1260  // iterators... that had better not change
1261  if (!semilocal_elems[elem->id()])
1262  mesh.delete_elem(elem);
1263  }
1264  }
1265 
1266  // Delete all the nodes we have no reason to save
1267  MeshBase::node_iterator node_it = mesh.nodes_begin(),
1268  node_end = mesh.nodes_end();
1269  for (node_it = mesh.nodes_begin(); node_it != node_end; ++node_it)
1270  {
1271  Node *node = *node_it;
1272  libmesh_assert(node);
1273  if (!semilocal_nodes[node->id()])
1274  mesh.delete_node(node);
1275  }
