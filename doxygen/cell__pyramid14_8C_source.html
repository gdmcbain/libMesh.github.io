122 dof_id_type Pyramid14::key (const unsigned int s) const
123 {
124  libmesh_assert_less (s, this->n_sides());
125 
126  switch (s)
127  {

147 UniquePtr<Elem> Pyramid14::build_side (const unsigned int i, bool proxy) const
148 {
149  libmesh_assert_less (i, this->n_sides());
150 
151  if (proxy)
152  {
153  switch (i)
154  {
155  case 0:
156  case 1:
157  case 2:
158  case 3:
159  return UniquePtr<Elem>(new Side<Tri6,Pyramid14>(this,i));
160 
161  case 4:
162  return UniquePtr<Elem>(new Side<Quad9,Pyramid14>(this,i));
163 
164  default:
165  libmesh_error_msg("Invalid side i = " << i);
166  }
167  }
168 
169  else
170  {
171  // Create NULL pointer to be initialized, returned later.
172  Elem* face = NULL;
173 
174  switch (i)
175  {
176  case 0: // triangular face 1
177  case 1: // triangular face 2
178  case 2: // triangular face 3
179  case 3: // triangular face 4
180  {
181  face = new Tri6;
182  break;
183  }
184  case 4: // the quad face at z=0
185  {
186  face = new Quad9;
187  break;
188  }
189  default:
190  libmesh_error_msg("Invalid side i = " << i);
191  }
192 
193  face->subdomain_id() = this->subdomain_id();
194 
195  // Set the nodes
196  for (unsigned n=0; n<face->n_nodes(); ++n)
197  face->set_node(n) = this->get_node(Pyramid14::side_nodes_map[i][n]);
198 
199  return UniquePtr<Elem>(face);
200  }
201 
202  libmesh_error_msg("We'll never get here!");
203  return UniquePtr<Elem>();
204 }
