43 ErrorVectorReal ErrorVector::minimum() const
44 {
45  START_LOG ("minimum()", "ErrorVector");
46 
47  const dof_id_type n = cast_int<dof_id_type>(this->size());
48  ErrorVectorReal min = std::numeric_limits<ErrorVectorReal>::max();
49 
50  for (dof_id_type i=0; i<n; i++)
51  {
52  // Only positive (or zero) values in the error vector
53  libmesh_assert_greater_equal ((*this)[i], 0.);
54  if (this->is_active_elem(i))
55  min = std::min (min, (*this)[i]);
56  }
57  STOP_LOG ("minimum()", "ErrorVector");
58 
59  // ErrorVectors are for positive values
60  libmesh_assert_greater_equal (min, 0.);
61 
62  return min;
63 }

92 Real ErrorVector::median()
93 {
94  const dof_id_type n = cast_int<dof_id_type>(this->size());
95 
96  if (n == 0)
97  return 0.;
98 
99 
100  // Build a StatisticsVector<ErrorVectorReal> containing
101  // only our active entries and take its mean
102  StatisticsVector<ErrorVectorReal> sv;
103 
104  sv.reserve (n);
105 
106  for (dof_id_type i=0; i<n; i++)
107  if(this->is_active_elem(i))
108  sv.push_back((*this)[i]);
109 
110  return sv.median();
111 }

126 Real ErrorVector::variance(const Real mean_in) const
127 {
128  const dof_id_type n = cast_int<dof_id_type>(this->size());
129 
130  START_LOG ("variance()", "ErrorVector");
131 
132  Real the_variance = 0;
133  dof_id_type nnz = 0;
134 
135  for (dof_id_type i=0; i<n; i++)
136  if (this->is_active_elem(i))
137  {
138  const Real delta = ( static_cast<Real>((*this)[i]) - mean_in );
139  the_variance += (delta * delta - the_variance) / (nnz + 1);
140 
141  nnz++;
142  }
143 
144  STOP_LOG ("variance()", "ErrorVector");
145 
146  return the_variance;
147 }

217 void ErrorVector::plot_error(const std::string& filename,
218  const MeshBase& oldmesh) const
219 {
220  AutoPtr<MeshBase> meshptr = oldmesh.clone();
221  MeshBase &mesh = *meshptr;
222 
223  // The all_first_order routine requires that renumbering be allowed
224  mesh.allow_renumbering(true);
225 
226  mesh.all_first_order();
227  EquationSystems temp_es (mesh);
228  ExplicitSystem& error_system
229  = temp_es.add_system<ExplicitSystem> ("Error");
230  error_system.add_variable("error", CONSTANT, MONOMIAL);
231  temp_es.init();
232 
233  const DofMap& error_dof_map = error_system.get_dof_map();
234 
235  MeshBase::const_element_iterator el =
236  mesh.active_local_elements_begin();
237  const MeshBase::const_element_iterator end_el =
238  mesh.active_local_elements_end();
239  std::vector<dof_id_type> dof_indices;
240 
241  for ( ; el != end_el; ++el)
242  {
243  const Elem* elem = *el;
244 
245  error_dof_map.dof_indices(elem, dof_indices);
246 
247  const dof_id_type elem_id = elem->id();
248 
249  //0 for the monomial basis
250  const dof_id_type solution_index = dof_indices[0];
251 
252  // libMesh::out << "elem_number=" << elem_number << std::endl;
253  libmesh_assert_less (elem_id, (*this).size());
254 
255  // We may have zero error values in special circumstances
256  // libmesh_assert_greater ((*this)[elem_id], 0.);
257  error_system.solution->set(solution_index, (*this)[elem_id]);
258  }
259 
260  // We may have to renumber if the original numbering was not
261  // contiguous. Since this is just a temporary mesh, that's probably
262  // fine.
263  if (mesh.max_elem_id() != mesh.n_elem() ||
264  mesh.max_node_id() != mesh.n_nodes())
265  {
266  mesh.allow_renumbering(true);
267  mesh.renumber_nodes_and_elements();
268  }
269 
270  if (filename.rfind(".gmv") < filename.size())
271  {
272  GMVIO(mesh).write_discontinuous_gmv(filename,
273  temp_es, false);
274  }
275  else if (filename.rfind(".plt") < filename.size())
276  {
277  TecplotIO (mesh).write_equation_systems
278  (filename, temp_es);
279  }
280 #ifdef LIBMESH_HAVE_EXODUS_API
281  else if( (filename.rfind(".exo") < filename.size()) ||
282  (filename.rfind(".e") < filename.size()) )
283  {
284  ExodusII_IO io(mesh);
285  io.write(filename);
286  io.write_element_data(temp_es);
287  }
288 #endif
289  else
290  {
291  libmesh_here();
292  libMesh::err << "Warning: ErrorVector::plot_error currently only"
293  << " supports .gmv and .plt and .exo/.e (if enabled) output;" << std::endl;
294  libMesh::err << "Could not recognize filename: " << filename
295  << std::endl;
296  }
297 }
