774 inline
775 void NumericVector<Complex>::print(std::ostream& os) const
776 {
777  libmesh_assert (this->initialized());
778  os << "Size\tglobal = " << this->size()
779  << "\t\tlocal = " << this->local_size() << std::endl;
780 
781  // std::complex<>::operator<<() is defined, but use this form
782  os << "#\tReal part\t\tImaginary part" << std::endl;
783  for (numeric_index_type i=this->first_local_index(); i<this->last_local_index(); i++)
784  os << i << "\t"
785  << (*this)(i).real() << "\t\t"
