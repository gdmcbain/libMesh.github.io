160 void MeshData::translate (const MeshBase & out_mesh,
161  std::vector<Number> & values,
162  std::vector<std::string> & names) const
163 {
164  libmesh_assert (_active || _compatibility_mode);
165 
166  START_LOG("translate()", "MeshData");
167 
168  const unsigned int n_comp = this->n_val_per_node();
169 
170  // transfer our nodal data to a vector
171  // that may be written concurrently
172  // with the \p out_mesh.
173  {
174  // reserve memory for the nodal data
175  values.reserve(n_comp*out_mesh.n_nodes());
176 
177  // iterate over the mesh's nodes
178  MeshBase::const_node_iterator nodes_it = out_mesh.nodes_begin();
179  const MeshBase::const_node_iterator nodes_end = out_mesh.nodes_end();
180 
181  // Do not use the \p get_data() method, but the operator()
182  // method, since this returns by default a zero value,
183  // when there is no nodal data.
184  for (; nodes_it != nodes_end; ++nodes_it)
185  {
186  const Node * node = *nodes_it;
187 
188  for (unsigned int c= 0; c<n_comp; c++)
189  values.push_back(this->operator()(node, c));
190  }
191  }
192 
193 
194 
195  // Now we have the data, nicely stored in \p values.
196  // It remains to give names to the data, then write to
197  // file.
198  {
199  names.reserve(n_comp);
200 
201  // this naming scheme only works up to n_comp=100
202  // (at least for gmv-accepted variable names)
203  libmesh_assert_less (n_comp, 100);
204 
205  for (unsigned int n=0; n<n_comp; n++)
206  {
207  std::ostringstream name_buf;
208  name_buf << "bc_" << n;
209  names.push_back(name_buf.str());
210  }
211  }
212 
213  STOP_LOG("translate()", "MeshData");
214 }

237 void MeshData::read (const std::string & name)
238 {
239  START_LOG("read()", "MeshData");
240 
241  libmesh_assert (_active || _compatibility_mode);
242 
243  // the id maps have to be closed before reading
244  // (note that in compatibility mode these are also true)
245  libmesh_assert (_elem_id_map_closed && _node_id_map_closed);
246 
247 #ifdef DEBUG
248  if (this->compatibility_mode())
249  libMesh::err << "WARNING: MeshData in compatibility mode, node and element ids" << std::endl
250  << " stored in file may be totally different from libMesh ids!" << std::endl;
251 #endif
252 
253  // Read the file based on extension. We let all processors read the
254  // data because it would be inaccurate to let only one processor
255  // have it and we're too lazy to code up a proper parallel read or
256  // read+broadcast right now.
257 
258  if (name.rfind(".xta") < name.size())
259  this->read_xdr (name, READ);
260 
261  else if (name.rfind(".xtr") < name.size())
262  this->read_xdr (name, DECODE);
263 
264  else if (name.rfind(".unv") < name.size())
265  this->read_unv (name);
266 
267  else if ((name.rfind(".node") < name.size()) ||
268  (name.rfind(".ele") < name.size()))
269  this->read_tetgen (name);
270 
271  else
272  libmesh_error_msg(" ERROR: Unrecognized file extension: " << name \
273  << "\n I understand the following:\n\n" \
274  << " *.xta -- Internal ASCII data format\n" \
275  << " *.xtr -- Internal binary data format\n" \
276  << " *.unv -- I-deas format");
277 
278  STOP_LOG("read()", "MeshData");
279 }

502 void MeshData::insert_node_data (std::map<const Node *,
503  std::vector<Number> > & nd,
504  const bool close_elem_data)
505 {
506  libmesh_assert (this->_active || this->_compatibility_mode);
507  // these are also true in compatibility mode
508  libmesh_assert (this->_node_id_map_closed);
509 
510  if (this->_node_data_closed)
511  libmesh_error_msg("ERROR: Nodal data already closed! Use clear() first!");
512 
513  libmesh_assert (this->_node_data.empty());
514 
515 #ifdef DEBUG
516  std::map<const Node *,
517  std::vector<Number> >::const_iterator nd_pos = nd.begin();
518  std::map<const Node *,
519  std::vector<Number> >::const_iterator nd_end = nd.end();
520 
521  // Compare entity-by-entity that the
522  // sizes of the std::vector's are identical.
523  // For this, simply take the length of the 0th
524  // entry as reference length, and compare this
525  // with the length of the 1st, 2nd...
526  libmesh_assert (nd_pos != nd_end);
527  const std::size_t reference_length = (*nd_pos).second.size();
528 
529  // advance, so that we compare with the 1st
530  ++nd_pos;
531 
532  for (; nd_pos != nd_end; ++nd_pos)
533  if ( (*nd_pos).second.size() != reference_length)
534  libmesh_error_msg("ERROR: Size mismatch.");
535 #endif
536 
537  // copy over
538  _node_data = nd;
539 
540  // we may freely trash the nd
541  nd.clear();
542 
543  // close node data
544  this->_node_data_closed = true;
545 
546  // if user wants to, then close elem data, too
547  if (close_elem_data)
548  {
549  libmesh_assert((this->_elem_id_map_closed));
550  this->_elem_data_closed = true;
551  }
552 }

668 void MeshData::assign (const MeshData & omd)
669 {
670  this->_data_descriptor = omd._data_descriptor;
671  this->_node_id_map_closed = omd._node_id_map_closed;
672  this->_node_data_closed = omd._node_data_closed;
673 
674  // we have to be able to modify our elem id maps
675  libmesh_assert (!this->_elem_id_map_closed);
676 
677  this->_elem_data_closed = omd._elem_data_closed;
678  this->_active = omd._active;
679  this->_compatibility_mode = omd._compatibility_mode;
680 
681  // this is ok because we do not manage the UnvHeader
682  // in terms of memory, but only hold a pointer to it...
683  this->_unv_header = omd._unv_header;
684 
685  // Now copy the foreign id maps -- but only for the
686  // nodes. The nodes of the boundary mesh are actually
687  // nodes of the volume mesh.
688  this->_node_id = omd._node_id;
689  this->_id_node = omd._id_node;
690 
691  // The element vector of the boundary mesh contains elements
692  // that are new, and there _cannot_ be any associated
693  // foreign id in the maps. Therefore, fill the maps with
694  // the libMesh id's. But only when the other MeshData
695  // has element ids.
696  if ((this->_active) && (omd._elem_id.size() != 0))
697  {
698 
699  MeshBase::const_element_iterator elem_it = _mesh.elements_begin();
700  const MeshBase::const_element_iterator elem_end = _mesh.elements_end();
701 
702  for (; elem_it != elem_end; ++elem_it)
703  {
704  const Elem * elem = *elem_it;
705  this->add_foreign_elem_id(elem, elem->id());
706  }
707  }
708 
709  // now we can safely assign omd's value
710  this->_elem_id_map_closed = omd._elem_id_map_closed;
711 
712 
713  // and finally the node- and element-associated data
714  this->_node_data = omd._node_data;
715  this->_elem_data = omd._elem_data;
716 }
