97 void FrequencySystem::clear_all ()
98 {
99  this->clear ();
100 
101  EquationSystems& es =

104  // clear frequencies in the parameters section of the

118 void FrequencySystem::init_data ()
119 {
120  // initialize parent data and additional solution vectors
121  LinearImplicitSystem::init_data();
122 
123  // Log how long initializing the system takes
124  START_LOG("init()", "FrequencySystem");
125 
126  EquationSystems& es =
127  this->get_equation_systems();
128 
129  // make sure we have frequencies to solve for
130  if (!_finished_set_frequencies)
131  {
132  /*
133  * when this system was read from file, check
134  * if this has a "n_frequencies" parameter,
135  * and initialize us with these.
136  */
137  if (es.parameters.have_parameter<unsigned int> ("n_frequencies"))
138  {
139  const unsigned int n_freq =
140  es.parameters.get<unsigned int>("n_frequencies");
141 
142  libmesh_assert_greater (n_freq, 0);
143 
144  _finished_set_frequencies = true;
145 
146  this->set_current_frequency(0);
147  }
148  else
149  libmesh_error_msg("ERROR: Need to set frequencies before calling init().");
150  }
151 
152  _finished_init = true;
153 
154  // Stop logging init()
155  STOP_LOG("init()", "FrequencySystem");

186 void FrequencySystem::set_frequencies_by_steps (const Real base_freq,
187  const Real freq_step,
188  const unsigned int n_freq,
189  const bool allocate_solution_duplicates)
190 {
191  this->_keep_solution_duplicates = allocate_solution_duplicates;
192 
193  // sanity check
194  if (_finished_set_frequencies)
195  libmesh_error_msg("ERROR: frequencies already initialized.");
196 
197  EquationSystems& es =
198  this->get_equation_systems();
199 
200  // store number of frequencies as parameter
201  es.parameters.set<unsigned int>("n_frequencies") = n_freq;
202 
203  for (unsigned int n=0; n<n_freq; n++)
204  {
205  // remember frequencies as parameters
206  es.parameters.set<Real>(this->form_freq_param_name(n)) =
207  base_freq + n * freq_step;
208 
209  // build storage for solution vector, if wanted
210  if (this->_keep_solution_duplicates)
211  this->add_vector(this->form_solu_vec_name(n));
212  }
213 
214  _finished_set_frequencies = true;
215 
216  // set the current frequency
217  this->set_current_frequency(0);
218 }

226 {

318  const unsigned int n_stop)
319 {

321  if (!_finished_assemble)
322  this->assemble ();
323

325  libmesh_assert(solve_system);

329  libmesh_assert_less (n_stop, this->n_frequencies());

333

335  // the user-specified maximum # of linear solver iterations,
336  // the user-specified wave speed
337  const Real tol =
338  es.parameters.get<Real>("linear solver tolerance");
339  const unsigned int maxits =

342 
343  // // return values
