82 template <typename T>
83 void PetscDMNonlinearSolver<T>::init(const char* name)
84 {
85  PetscErrorCode ierr;
86  DM dm;
87  this->PetscNonlinearSolver<T>::init();
88 
89  // Attaching a DM with the function and Jacobian callbacks to SNES.
90  ierr = DMCreateLibMesh(this->comm().get(), this->system(), &dm); LIBMESH_CHKERRABORT(ierr);
91  if (name)
92  ierr = DMSetOptionsPrefix(dm,name); LIBMESH_CHKERRABORT(ierr);
93  ierr = DMSetFromOptions(dm); LIBMESH_CHKERRABORT(ierr);
94  ierr = DMSetUp(dm); LIBMESH_CHKERRABORT(ierr);
95  ierr = SNESSetDM(this->_snes, dm); LIBMESH_CHKERRABORT(ierr);
96  // SNES now owns the reference to dm.
97  ierr = DMDestroy(&dm); LIBMESH_CHKERRABORT(ierr);
98  KSP ksp;
99  ierr = SNESGetKSP (this->_snes, &ksp); LIBMESH_CHKERRABORT(ierr);
100 
101  // Set the tolerances for the iterative solver. Use the user-supplied
102  // tolerance for the relative residual & leave the others at default values
103  ierr = KSPSetTolerances (ksp, this->initial_linear_tolerance, PETSC_DEFAULT,PETSC_DEFAULT, this->max_linear_iterations); LIBMESH_CHKERRABORT(ierr);
104 
105  // Set the tolerances for the non-linear solver.
106  ierr = SNESSetTolerances(this->_snes,
107  this->absolute_residual_tolerance,
108  this->relative_residual_tolerance,
109  this->absolute_step_tolerance,
110  this->max_nonlinear_iterations,
111  this->max_function_evaluations);
112  LIBMESH_CHKERRABORT(ierr);
113 
114  //Pull in command-line options
115  KSPSetFromOptions(ksp);
116  SNESSetFromOptions(this->_snes);
117 }

120 template <typename T>
121 std::pair<unsigned int, Real>
122 PetscDMNonlinearSolver<T>::solve (SparseMatrix<T>& jac_in, // System Jacobian Matrix
123  NumericVector<T>& x_in, // Solution vector
124  NumericVector<T>& r_in, // Residual vector
125  const double, // Stopping tolerance
126  const unsigned int)
127 {
128  START_LOG("solve()", "PetscNonlinearSolver");
129  this->init ();
130 
131  // Make sure the data passed in are really of Petsc types
132  cast_ptr<PetscMatrix<T>*>(&jac_in);
133  cast_ptr<PetscVector<T>*>(&r_in);
134 
135  // Extract solution vector
136  PetscVector<T>* x = cast_ptr<PetscVector<T>*>(&x_in);
137 
138  PetscErrorCode ierr=0;
139  PetscInt n_iterations =0;
140 
141 
142  PetscReal final_residual_norm=0.;
143 
144  if (this->user_presolve)
145  this->user_presolve(this->system());
146 
147  //Set the preconditioning matrix
148  if (this->_preconditioner)
149  this->_preconditioner->set_matrix(jac_in);
150 
151  ierr = SNESSolve (this->_snes, PETSC_NULL, x->vec());
152  LIBMESH_CHKERRABORT(ierr);
153 
154  ierr = SNESGetIterationNumber(this->_snes,&n_iterations);
155  LIBMESH_CHKERRABORT(ierr);
156 
157  ierr = SNESGetLinearSolveIterations(this->_snes, &this->_n_linear_iterations);
158  LIBMESH_CHKERRABORT(ierr);
159 
160 #if PETSC_VERSION_LESS_THAN(3,5,0)
161  // SNESGetFunctionNorm was removed in PETSc 3.5.0
162  ierr = SNESGetFunctionNorm(this->_snes,&final_residual_norm);
163  LIBMESH_CHKERRABORT(ierr);
164 #else
165  {
166  // PB: Not sure where r_in is coming from and it's not used here, so we'll
167  // just get the residual from PETSc
168  Vec r;
169  ierr = SNESGetFunction(this->_snes,&r,NULL,NULL);LIBMESH_CHKERRABORT(ierr);
170  ierr = VecNorm(r,NORM_2,&final_residual_norm);LIBMESH_CHKERRABORT(ierr);
171  }
172 #endif
173 
174  // Get and store the reason for convergence
175  SNESGetConvergedReason(this->_snes, &this->_reason);
176 
177  //Based on Petsc 2.3.3 documentation all diverged reasons are negative
178  this->converged = (this->_reason >= 0);
179 
180  this->clear();
181 
182  STOP_LOG("solve()", "PetscNonlinearSolver");
183 
184  // return the # of its. and the final residual norm.
185  return std::make_pair(n_iterations, final_residual_norm);
186 }
