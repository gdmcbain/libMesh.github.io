36 template <unsigned int N>
37 Tree<N>::Tree (const MeshBase & m,
38  unsigned int target_bin_size,
39  const Trees::BuildType bt) :
40  TreeBase(m),
41  root(m,target_bin_size),
42  build_type(bt)
43 {
44  // Set the root node bounding box equal to the bounding
45  // box for the entire domain.
46  root.set_bounding_box (MeshTools::bounding_box(mesh));
47 
48  if (build_type == Trees::NODES)
49  {
50  // Add all the nodes to the root node. It will
51  // automagically build the tree for us.
52  MeshBase::const_node_iterator it = mesh.nodes_begin();
53  const MeshBase::const_node_iterator end = mesh.nodes_end();
54 
55  for (; it != end; ++it)
56  {
57 #ifndef NDEBUG
58  bool node_was_inserted =
59 #endif
60  root.insert (*it);
61  libmesh_assert(node_was_inserted);
62  }
63 
64  // Now the tree contains the nodes.
65  // However, we want element pointers, so here we
66  // convert between the two.
67  std::vector<std::vector<const Elem *> > nodes_to_elem;
68 
69  MeshTools::build_nodes_to_elem_map (mesh, nodes_to_elem);
70  root.transform_nodes_to_elements (nodes_to_elem);
71  }
72 
73  else if (build_type == Trees::ELEMENTS)
74  {
75  // Add all active elements to the root node. It will
76  // automatically build the tree for us.
77  MeshBase::const_element_iterator it = mesh.active_elements_begin();
78  const MeshBase::const_element_iterator end = mesh.active_elements_end();
79 
80  for (; it != end; ++it)
81  {
82 #ifndef NDEBUG
83  bool elem_was_inserted =
84 #endif
85  root.insert (*it);
86  libmesh_assert(elem_was_inserted);
87  }
88  }
89 
90  else if (build_type == Trees::LOCAL_ELEMENTS)
91  {
92  // Add all active, local elements to the root node. It will
93  // automatically build the tree for us.
94  MeshBase::const_element_iterator it = mesh.active_local_elements_begin();
95  const MeshBase::const_element_iterator end = mesh.active_local_elements_end();
96 
97  for (; it != end; ++it)
98  {
99 #ifndef NDEBUG
100  bool elem_was_inserted =
101 #endif
102  root.insert (*it);
103  libmesh_assert(elem_was_inserted);
104  }
105  }
106 
107  else
108  libmesh_error_msg("Unknown build_type = " << build_type);
109 }
